123  @Override
124  public String getShortForm(OWLEntity entity) {
125  int lastURIMatchIndex = Integer.MAX_VALUE;
126  int lastLangMatchIndex = Integer.MAX_VALUE;
127  if (!(entity instanceof OWLIndividual)) {
128  return alternateShortFormProvider.getShortForm(entity);
129  }
130  OWLIndividual individual = (OWLIndividual) entity;
131  // The candidate value to be rendered, we select this based on
132  // ranking of annotation URI and ranking of lang (if present)
133  OWLObject candidateValue = null;
134  for (OWLOntology ontology : ontologySetProvider.getOntologies()) {
135 
136  for (OWLObjectPropertyAssertionAxiom ax : ontology.getObjectPropertyAssertionAxioms(individual)) {
137  int index = properties.indexOf(ax.getProperty());
138  if (index == -1) {
139  continue;
140  }
141  if (index < lastURIMatchIndex) {
142  candidateValue = ax.getObject();
143  }
144  }
145  for (OWLDataPropertyAssertionAxiom ax : ontology.getDataPropertyAssertionAxioms(individual)) {
146  int index = properties.indexOf(ax.getProperty());
147  if (index == -1) {
148  continue;
149  }
150  if (index == lastURIMatchIndex) {
151  // Different property value but same prop, as previous candidate - look at lang tag for that URI
152  // and see if we take priority over the previous one
153  OWLObject obj = ax.getObject();
154  if (obj instanceof OWLLiteral) {
155  List<String> langList = preferredLanguageMap.get(ax.getProperty());
156  if (langList != null) {
157  // There is no need to check if lang is null. It may well be that no
158  // lang is preferred over any other lang.
159  OWLLiteral lit = (OWLLiteral) obj;
160  int langIndex = langList.indexOf(lit.getLang());
161  if (langIndex != -1 && langIndex < lastLangMatchIndex) {
162  lastLangMatchIndex = langIndex;
163  candidateValue = ax.getObject();
164  }
165  }
166 
167  }
168  } else if (index < lastURIMatchIndex) {
169  // Better match than previous URI - wipe out previous match!
170  lastURIMatchIndex = index;
171  candidateValue = ax.getObject();
172  }
173  }
174 
175 
176  }
177 
178  if (candidateValue != null) {
179  return getRendering(candidateValue);
180  } else {
181  return alternateShortFormProvider.getShortForm(entity);
182  }
183  }

196  private String getRendering(OWLObject object) {
197  // We return the literal value of constants or use the alternate
198  // short form provider to render individuals.
199  if (object instanceof OWLLiteral) {
200  return ((OWLLiteral) object).getLiteral();
201  } else {
202  return alternateShortFormProvider.getShortForm((OWLEntity) object);
203  }
204  }
