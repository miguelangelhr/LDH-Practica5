200  public void setPrefixesFromOntologyFormat(OWLOntology ontology, OWLOntologyManager manager, boolean processImportedOntologies) {
201  resetShortFormProvider();
202  if (processImportedOntologies) {
203  for (OWLOntology importedOntology : manager.getImportsClosure(ontology)) {
204  if (!importedOntology.equals(ontology)) {
205  copyPrefixes(manager.getOntologyFormat(importedOntology));
206  }
207  }
208  }
209  OWLOntologyFormat format = manager.getOntologyFormat(ontology);
210  copyPrefixes(format);
211  }

213  private void copyPrefixes(OWLOntologyFormat ontologyFormat) {
214  if (!(ontologyFormat instanceof PrefixOWLOntologyFormat)) {
215  return;
216  }
217  PrefixOWLOntologyFormat prefixFormat = (PrefixOWLOntologyFormat) ontologyFormat;
218  for (String prefixName : prefixFormat.getPrefixName2PrefixMap().keySet()) {
219  String prefix = prefixFormat.getPrefixName2PrefixMap().get(prefixName);
220  setPrefix(prefixName, prefix);
221  }
222  }

229  public void setPrefix(String prefixName, String prefix) {
230  if (!isUsingDefaultShortFormProvider()) {
231  resetShortFormProvider();
232  }
233  ((DefaultPrefixManager) shortFormProvider).setPrefix(prefixName, prefix);
234  }
