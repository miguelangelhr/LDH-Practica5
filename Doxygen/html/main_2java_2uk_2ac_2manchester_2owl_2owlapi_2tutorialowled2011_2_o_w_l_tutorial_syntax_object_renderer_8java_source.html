225  private void writeTable(Collection<? extends OWLObject> objects) {
226  writeTableStart();
227  int count = 0;
228  for (Iterator<? extends OWLObject> it = objects.iterator(); it.hasNext();) {
229  if (count % TABLE_COLUMNS == 0) {
230  if (count > 0) {
231  writeTableRowEnd();
232  }
233  writeTableRowStart();
234  }
235  writeTableCellStart();
236  it.next().accept(this);
237  writeTableCellEnd();
238  count++;
239  }
240  writeTableRowEnd();
241  writeTableEnd();
242  }

244  private void writeList(Collection<? extends OWLObject> objects) {
245  writeListStart();
246  for (Iterator<? extends OWLObject> it = objects.iterator(); it.hasNext();) {
247  writeListItemStart();
248  it.next().accept(this);
249  writeListItemEnd();
250  }
251  writeListEnd();
252  }
