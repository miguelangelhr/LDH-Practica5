372  private void renderUntypedIRIAnnotationAssertions() throws IOException {
373  Set<IRI> annotatedIRIs = new HashSet<IRI>();
374  for (OWLAnnotationAssertionAxiom ax : ontology.getAxioms(AxiomType.ANNOTATION_ASSERTION)) {
375  OWLAnnotationSubject subject = ax.getSubject();
376  if (subject instanceof IRI) {
377  IRI iri = (IRI) subject;
378  if (!ontology.containsEntityInSignature(iri)) {
379  annotatedIRIs.add(iri);
380  }
381  }
382  }
383  if (!annotatedIRIs.isEmpty()) {
384  writeBanner(ANNOTATED_IRIS_BANNER_TEXT);
385  for (IRI iri : annotatedIRIs) {
386  beginObject();
387  createGraph(ontology.getAnnotationAssertionAxioms(iri));
388  render(new RDFResourceNode(iri));
389  renderAnonRoots();
390  endObject();
391  }
392  }

395  private void renderAnonymousIndividuals() throws IOException {
396  for (OWLAnonymousIndividual anonInd : ontology.getReferencedAnonymousIndividuals()) {
397  boolean anonRoot = true;
398  Set<OWLAxiom> axioms = new HashSet<OWLAxiom>();
399  for (OWLAxiom ax : ontology.getReferencingAxioms(anonInd)) {
400  if (!(ax instanceof OWLDifferentIndividualsAxiom)) {
401  AxiomSubjectProvider subjectProvider = new AxiomSubjectProvider();
402  OWLObject obj = subjectProvider.getSubject(ax);
403  if (!obj.equals(anonInd)) {
404  anonRoot = false;
405  break;
406  }
407  else {
408  axioms.add(ax);
409  }
410  }
411  }
412  if (anonRoot) {
413  createGraph(axioms);
414  renderAnonRoots();
415  }
416  }

456  private Set<OWLAxiom> getGeneralAxioms() {
457  Set<OWLAxiom> generalAxioms = new HashSet<OWLAxiom>();
458  generalAxioms.addAll(ontology.getGeneralClassAxioms());
459  generalAxioms.addAll(ontology.getAxioms(AxiomType.DIFFERENT_INDIVIDUALS));
460  for (OWLDisjointClassesAxiom ax : ontology.getAxioms(AxiomType.DISJOINT_CLASSES)) {
461  if (ax.getClassExpressions().size() > 2) {
462  generalAxioms.add(ax);
463  }
464  }
465  for (OWLDisjointObjectPropertiesAxiom ax : ontology.getAxioms(AxiomType.DISJOINT_OBJECT_PROPERTIES)) {
466  if (ax.getProperties().size() > 2) {
467  generalAxioms.add(ax);
468  }
469  }
470  for (OWLDisjointDataPropertiesAxiom ax : ontology.getAxioms(AxiomType.DISJOINT_DATA_PROPERTIES)) {
471  if (ax.getProperties().size() > 2) {
472  generalAxioms.add(ax);
473  }
474  }
475  for (OWLHasKeyAxiom ax : ontology.getAxioms(AxiomType.HAS_KEY)) {
476  if (ax.getClassExpression().isAnonymous()) {
477  generalAxioms.add(ax);
478  }
479  }
480  return generalAxioms;
481  }

527  private void addAnnotationsToOntologyHeader(RDFResourceNode ontologyHeaderNode) {

542  return RDFTranslator.translateLiteralNode(literal);

553  private boolean createGraph(OWLEntity entity) {
554  final Set<OWLAxiom> axioms = new HashSet<OWLAxiom>();
555  // Don't write out duplicates for punned annotations!
556  if (!isIndividualAndClass(entity)) {
557  axioms.addAll(entity.getAnnotationAssertionAxioms(ontology));
558  }
559  axioms.addAll(ontology.getDeclarationAxioms(entity));
560 
561  entity.accept(new OWLEntityVisitor() {
562  @Override
563  public void visit(OWLClass cls) {
564  for (OWLAxiom ax : ontology.getAxioms(cls)) {
565  if (ax instanceof OWLDisjointClassesAxiom) {
566  OWLDisjointClassesAxiom disjAx = (OWLDisjointClassesAxiom) ax;
567  if (disjAx.getClassExpressions().size() > 2) {
568  continue;
569  }
570  }
571  axioms.add(ax);
572  }
573  for (OWLHasKeyAxiom ax : ontology.getAxioms(AxiomType.HAS_KEY)) {
574  if (ax.getClassExpression().equals(cls)) {
575  axioms.add(ax);
576  }
577  }
578 
579  }
580 
581 
582  @Override
583  public void visit(OWLDatatype datatype) {
584  axioms.addAll(ontology.getDatatypeDefinitions(datatype));
585  createGraph(axioms);
586  }
587 
588 
589  @Override
590  public void visit(OWLNamedIndividual individual) {
591  for (OWLAxiom ax : ontology.getAxioms(individual)) {
592  if (ax instanceof OWLDifferentIndividualsAxiom) {
593  continue;
594  }
595  axioms.add(ax);
596  }
597  }
598 
599 
600  @Override
601  public void visit(OWLDataProperty property) {
602  for (OWLAxiom ax : ontology.getAxioms(property)) {
603  if (ax instanceof OWLDisjointDataPropertiesAxiom) {
604  if (((OWLDisjointDataPropertiesAxiom) ax).getProperties().size() > 2) {
605  continue;
606  }
607  }
608  axioms.add(ax);
609  }
610  }
611 
612 
613  @Override
614  public void visit(OWLObjectProperty property) {
615  for (OWLAxiom ax : ontology.getAxioms(property)) {
616  if (ax instanceof OWLDisjointObjectPropertiesAxiom) {
617  if (((OWLDisjointObjectPropertiesAxiom) ax).getProperties().size() > 2) {
618  continue;
619  }
620  }
621  axioms.add(ax);
622  }
623  for (OWLSubPropertyChainOfAxiom ax : ontology.getAxioms(AxiomType.SUB_PROPERTY_CHAIN_OF)) {
624  if (ax.getSuperProperty().equals(property)) {
625  axioms.add(ax);
626  }
627  }
628  axioms.addAll(ontology.getAxioms(ontology.getOWLOntologyManager()
629  .getOWLDataFactory().getOWLObjectInverseOf(property)));
630  }
631 
632  @Override
633  public void visit(OWLAnnotationProperty property) {
634  axioms.addAll(ontology.getAxioms(property));
635  }
636  });
637 
638  if (axioms.isEmpty() && shouldInsertDeclarations()) {
639  if (RDFOntologyFormat.isMissingType(entity, ontology)) {
640  axioms.add(ontology.getOWLOntologyManager().getOWLDataFactory().getOWLDeclarationAxiom(entity));
641  }
642  }
643  createGraph(axioms);
644 
645  return !axioms.isEmpty();
646  }

691  protected boolean isObjectList(RDFResourceNode node) {
692  for (RDFTriple triple : graph.getSortedTriplesForSubject(node, false)) {
693  if (triple.getProperty().getIRI().equals(RDF_TYPE.getIRI())) {
694  if (!triple.getObject().isAnonymous()) {
695  if (triple.getObject().getIRI().equals(RDF_LIST.getIRI())) {
696  List<RDFNode> items = new ArrayList<RDFNode>();
697  toJavaList(node, items);
698  for (RDFNode n : items) {
699  if (n.isLiteral()) {
700  return false;
701  }
702  }
703  return true;
704  }
705  }
706  }
707  }
708  return false;
709  }

712  protected void toJavaList(RDFNode n, List<RDFNode> list) {
713  RDFNode currentNode = n;
714  while (currentNode != null) {
715  for (RDFTriple triple : graph.getSortedTriplesForSubject(currentNode, false)) {
716  if (triple.getProperty().getIRI().equals(RDF_FIRST.getIRI())) {
717  list.add(triple.getObject());
718  }
719  }
720  for (RDFTriple triple : graph.getSortedTriplesForSubject(currentNode, false)) {
721  if (triple.getProperty().getIRI().equals(RDF_REST.getIRI())) {
722  if (!triple.getObject().isAnonymous()) {
723  if (triple.getObject().getIRI().equals(RDF_NIL.getIRI())) {
724  // End of list
725  currentNode = null;
726  }
727  }
728  else {
729  // Should be another list
730  currentNode = triple.getObject();
731 // toJavaList(triple.getObject(), list);
732  }
733  }
734  }
