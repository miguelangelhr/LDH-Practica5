244  private final void flatten(Set<OWLClassExpression> classExpressions) {
245  if (classExpressions.isEmpty()) {
246  return;
247  }
248  final OWLClassExpression desc = classExpressions.iterator().next();
249  if (classExpressions.size() == 1) {
250  write(desc);
251  return;
252  }
253  classExpressions.remove(desc);
254  writeOpenBracket();
255  write(AND);
256  write(desc);
257  flatten(classExpressions);
258  writeCloseBracket();
259  }

261  @Override
262  public final void visit(OWLOntology onto) {
263  for (final OWLClass eachClass : onto.getClassesInSignature()) {
264  final boolean primitive = !eachClass.isDefined(onto);
265  if (primitive) {
266  writeOpenBracket();
267  write(DEFINE_PRIMITIVE_CONCEPT);
268  write(eachClass);
269  writeSpace();
270  Set<OWLClassExpression> superClasses = eachClass.getSuperClasses(onto);
271  if (superClasses.size() == 1) {
272  write(superClasses.iterator().next());
273  } else {
274  flatten(superClasses);
275  }
276  writeCloseBracket(); //==> end definition of primitive-concept
277  writeln();
278  for (OWLClassExpression classExpression : eachClass.getEquivalentClasses(onto)) {
279  writeOpenBracket();
280  write(eachClass);
281  write(EQUIVALENT);
282  writeSpace();
283  classExpression.accept(this);
284  writeCloseBracket();
285  writeln();
286  }
287  } else {
288  writeOpenBracket();
289  write(DEFINE_CONCEPT);
290  write(eachClass);
291  Set<OWLClassExpression> equivalentClasses = eachClass.getEquivalentClasses(onto);
292  if (equivalentClasses.isEmpty()) {
293  //?
294  writeCloseBracket();
295  writeln();
296  } else if (equivalentClasses.size() == 1) {
297  write(equivalentClasses.iterator().next());
298  writeCloseBracket();
299  writeln();
300  } else {
301  Iterator<OWLClassExpression> iter = equivalentClasses.iterator();
302  write(iter.next());
303  writeCloseBracket();
304  writeln();
305  for (; iter.hasNext();) {
306  writeOpenBracket();
307  write(EQUIVALENT);
308  write(eachClass);
309  writeSpace();
310  iter.next().accept(this);
311  writeCloseBracket();
312  writeln();
313  }
314  }
315 
316  }
317  }
318  for (final OWLClassAxiom axiom : onto.getGeneralClassAxioms()) {
319  axiom.accept(this);
320  }
321 
322  for (final OWLObjectProperty property : onto.getObjectPropertiesInSignature()) {
323  writeOpenBracket();
324  write(DEFINE_PRIMITIVE_ROLE);
325  write(property);
326 
327  if (property.isTransitive(onto)) {
328  writeAttribute(TRANSITIVE_ATTR);
329  writeSpace();
330  write(TRUE);
331  }
332  if (property.isSymmetric(onto)) {
333  writeAttribute(SYMMETRIC_ATTR);
334  writeSpace();
335  write(TRUE);
336  }
337  final Set<OWLClassExpression> domains = property.getDomains(onto);
338  if (!domains.isEmpty()) {
339  writeAttribute(DOMAIN);
340  flatten(domains);
341  }
342  final Set<OWLClassExpression> ranges = property.getDomains(onto);
343  if (!ranges.isEmpty()) {
344  writeAttribute(RANGE_ATTR);
345  flatten(ranges);
346  }
347  final Set<OWLObjectPropertyExpression> superProperties = property.getSuperProperties(onto);
348  if (!superProperties.isEmpty()) {
349  writeAttribute(PARENTS_ATTR);
350  writeOpenBracket();
351  for (OWLObjectPropertyExpression express : superProperties) {
352  write(express);
353  }
354  writeCloseBracket();
355  }
356 
357  writeCloseBracket();
358  }
359  try {
360  writer.flush();
361  } catch (IOException io) {
362  io.printStackTrace();
363  }
364 
365  }
