183  public static OWL2Datatype getDatatype(IRI datatype) {
184  if (!isBuiltIn(datatype)) {
185  throw new OWLRuntimeException(datatype + " is not a built in datatype!");
186  }
187  for (OWL2Datatype v : values()) {
188  if (v.iri.equals(datatype)) {
189  return v;
190  }
191  }
192  throw new OWLRuntimeException(datatype + " is not a built in datatype!");
193  }

369  public String getNormalisedString(String s) {
370  switch (this) {
371  case REPLACE:
372  return s.replaceAll("\\t|\\n|\\r", " ");
373  case COLLAPSE:
374  return REPLACE.getNormalisedString(s).replaceAll("\\s+", " ").trim();
