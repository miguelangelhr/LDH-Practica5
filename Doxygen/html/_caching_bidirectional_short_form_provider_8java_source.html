105  public void add(OWLEntity entity) {
106  String shortForm = generateShortForm(entity);
107  entity2ShortFormMap.put(entity, shortForm);
108  Set<OWLEntity> entities = shortForm2EntityMap.get(shortForm);
109  if (entities == null) {

113  entities.add(entity);

115  }
