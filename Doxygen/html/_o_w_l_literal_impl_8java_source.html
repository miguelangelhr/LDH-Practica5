90  public OWLLiteralImpl(String literal, String lang, OWLDatatype datatype) {
91  super();
92  this.literal = new LiteralWrapper(literal);
93  if (lang == null || lang.length() == 0) {
94  this.lang = "";
95  this.datatype = datatype;
96  } else {
97  if (datatype != null && !datatype.isRDFPlainLiteral()) {
98  // ERROR: attempting to build a literal with a language tag and
99  // type different from plain literal
100  throw new OWLRuntimeException("Error: cannot build a literal with type: "
101  + datatype.getIRI() + " and language: " + lang);
102  }
103  this.lang = lang;
104  this.datatype = new OWLDatatypeImpl(OWL2Datatype.RDF_PLAIN_LITERAL.getIRI());
105  }
106  hashcode = getHashCode();
107  }

285  private static final class LiteralWrapper implements Serializable {
286  private static final long serialVersionUID = 30406L;
287  String l;
288  byte[] bytes;
289 
290  LiteralWrapper(String s) {
291  if (s.length() > COMPRESSION_LIMIT) {
292  try {
293  bytes = compress(s);
294  l = null;
295  } catch (IOException e) {
296  // some problem happened - defaulting to no compression
297  l = s;
298  bytes = null;
299  }
300  } else {
301  bytes = null;
302  l = s;
303  }
304  }
305 
306  String get() {
307  if (l != null) {
308  return l;
309  }
310  try {
311  return decompress(bytes);
312  } catch (IOException e) {
313  // some problem has happened - cannot recover from this
314  e.printStackTrace();
315  return null;
316  }
317  }
318 
319  byte[] compress(String s) throws IOException {
320  ByteArrayOutputStream out = new ByteArrayOutputStream();
321  GZIPOutputStream zipout;
322  zipout = new GZIPOutputStream(out);
323  Writer writer = new OutputStreamWriter(zipout, COMPRESSED_ENCODING);
324  writer.write(s);
325  writer.flush();
326  zipout.finish();
327  zipout.flush();
328  return out.toByteArray();
329  }
330 
331  String decompress(byte[] result) throws IOException {
332  ByteArrayInputStream in = new ByteArrayInputStream(result);
333  GZIPInputStream zipin = new GZIPInputStream(in);
334  Reader reader = new InputStreamReader(zipin, COMPRESSED_ENCODING);
335  StringBuilder b = new StringBuilder();
336  int c = reader.read();
337  while (c > -1) {
338  b.append((char) c);
339  c = reader.read();
340  }
341  return b.toString();
342  }
343 
344  private static final String COMPRESSED_ENCODING = "UTF-16";
345  }
