150  @Override
151  public OWLClassExpression visit(OWLClass desc) {
152  if (negated) {
153  if(desc.isOWLNothing()) {
154  return dataFactory.getOWLThing();
155  }
156  else if(desc.isOWLThing()) {
157  return dataFactory.getOWLNothing();
158  }
159  else {
160  return getNegation(desc);
161  }
162 
163  } else {
164  return desc;
165  }
166  }

169  @Override
170  public OWLClassExpression visit(OWLObjectIntersectionOf desc) {
171  Set<OWLClassExpression> ops = new HashSet<OWLClassExpression>();
172  for (OWLClassExpression op : desc.getOperands()) {
173  ops.add(op.accept(this));
174  }
175  if (negated) {
176  return dataFactory.getOWLObjectUnionOf(ops);
177  } else {
178  return dataFactory.getOWLObjectIntersectionOf(ops);
179  }
180  }

184  public OWLClassExpression visit(OWLObjectUnionOf desc) {
185  Set<OWLClassExpression> ops = new HashSet<OWLClassExpression>();
186  for (OWLClassExpression op : desc.getOperands()) {
187  ops.add(op.accept(this));
188  }
189  if (negated) {
190  // Flip to an intersection
191  return dataFactory.getOWLObjectIntersectionOf(ops);
192  } else {
193  return dataFactory.getOWLObjectUnionOf(ops);
194  }

198  @Override
199  public OWLClassExpression visit(OWLObjectComplementOf desc) {
200  if (negated) {
201  // Cancels out.
202  // Save and then restore.
203  boolean neg = negated;
204  negated = false;
205  OWLClassExpression negDesc = desc.getOperand().accept(this);
206  negated = neg;
207  return negDesc;
208  } else {
209  // Save and then restore
210  boolean neg = negated;
211  negated = true;
212  OWLClassExpression negDesc = desc.getOperand().accept(this);
213  negated = neg;
214  return negDesc;
215  }
216  }

247  @Override
248  public OWLClassExpression visit(OWLObjectMinCardinality desc) {
249  boolean neg = negated;
250  int card = desc.getCardinality();
251  if (negated) {
252  card = desc.getCardinality() - 1;
253  if (card < 0) {
254  card = 0;
255  }
256  }
257  negated = false;
258  OWLClassExpression filler = desc.getFiller().accept(this);
259  OWLClassExpression nnf = null;
260  if (neg) {
261  nnf = dataFactory.getOWLObjectMaxCardinality(card, desc.getProperty(), filler);
262  } else {
263  nnf = dataFactory.getOWLObjectMinCardinality(card, desc.getProperty(), filler);
264  }
265  negated = neg;
266  return nnf;
267  }

418  @Override
419  public OWLDataRange visit(OWLDataOneOf node) {
420  if (node.getValues().size() == 1) {
421  if (negated) {
422  return dataFactory.getOWLDataComplementOf(node);

430  oneOfs.add(dataFactory.getOWLDataOneOf(lit));
431  }
432  return dataFactory.getOWLDataUnionOf(oneOfs).accept(this);
433  }
434 
435  }
