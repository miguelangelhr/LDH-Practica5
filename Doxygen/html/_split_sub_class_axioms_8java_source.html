101  public SplitSubClassAxioms(Set<OWLOntology> ontologies, OWLDataFactory dataFactory) {
102  super(dataFactory);
103  changes = new ArrayList<OWLOntologyChange>();
104  for (OWLOntology ont : ontologies) {
105  for (OWLSubClassOfAxiom ax : ont.getAxioms(AxiomType.SUBCLASS_OF)) {
106  ConjunctSplitter splitter = new ConjunctSplitter();
107  ax.getSuperClass().accept(splitter);
108  if (splitter.result.size() > 1) {
109  changes.add(new RemoveAxiom(ont, ax));
110  for (OWLClassExpression desc : splitter.result) {
111  OWLAxiom replAx = getDataFactory().getOWLSubClassOfAxiom(ax.getSubClass(), desc);
112  changes.add(new AddAxiom(ont, replAx));
113  }
114  }
115  }
116  }
117  }
