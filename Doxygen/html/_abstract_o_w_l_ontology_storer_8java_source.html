83  @Override
84  public final void storeOntology(OWLOntology ontology, IRI documentIRI,
85  OWLOntologyFormat ontologyFormat) throws OWLOntologyStorageException {
86  try {
87  if (!documentIRI.isAbsolute()) {
88  throw new OWLOntologyStorageException("Document IRI must be absolute: "
89  + documentIRI);
90  }
91  File tempFile = null;
92  FileOutputStream tempOutputStream = null;
93  FileInputStream tempInputStream = null;
94  InputStreamReader inputStreamReader = null;
95  OutputStreamWriter outputStreamWriter = null;
96  BufferedReader br = null;
97  BufferedWriter w = null;
98  OutputStream os = null;
99  try {
100  // prepare actual output
101  os = prepareActualOutput(documentIRI);
102  boolean useTemp = Boolean.parseBoolean(ontologyFormat.getParameter(
103  OWLOntologyFormat.USE_INTERMEDIATE_OUTPUT_FILE, Boolean.FALSE)
104  .toString());
105  if (useTemp) {
106  tempFile = File.createTempFile("owlapi", ".owl");
107  tempOutputStream = new FileOutputStream(tempFile);
108  store(ontology, ontologyFormat, tempOutputStream);
109  // Now copy across
110  tempInputStream = new FileInputStream(tempFile);
111  inputStreamReader = new InputStreamReader(tempInputStream, UTF_8);
112  br = new BufferedReader(inputStreamReader);
113  outputStreamWriter = new OutputStreamWriter(os, UTF_8);
114  w = new BufferedWriter(outputStreamWriter);
115  String line = br.readLine();
116  while (line != null) {
117  w.write(line);
118  w.write("\n");
119  line = br.readLine();
120  }
121  } else {
122  store(ontology, ontologyFormat, os);
123  }
124  } finally {
125  if (br != null) {
126  br.close();
127  }
128  if (w != null) {
129  w.close();
130  }
131  if (os != null) {
132  os.close();
133  }
134  if (tempOutputStream != null) {
135  tempOutputStream.close();
136  }
137  if (tempInputStream != null) {
138  tempInputStream.close();
139  }
140  if (inputStreamReader != null) {
141  inputStreamReader.close();
142  }
143  if (outputStreamWriter != null) {
144  outputStreamWriter.close();
145  }
146  if (tempFile != null && !tempFile.delete()) {
147  LOGGER.warning("Temporary file " + tempFile.getAbsolutePath()
148  + " cannot be deleted.");
149  }
150  }
151  } catch (IOException e) {
152  throw new OWLOntologyStorageException(e);
153  }
154  }

156  private OutputStream prepareActualOutput(IRI documentIRI)
157  throws FileNotFoundException, MalformedURLException, IOException {
158  OutputStream os;
159  if (documentIRI.getScheme().equals("file")) {
160  File file = new File(documentIRI.toURI());
161  // Ensure that the necessary directories exist.
162  file.getParentFile().mkdirs();
163  os = new FileOutputStream(file);
164  } else {
165  URL url = documentIRI.toURI().toURL();
166  URLConnection conn = url.openConnection();
167  os = conn.getOutputStream();
168  }
169  return os;
170  }

189  @Override
190  public final void storeOntology(OWLOntology ontology,
191  OWLOntologyDocumentTarget target, OWLOntologyFormat format)
192  throws OWLOntologyStorageException {
193  if (target.isWriterAvailable()) {
194  try {
195  Writer writer = target.getWriter();
196  storeOntology(ontology, writer, format);
197  writer.flush();
198  } catch (IOException e) {
199  throw new OWLOntologyStorageException(e);
200  }
201  } else if (target.isOutputStreamAvailable()) {
202  Writer writer = null;
203  try {
204  writer = new BufferedWriter(new OutputStreamWriter(
205  target.getOutputStream(), UTF_8));
206  storeOntology(ontology, writer, format);
207  writer.flush();
208  } catch (IOException e) {
209  throw new OWLOntologyStorageException(e);
210  }
211  } else if (target.isDocumentIRIAvailable()) {
212  storeOntology(ontology, target.getDocumentIRI(), format);
213  } else {
214  throw new OWLOntologyStorageException(
215  "Neither a Writer, OutputStream or Document IRI could be obtained to store the ontology");
216  }
217  }
