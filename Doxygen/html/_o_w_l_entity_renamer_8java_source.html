94  public List<OWLOntologyChange> changeIRI(IRI uri, IRI newIRI) {
95  Map<IRI, IRI> uriMap = new HashMap<IRI, IRI>();
96  uriMap.put(uri, newIRI);
97  List<OWLOntologyChange> changes = new ArrayList<OWLOntologyChange>();
98  OWLObjectDuplicator dup = new OWLObjectDuplicator(owlOntologyManager.getOWLDataFactory(), uriMap);
99  for (OWLOntology ont : ontologies) {
100  fillListWithTransformChanges(changes, getAxioms(ont, uri), ont, dup);
101  }
102  return changes;
103  }

139  private static Set<OWLAxiom> getAxioms(OWLOntology ont, OWLEntity entity) {
140  Set<OWLAxiom> axioms = ont.getReferencingAxioms(entity);
141  axioms.addAll(ont.getDeclarationAxioms(entity));
142  axioms.addAll(ont.getAnnotationAssertionAxioms(entity.getIRI()));
143  return axioms;
144  }

146  private Set<OWLAxiom> getAxioms(OWLOntology ont, IRI iri) {
147  Set<OWLAxiom> axioms = new HashSet<OWLAxiom>();
148  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLClass(iri)));
149  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLObjectProperty(iri)));
150  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLDataProperty(iri)));
151  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLNamedIndividual(iri)));
152  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLDatatype(iri)));
153  axioms.addAll(ont.getReferencingAxioms(owlOntologyManager.getOWLDataFactory().getOWLAnnotationProperty(iri)));
154  axioms.addAll(ont.getAnnotationAssertionAxioms(iri));
155  return axioms;
156  }

168  private static void fillListWithTransformChanges(List<OWLOntologyChange> changes, Set<OWLAxiom> axioms, OWLOntology ont, OWLObjectDuplicator duplicator) {
169  for (OWLAxiom ax : axioms) {
170  changes.add(new RemoveAxiom(ont, ax));
171  OWLAxiom dupAx = duplicator.duplicateObject(ax);
172  changes.add(new AddAxiom(ont, dupAx));
173  }
174  }
