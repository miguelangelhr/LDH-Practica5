108  private void setupEntities() {
109  List<String> namespaces = new ArrayList<String>(xmlWriterNamespaceManager.getNamespaces());
110  Collections.sort(namespaces, new StringLengthOnlyComparator());
111  entities = new LinkedHashMap<String, String>();
112  for (String curNamespace : namespaces) {
113  String curPrefix = "";
114  if (xmlWriterNamespaceManager.getDefaultNamespace().equals(curNamespace)) {
115  curPrefix = xmlWriterNamespaceManager.getDefaultPrefix();
116  }
117  else {
118  curPrefix = xmlWriterNamespaceManager.getPrefixForNamespace(curNamespace);
119 
120  }
121  if (curPrefix.length() > 0) {
122  entities.put(curNamespace, "&" + curPrefix + ";");
123  }
124  }
125  }

128  protected String swapForEntity(String value) {
129  for (String curEntity : entities.keySet()) {
130  String entityVal = entities.get(curEntity);
131  if (value.length() > curEntity.length()) {
132  String repVal = value.replace(curEntity, entityVal);
133  if (repVal.length() < value.length()) {
134  return repVal;
135  }
136  }
137  }
138  return value;
139  }

167  private boolean isValidQName(String name) {
168  if(name == null) {
169  return false;
170  }
171  int colonIndex = name.indexOf(":");
172  boolean valid = false;
173  if (colonIndex == -1) {
174  valid = OWL2Datatype.XSD_NCNAME.getPattern().matcher(name).matches();
175  }
176  else {
177  valid = OWL2Datatype.XSD_NCNAME.getPattern().matcher(name.substring(0, colonIndex - 1)).matches();
178  if (valid) {
179  valid = OWL2Datatype.XSD_NAME.getPattern().matcher(name.substring(colonIndex + 1)).matches();
180  }
181  }
182  return valid;
183  }

263  private void writeEntities(String rootName) throws IOException {
264  String qName = xmlWriterNamespaceManager.getQName(rootName);
265  if (qName == null) {
266  throw new IOException("Cannot create valid XML: qname for " + rootName
267  + " is null");
268  }
269  writer.write("\n\n<!DOCTYPE " + qName + " [\n");
270  for (String entityVal : entities.keySet()) {
271  String entity = entities.get(entityVal);
272  entity = entity.substring(1, entity.length() - 1);
273  writer.write(" <!ENTITY ");
274  writer.write(entity);
275  writer.write(" \"");
276  entityVal = XMLUtils.escapeXML(entityVal);
277  entityVal = entityVal.replace("%", PERCENT_ENTITY);
278  writer.write(entityVal);
279  writer.write("\" >\n");
280  }
281  writer.write("]>\n\n\n");
282  }

285  @Override
286  public void startDocument(String rootElementName) throws IOException {
287  String encodingString = "";
288  if (encoding.length() > 0) {
289  encodingString = " encoding=\"" + encoding + "\"";
290  }
291  writer.write("<?xml version=\"1.0\"" + encodingString + "?>\n");
292  if (XMLWriterPreferences.getInstance().isUseNamespaceEntities()) {
293  writeEntities(rootElementName);
294  }
295  preambleWritten = true;
296  while (!elementStack.isEmpty()) {
297  elementStack.pop().writeElementStart(true);
298  }
299  writeStartElement(rootElementName);
300  setWrapAttributes(true);
301  writeAttribute("xmlns", xmlWriterNamespaceManager.getDefaultNamespace());
302  if (xmlBase.length() != 0) {
303  writeAttribute("xml:base", xmlBase);
304  }
305  for (String curPrefix : xmlWriterNamespaceManager.getPrefixes()) {
306  if (curPrefix.length() > 0) {
307  writeAttribute("xmlns:" + curPrefix, xmlWriterNamespaceManager.getNamespaceForPrefix(curPrefix));
308  }
309  }
310  }

381  public void writeElementStart(boolean close) throws IOException {
382  if (!startWritten) {
383  startWritten = true;
384  insertIndentation();
385  if (name != null) {
386  writer.write('<');
387  writer.write(name);
388  writeAttributes();
389  if (textContent != null) {
390  boolean wrap = textContent.length() > TEXT_CONTENT_WRAP_LIMIT;
391  if (wrap) {
392  writeNewLine();
393  indentation++;
394  insertIndentation();
395  }
396  writer.write('>');
397  writeTextContent();
398  if (wrap) {
399  indentation--;
400  }
401  }
402  if (close) {
403  if (textContent != null) {
404  writeElementEnd();
405  }
406  else {
407  writer.write("/>");
408  writeNewLine();
409  }
410  }
411  else {
412  if (textContent == null) {
413  writer.write('>');
414  writeNewLine();
415  }
416  }
417  }
418  else {
419  // Name is null so by convension this is a comment
420  if (textContent != null) {
421  writer.write("\n\n\n");
422  StringTokenizer tokenizer = new StringTokenizer(textContent, "\n", true);
423  while (tokenizer.hasMoreTokens()) {
424  String token = tokenizer.nextToken();
425  if (!token.equals("\n")) {
426  insertIndentation();
427  }
428  writer.write(token);
429  }
430  writer.write("\n\n");
431  }
432  }
433  }
434  }

455  private void writeAttribute(String attr, String val) throws IOException {
456  writer.write(attr);
457  writer.write('=');

459  if (XMLWriterPreferences.getInstance().isUseNamespaceEntities()) {
460  writer.write(swapForEntity(XMLUtils.escapeXML(val)));

464  }

469  private void writeAttributes() throws IOException {
470  for (Iterator<String> it = attributes.keySet().iterator(); it.hasNext();) {
471  String attr = it.next();
472  String val = attributes.get(attr);
473  writer.write(' ');
474  writeAttribute(attr, val);
475  if (it.hasNext() && wrapAttributes) {
476  writer.write("\n");
477  indentation++;
478  insertIndentation();
479  indentation--;
480  }
481  }
482  }
