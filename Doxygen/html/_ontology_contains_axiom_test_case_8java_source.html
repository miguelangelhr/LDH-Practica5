136  private void runTestOntologyContainsAxioms1(OWLOntologyFormat format)
137  throws Exception {
138  OWLOntology ont1 = getOWLOntology("testont1A");
139  IRI ont1_iri = ont1.getOntologyID().getOntologyIRI();
140  OWLOntology ont2 = getOWLOntology("testont2A");
141  IRI ont2_iri = ont2.getOntologyID().getOntologyIRI();
142  OWLImportsDeclaration ont2_import = ImportsDeclaration(ont1_iri);
143  getManager().applyChange(new AddImport(ont2, ont2_import));
144  OWLAnnotationProperty annoProp = AnnotationProperty(getIRI("annoProp"));
145  OWLAxiom ax_annoProp_decl = Declaration(annoProp);
146  getManager().addAxiom(ont1, ax_annoProp_decl);
147  OWLAnnotation in_ont1_anno = Annotation(annoProp, ont1_iri);
148  OWLAnnotation in_ont2_anno = Annotation(annoProp, ont2_iri);
149  OWLClass A = Class(getIRI("A"));
150  OWLAxiom ax_A_decl = Declaration(A, Collections.singleton(in_ont1_anno));
151  getManager().addAxiom(ont1, ax_A_decl);
152  OWLClass B = Class(getIRI("B"));
153  OWLAxiom ax_B_decl = Declaration(B, Collections.singleton(in_ont2_anno));
154  getManager().addAxiom(ont2, ax_B_decl);
155  OWLAxiom ax_AsubB = SubClassOf(Class(getIRI("A")), Class(getIRI("B")),
156  Collections.singleton(in_ont2_anno));
157  getManager().addAxiom(ont2, ax_AsubB);
158  // annoProp is in ont1 and in the import closure of ont2
159  assertTrue(ont1.containsAxiom(ax_annoProp_decl, false));
160  assertFalse(ont2.containsAxiom(ax_annoProp_decl, false));
161  assertTrue(ont2.containsAxiom(ax_annoProp_decl, true));
162  // A is in ont1 and in the import closure of ont2
163  assertTrue(ont1.containsAxiom(ax_A_decl, false));
164  assertFalse(ont2.containsAxiom(ax_A_decl, false));
165  assertTrue(ont2.containsAxiom(ax_A_decl, true));
166  // B is in only in ont2
167  assertFalse(ont1.containsAxiom(ax_B_decl, true));
168  assertTrue(ont2.containsAxiom(ax_B_decl, false));
169  assertTrue(ont2.containsAxiom(ax_B_decl, true));
170  // A is a subclass of B is in only in ont2
171  assertFalse(ont1.containsAxiom(ax_AsubB, true));
172  assertTrue(ont2.containsAxiom(ax_AsubB, false));
173  assertTrue(ont2.containsAxiom(ax_AsubB, true));
174  File savedLocation1 = File.createTempFile("testont1A", ".owl");
175  FileOutputStream out1 = new FileOutputStream(savedLocation1);
176  StreamDocumentTarget writer1 = new StreamDocumentTarget(out1);
177  getManager().saveOntology(ont1, format, writer1);
178  File savedLocation2 = File.createTempFile("testont2A.owl", ".owl");
179  FileOutputStream out2 = new FileOutputStream(savedLocation2);
180  StreamDocumentTarget writer2 = new StreamDocumentTarget(out2);
181  getManager().saveOntology(ont2, format, writer2);
182  OWLOntologyManager man = Factory.getManager();
183  OWLOntology ont1L = man.loadOntologyFromOntologyDocument(savedLocation1);
184  OWLOntology ont2L = man.loadOntologyFromOntologyDocument(savedLocation2);
185  // annoProp is in ont1 and in the import closure of ont2
186  assertTrue(ont1L.containsAxiom(ax_annoProp_decl, false));
187  assertFalse(ont2L.containsAxiom(ax_annoProp_decl, false));
188  assertTrue(ont2L.containsAxiom(ax_annoProp_decl, true));
189  // A is in ont1 and in the import closure of ont2
190  assertTrue(ont1L.containsAxiom(ax_A_decl, false));
191  assertFalse(ont2L.containsAxiom(ax_A_decl, false));
192  assertTrue(ont2L.containsAxiom(ax_A_decl, true));
193  // B is in only in ont2
194  assertFalse(ont1L.containsAxiom(ax_B_decl, true));
195  assertTrue(ont2L.containsAxiom(ax_B_decl, false));
196  assertTrue(ont2L.containsAxiom(ax_B_decl, true));
197  // A is a subclass of B is in only in ont2
198  assertFalse(ont1L.containsAxiom(ax_AsubB, true));
199  assertTrue(ont2L.containsAxiom(ax_AsubB, false));
200  assertTrue(ont2L.containsAxiom(ax_AsubB, true));
201  }
