123  private void generateChanges() {
124  // We remove the existing superclasses and then combine these
125  // into an intersection which is made equivalent.
126  changes = new ArrayList<OWLOntologyChange>();
127  Set<OWLClassExpression> descs = new HashSet<OWLClassExpression>();
128  for (OWLOntology ont : ontologies) {
129  for (OWLSubClassOfAxiom ax : ont.getSubClassAxiomsForSubClass(cls)) {
130  changes.add(new RemoveAxiom(ont, ax));
131  descs.add(ax.getSuperClass());
132  }
133  }
134  OWLClassExpression equivalentClass = getDataFactory().getOWLObjectIntersectionOf(descs);
135  Set<OWLClassExpression> equivalentClasses = new HashSet<OWLClassExpression>();
136  equivalentClasses.add(cls);
137  equivalentClasses.add(equivalentClass);
138  changes.add(new AddAxiom(targetOntology, getDataFactory().getOWLEquivalentClassesAxiom(equivalentClasses)));
139  }
140

142  @Override
143  public List<OWLOntologyChange> getChanges() {
144  return changes;
145  }
