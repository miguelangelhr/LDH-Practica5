226  @Override
227  public void visit(OWLEquivalentClassesAxiom axiom) {
228  Iterator<OWLClassExpression> classes = axiom.getClassExpressions().iterator();
229  OWLClassExpression c1 = classes.next();
230  OWLClassExpression c2 = classes.next();
231  if (classes.hasNext()) {
232  logger.warning("EquivalentClassesAxiom with more than two elements not supported!");
233  }
234  // apply simplification for the cases where either concept is
235  // owl:Thing or owlapi:Nothin
236  if (c1.isOWLNothing()) {
237  result = c2;
238  } else if (c2.isOWLNothing()) {
239  result = c1;
240  } else if (c1.isOWLThing()) {
241  result = not(c2);
242  } else if (c2.isOWLThing()) {
243  result = not(c1);
244  } else {
245  result = or(and(c1, not(c2)), and(not(c1), c2));
246  }
247  }
