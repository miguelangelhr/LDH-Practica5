153  private void writeAnnotations(OWLAxiom axiom) {
154  for (OWLAnnotation anno : axiom.getAnnotations()) {
155  anno.accept(this);

159  @Override
160  public void visit(OWLOntology ontology) {
161  for (OWLImportsDeclaration decl : ontology.getImportsDeclarations()) {
162  writer.writeStartElement(IMPORT);
163  writer.writeTextContent(decl.getURI().toString());
164  writer.writeEndElement();
165  }
166  for (OWLAnnotation annotation : ontology.getAnnotations()) {
167  annotation.accept(this);
168  }
169  List<OWLAxiom> axioms = new ArrayList<OWLAxiom>(ontology.getAxioms());
170  Collections.sort(axioms);
171  for (OWLAxiom ax : axioms) {
172  ax.accept(this);
173  }
174  }
