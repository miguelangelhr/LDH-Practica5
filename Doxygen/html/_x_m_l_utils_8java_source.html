140  public static boolean isNCName(CharSequence s) {
141  if (isNullOrEmpty(s)) {
142  return false;
143  }
144  int firstCodePoint = Character.codePointAt(s, 0);
145  if(!isNCNameStartChar(firstCodePoint)) {
146  return false;
147  }
148  for(int i = Character.charCount(firstCodePoint); i < s.length(); ) {
149  int codePoint = Character.codePointAt(s, i);
150  if(!isNCNameChar(codePoint)) {
151  return false;
152  }
153  i += Character.charCount(codePoint);
154  }
155  return true;
156  }

165  public static boolean isQName(CharSequence s) {
166  if (isNullOrEmpty(s)) {
167  return false;
168  }
169  boolean foundColon = false;
170  boolean inNCName = false;
171  for(int i = 0; i < s.length(); ) {
172  int codePoint = Character.codePointAt(s, i);
173  if(codePoint == ':') {
174  if(foundColon) {
175  return false;
176  }
177  foundColon = true;
178  if(!inNCName) {
179  return false;
180  }
181  inNCName = false;
182  }
183  else {
184  if(!inNCName) {
185  if(!isXMLNameStartCharacter(codePoint)) {
186  return false;
187  }
188  inNCName = true;
189  }
190  else {
191  if(!isXMLNameChar(codePoint)) {
192  return false;
193  }
194  }
195  }
196  i += Character.charCount(codePoint);
197  }
198  return true;
199  }

217  public static int getNCNameSuffixIndex(CharSequence s) {
218  // identify bnode labels and do not try to split them
219  if (s.length() > 1 && s.charAt(0) == '_' && s.charAt(1) == ':') {
220  return -1;
221  }
222  int index = -1;
223  for(int i = s.length() - 1; i > -1; i--) {
224  if (!Character.isLowSurrogate(s.charAt(i))) {
225  int codePoint = Character.codePointAt(s, i);
226  if(isNCNameStartChar(codePoint)) {
227  index = i;
228  }
229  if(!isNCNameChar(codePoint)) {
230  break;
231  }
232  }
233  }
234  return index;
235  }

244  public static String getNCNameSuffix(CharSequence s) {
245  if (s.length() > 1 && s.charAt(0) == '_' && s.charAt(1) == ':') {
246  return null;

278  public static String escapeXML(CharSequence s) {
279  // double quote -- quot
280  // ampersand -- amp
281  // less than -- lt
282  // greater than -- gt
283  // apostrophe -- apos
284  StringBuilder sb = new StringBuilder(s.length() * 2);
285  for (int i = 0; i < s.length(); ) {
286  int codePoint = Character.codePointAt(s, i);
287  if (codePoint == '<') {
288  sb.append(LT);
289  }
290  else if (codePoint == '>') {
291  sb.append(GT);
292  }
293  else if (codePoint == '\"') {
294  sb.append(QUOT);
295  }
296  else if (codePoint == '&') {
297  sb.append(AMP);
298  }
299  else if (codePoint == '\'') {
300  sb.append(APOS);
301  }
302  else {
303  sb.appendCodePoint(codePoint);
304  }
305  i += Character.charCount(codePoint);
306  }
307  return sb.toString();
308  }
