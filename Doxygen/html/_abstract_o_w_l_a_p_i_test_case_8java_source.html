75  public boolean equal(OWLOntology ont1, OWLOntology ont2) {
76  if (!ont1.isAnonymous() && !ont2.isAnonymous()) {
77  assertEquals("Ontologies supposed to be the same", ont1.getOntologyID(),
78  ont2.getOntologyID());
79  }
80  assertEquals("Annotations supposed to be the same", ont1.getAnnotations(),
81  ont2.getAnnotations());
82  Set<OWLAxiom> axioms1 = ont1.getAxioms();
83  Set<OWLAxiom> axioms2 = ont2.getAxioms();
84  // This isn't great - we normalise axioms by changing the ids of
85  // individuals. This relies on the fact that
86  // we iterate over objects in the same order for the same set of axioms!
87  OWLDataFactory df = ont1.getOWLOntologyManager().getOWLDataFactory();
88  AnonymousIndividualsNormaliser normaliser1 = new AnonymousIndividualsNormaliser(
89  df);
90  axioms1 = normaliser1.getNormalisedAxioms(axioms1);
91  AnonymousIndividualsNormaliser normaliser2 = new AnonymousIndividualsNormaliser(
92  df);
93  axioms2 = normaliser2.getNormalisedAxioms(axioms2);
94  if (!axioms1.equals(axioms2)) {
95  int counter = 0;
96  StringBuilder sb = new StringBuilder();
97  for (OWLAxiom ax : axioms1) {
98  if (!axioms2.contains(ax)) {
99  if (!isIgnorableAxiom(ax, false)) {
100  sb.append("Rem axiom: ");
101  sb.append(ax);
102  sb.append("\n");
103  counter++;
104  }
105  }
106  }
107  for (OWLAxiom ax : axioms2) {
108  if (!axioms1.contains(ax)) {
109  if (!isIgnorableAxiom(ax, true)) {
110  sb.append("Add axiom: ");
111  sb.append(ax);
112  sb.append("\n");
113  counter++;
114  }
115  }
116  }
117  if (counter > 0) {
118  System.out.println(this.getClass().getSimpleName()
119  + " roundTripOntology() Failing to match axioms: "
120  + sb.toString());
121  return false;
122  } else {
123  return true;
124  }
125  }
126  assertEquals(axioms1, axioms2);
127  return true;
128  }

137  public boolean isIgnorableAxiom(OWLAxiom ax, boolean parse) {
138  if (ax instanceof OWLDeclarationAxiom) {
139  OWLDeclarationAxiom d = (OWLDeclarationAxiom) ax;
140  if (parse) {
141  // all extra declarations in the parsed ontology are fine
142  return true;
143  }
144  // declarations of builtin and named individuals can be ignored
145  return d.getEntity().isBuiltIn() || d.getEntity().isOWLNamedIndividual();
146  }
147  return false;
148  }

217  public OWLOntology roundTripOntology(OWLOntology ont, OWLOntologyFormat format)
218  throws Exception {
219  StringDocumentTarget target = new StringDocumentTarget();
220  OWLOntologyFormat fromFormat = manager.getOntologyFormat(ont);
221  if (fromFormat instanceof PrefixOWLOntologyFormat
222  && format instanceof PrefixOWLOntologyFormat) {
223  PrefixOWLOntologyFormat fromPrefixFormat = (PrefixOWLOntologyFormat) fromFormat;
224  PrefixOWLOntologyFormat toPrefixFormat = (PrefixOWLOntologyFormat) format;
225  toPrefixFormat.copyPrefixesFrom(fromPrefixFormat);
226  }
227  if (format instanceof RDFOntologyFormat) {
228  ((RDFOntologyFormat) format).setAddMissingTypes(false);
229  }
230  manager.saveOntology(ont, format, target);
231  handleSaved(target, format);
232  OWLOntologyManager man = Factory.getManager();
233  OWLOntology ont2 = man.loadOntologyFromOntologyDocument(new StringDocumentSource(
234  target.toString()), new OWLOntologyLoaderConfiguration()
235  .setReportStackTraces(true));
236  equal(ont, ont2);
237  return ont2;
238  }
