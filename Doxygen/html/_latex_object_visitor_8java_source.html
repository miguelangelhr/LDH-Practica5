320  @Override
321  public void visit(OWLDisjointClassesAxiom axiom) {
322  if (axiom.getClassExpressions().size() != 2) {
323  for (OWLClassExpression left : axiom.getClassExpressions()) {
324  for (OWLClassExpression right : axiom.getClassExpressions()) {
325  if (left != right) {
326  if (left.equals(subject)) {
327  left.accept(this);
328  writeSpace();
329  write(SUBCLASS);
330  writeSpace();
331  write(NOT);
332  writeSpace();
333  right.accept(this);
334  } else {
335  right.accept(this);
336  writeSpace();
337  write(SUBCLASS);
338  writeSpace();
339  write(NOT);
340  writeSpace();
341  left.accept(this);
342  }
343  writer.writeNewLine();
344  }
345  }
346  }
347  } else {
348  Iterator<OWLClassExpression> it = axiom.getClassExpressions()
349  .iterator();
350  OWLClassExpression descA = it.next();
351  OWLClassExpression descB = it.next();
352  OWLClassExpression lhs;
353  OWLClassExpression rhs;
354  if (descA.equals(subject)) {
355  lhs = descA;
356  rhs = descB;
357  } else {
358  lhs = descB;
359  rhs = descA;
360  }
361  lhs.accept(this);
362  writeSpace();
363  write(SUBCLASS);
364  writeSpace();
365  write(NOT);
366  writeSpace();
367  rhs.accept(this);
368  }
369  }

371  @Override
372  public void visit(OWLEquivalentClassesAxiom axiom) {
373  if (axiom.getClassExpressions().size() > 2) {
374  Set<Set<OWLClassExpression>> rendered = new HashSet<Set<OWLClassExpression>>();
375  for (OWLClassExpression left : axiom.getClassExpressions()) {
376  for (OWLClassExpression right : axiom.getClassExpressions()) {
377  if (left != right) {
378  Set<OWLClassExpression> cur = CollectionFactory
379  .createSet(left, right);
380  if (!rendered.contains(cur)) {
381  rendered.add(cur);
382  left.accept(this);
383  writeSpace();
384  write(EQUIV);
385  writeSpace();
386  right.accept(this);
387  }
388  }
389  }
390  }
391  } else if (axiom.getClassExpressions().size() == 2) {
392  Iterator<OWLClassExpression> it = axiom.getClassExpressions()
393  .iterator();
394  OWLClassExpression descA = it.next();
395  OWLClassExpression descB = it.next();
396  OWLClassExpression lhs;
397  OWLClassExpression rhs;
398  if (subject.equals(descA)) {
399  lhs = descA;
400  rhs = descB;
401  } else {
402  lhs = descB;
403  rhs = descA;
404  }
405  lhs.accept(this);
406  writeSpace();
407  write(EQUIV);
408  writeSpace();
409  rhs.accept(this);
410  }
411  }
