125  protected void write(String s) {
126  if (s == null) {
127  return;
128  }
129  int indexOfNewLine = s.indexOf('\n');
130  if (indexOfNewLine != -1) {
131  lastNewLinePos = currentPos + indexOfNewLine;
132  }

145  protected void write(String s, int lineLen) {
146  StringTokenizer tokenizer = new StringTokenizer(s, " \n\t-", true);
147  int currentLineLength = 0;
148  while (tokenizer.hasMoreTokens()) {
149  String curToken = tokenizer.nextToken();
150  write(curToken);
151  if (curToken.equals("\n")) {
152  writeTab();
153  }
154  currentLineLength += curToken.length();
155  if (currentLineLength > lineLen && curToken.trim().length() != 0
156  && tokenizer.hasMoreTokens()) {
157  writeNewLine();
158  currentLineLength = 0;
159  }
160  }
161  }
