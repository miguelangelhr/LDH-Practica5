133  @Override
134  public Set<OWLClass> getNamedClasses() {
135  Set<OWLClass> toReturn = null;
136  if (namedClasses != null) {
137  toReturn = namedClasses.get();
138  }
139  if (toReturn == null) {
140  Set<OWLClass> clses = new HashSet<OWLClass>(1);
141  for (OWLClassExpression desc : getClassExpressions()) {
142  if (!desc.isAnonymous() && !desc.isOWLNothing() && !desc.isOWLThing()) {
143  clses.add(desc.asOWLClass());
144  }
145  }
146  toReturn = CollectionFactory
147  .getCopyOnRequestSetFromImmutableCollection(clses);
148  namedClasses = new WeakReference<Set<OWLClass>>(toReturn);
149  }
150  return toReturn;
151  }
