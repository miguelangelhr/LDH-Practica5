126  @Override
127  public Set<Set<OWLAxiom>> getExplanations(OWLClassExpression unsatClass,
128  int maxExplanations) {
129  if (maxExplanations < 0) {
130  throw new IllegalArgumentException();
131  }
132  if (log.isLoggable(Level.CONFIG)) {
133  log.config("Get " + (maxExplanations == 0 ? "all" : maxExplanations)
134  + " explanation(s) for: " + unsatClass);
135  }
136  try {
137  Set<OWLAxiom> firstMups = getExplanation(unsatClass);
138  if (firstMups.isEmpty()) {
139  return Collections.emptySet();
140  }
141  Set<Set<OWLAxiom>> allMups = new LinkedHashSet<Set<OWLAxiom>>();
142  progressMonitor.foundExplanation(firstMups);
143  allMups.add(firstMups);
144  Set<Set<OWLAxiom>> satPaths = new HashSet<Set<OWLAxiom>>();
145  Set<OWLAxiom> currentPathContents = new HashSet<OWLAxiom>();
146  singleExplanationGenerator.beginTransaction();
147  try {
148  constructHittingSetTree(unsatClass, firstMups, allMups, satPaths,
149  currentPathContents, maxExplanations);
150  } finally {
151  singleExplanationGenerator.endTransaction();
152  }
153  progressMonitor.foundAllExplanations();
154  return allMups;
155  } catch (OWLException e) {
156  throw new OWLRuntimeException(e);
157  }
158  }

172  private static List<OWLAxiom> getOrderedMUPS(List<OWLAxiom> mups,
173  final Set<Set<OWLAxiom>> allMups) {
174  Comparator<OWLAxiom> mupsComparator = new Comparator<OWLAxiom>() {
175  @Override
176  public int compare(OWLAxiom o1, OWLAxiom o2) {
177  // The axiom that appears in most MUPS has the lowest index
178  // in the list
179  int occ1 = getOccurrences(o1, allMups);
180  int occ2 = getOccurrences(o2, allMups);
181  return -occ1 + occ2;
182  }
183  };
184  Collections.sort(mups, mupsComparator);
185  return mups;
186  }

195  protected static int getOccurrences(OWLAxiom ax, Set<Set<OWLAxiom>> axiomSets) {
196  int count = 0;

233  private void constructHittingSetTree(OWLClassExpression unsatClass,
234  Set<OWLAxiom> mups, Set<Set<OWLAxiom>> allMups, Set<Set<OWLAxiom>> satPaths,
235  Set<OWLAxiom> currentPathContents, int maxExplanations) throws OWLException {
236  if (log.isLoggable(Level.FINE)) {
237  log.fine("MUPS " + allMups.size() + ": " + mups);
238  }
239  if (progressMonitor.isCancelled()) {
240  return;
241  }
242  // We go through the current mups, axiom by axiom, and extend the tree
243  // with edges for each axiom
244  List<OWLAxiom> orderedMups = getOrderedMUPS(new ArrayList<OWLAxiom>(mups),
245  allMups);
246  while (!orderedMups.isEmpty()) {
247  if (progressMonitor.isCancelled()) {
248  return;
249  }
250  OWLAxiom axiom = orderedMups.get(0);
251  orderedMups.remove(0);
252  if (allMups.size() == maxExplanations) {
253  if (log.isLoggable(Level.FINE)) {
254  log.fine("Computed " + maxExplanations + "explanations");
255  }
256  return;
257  }
258  if (log.isLoggable(Level.FINE)) {
259  log.fine("Removing axiom: " + axiom + " " + currentPathContents.size()
260  + " more removed: " + currentPathContents);
261  }
262  // Remove the current axiom from all the ontologies it is included
263  // in
264  Set<OWLOntology> ontologies = OntologyUtils.removeAxiom(axiom, getReasoner()
265  .getRootOntology().getImportsClosure(), getOntologyManager());
266  // Removal may have dereferenced some entities, if so declarations
267  // are added
268  Set<OWLEntity> sig = getSignature(axiom);
269  List<OWLDeclarationAxiom> temporaryDeclarations = new ArrayList<OWLDeclarationAxiom>(
270  sig.size());
271  for (OWLEntity e : sig) {
272  boolean referenced = false;
273  for (Iterator<OWLOntology> i = ontologies.iterator(); !referenced
274  && i.hasNext();) {
275  for (Iterator<OWLAxiom> j = i.next().getReferencingAxioms(e)
276  .iterator(); !referenced && j.hasNext();) {
277  OWLAxiom a = j.next();
278  referenced = a.isLogicalAxiom()
279  || a instanceof OWLDeclarationAxiom;
280  }
281  }
282  if (!referenced) {
283  OWLDeclarationAxiom declaration = getOntologyManager()
284  .getOWLDataFactory().getOWLDeclarationAxiom(e);
285  temporaryDeclarations.add(declaration);
286  }
287  }
288  for (OWLDeclarationAxiom decl : temporaryDeclarations) {
289  OntologyUtils.addAxiom(decl, getReasoner().getRootOntology()
290  .getImportsClosure(), getOntologyManager());
291  }
292  currentPathContents.add(axiom);
293  boolean earlyTermination = false;
294  // Early path termination. If our path contents are the superset of
295  // the contents of a path then we can terminate here.
296  for (Set<OWLAxiom> satPath : satPaths) {
297  if (currentPathContents.containsAll(satPath)) {
298  earlyTermination = true;
299  if (log.isLoggable(Level.FINE)) {
300  log.fine("Stop - satisfiable (early termination)");
301  }
302  break;
303  }
304  }
305  if (!earlyTermination) {
306  Set<OWLAxiom> newMUPS = null;
307  for (Set<OWLAxiom> foundMUPS : allMups) {
308  Set<OWLAxiom> foundMUPSCopy = new HashSet<OWLAxiom>(foundMUPS);
309  foundMUPSCopy.retainAll(currentPathContents);
310  if (foundMUPSCopy.isEmpty()) {
311  newMUPS = foundMUPS;
312  break;
313  }
314  }
315  if (newMUPS == null) {
316  newMUPS = getExplanation(unsatClass);
317  }
318  // Generate a new node - i.e. a new justification set
319  if (newMUPS.contains(axiom)) {
320  // How can this be the case???
321  throw new OWLRuntimeException("Explanation contains removed axiom: "
322  + axiom);
323  }
324  if (!newMUPS.isEmpty()) {
325  // Note that getting a previous justification does not mean
326  // we can stop. stopping here causes some justifications to
327  // be missed
328  allMups.add(newMUPS);
329  progressMonitor.foundExplanation(newMUPS);
330  // Recompute priority here?
331  constructHittingSetTree(unsatClass, newMUPS, allMups, satPaths,
332  currentPathContents, maxExplanations);
333  // We have found a new MUPS, so recalculate the ordering
334  // axioms in the MUPS at the current level
335  orderedMups = getOrderedMUPS(orderedMups, allMups);
336  } else {
337  if (log.isLoggable(Level.FINE)) {
338  log.fine("Stop - satisfiable");
339  }
340  // End of current path - add it to the list of paths
341  satPaths.add(new HashSet<OWLAxiom>(currentPathContents));
342  }
343  }
344  // Back track - go one level up the tree and run for the next axiom
345  currentPathContents.remove(axiom);
346  if (log.isLoggable(Level.FINE)) {
347  log.fine("Restoring axiom: " + axiom);
348  }
349  // Remove any temporary declarations
350  for (OWLDeclarationAxiom decl : temporaryDeclarations) {
351  OntologyUtils.removeAxiom(decl, getReasoner().getRootOntology()
352  .getImportsClosure(), getOntologyManager());
353  }
354  // Done with the axiom that was removed. Add it back in
355  OntologyUtils.addAxiom(axiom, ontologies, getOntologyManager());
356  }
357  }
