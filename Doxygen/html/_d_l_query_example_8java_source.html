114  @SuppressWarnings("javadoc")
115  public static void main(String[] args) {
116  try {
117  // Load an example ontology. In this case, we'll just load the pizza
118  // ontology.
119  OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
120  OWLOntology ontology = manager
121  .loadOntologyFromOntologyDocument(new StringDocumentSource(koala));
122  System.out.println("Loaded ontology: " + ontology.getOntologyID());
123  // We need a reasoner to do our query answering
124  OWLReasoner reasoner = createReasoner(ontology);
125  // Entities are named using IRIs. These are usually too long for use
126  // in user interfaces. To solve this
127  // problem, and so a query can be written using short class,
128  // property, individual names we use a short form
129  // provider. In this case, we'll just use a simple short form
130  // provider that generates short froms from IRI
131  // fragments.
132  ShortFormProvider shortFormProvider = new SimpleShortFormProvider();
133  // Create the DLQueryPrinter helper class. This will manage the
134  // parsing of input and printing of results
135  DLQueryPrinter dlQueryPrinter = new DLQueryPrinter(new DLQueryEngine(
136  reasoner, shortFormProvider), shortFormProvider);
137  // Enter the query loop. A user is expected to enter class
138  // expression on the command line.
139  doQueryLoop(dlQueryPrinter);
140  } catch (OWLOntologyCreationException e) {
141  System.out.println("Could not load ontology: " + e.getMessage());
142  } catch (IOException ioEx) {
143  System.out.println(ioEx.getMessage());
144  }
145  }

147  private static void doQueryLoop(DLQueryPrinter dlQueryPrinter) throws IOException {
148  while (true) {
149  // Prompt the user to enter a class expression
150  System.out
151  .println("Please type a class expression in Manchester Syntax and press Enter (or press x to exit):");
152  System.out.println("");
153  String classExpression = readInput();
154  // Check for exit condition
155  if (classExpression.equalsIgnoreCase("x")) {
156  break;
157  }
158  dlQueryPrinter.askQuery(classExpression.trim());
159  System.out.println();
160  System.out.println();
161  }
162  }

212  public Set<OWLClass> getSuperClasses(String classExpressionString, boolean direct) {
213  if (classExpressionString.trim().length() == 0) {
214  return Collections.emptySet();
215  }
216  OWLClassExpression classExpression = parser
217  .parseClassExpression(classExpressionString);
218  NodeSet<OWLClass> superClasses = reasoner
219  .getSuperClasses(classExpression, direct);
220  return superClasses.getFlattened();
221  }
222 
229  public Set<OWLClass> getEquivalentClasses(String classExpressionString) {
230  if (classExpressionString.trim().length() == 0) {
231  return Collections.emptySet();
232  }
233  OWLClassExpression classExpression = parser
234  .parseClassExpression(classExpressionString);
235  Node<OWLClass> equivalentClasses = reasoner.getEquivalentClasses(classExpression);
236  Set<OWLClass> result;
237  if (classExpression.isAnonymous()) {
238  result = equivalentClasses.getEntities();
239  } else {
240  result = equivalentClasses.getEntitiesMinus(classExpression.asOWLClass());
241  }
242  return result;
243  }

284 class DLQueryParser {
285  private final OWLOntology rootOntology;
286  private final BidirectionalShortFormProvider bidiShortFormProvider;
287 
297  public DLQueryParser(OWLOntology rootOntology, ShortFormProvider shortFormProvider) {
298  this.rootOntology = rootOntology;
299  OWLOntologyManager manager = rootOntology.getOWLOntologyManager();
300  Set<OWLOntology> importsClosure = rootOntology.getImportsClosure();
301  // Create a bidirectional short form provider to do the actual mapping.
302  // It will generate names using the input
303  // short form provider.
304  bidiShortFormProvider = new BidirectionalShortFormProviderAdapter(manager,
305  importsClosure, shortFormProvider);
306  }
307 
314  public OWLClassExpression parseClassExpression(String classExpressionString) {
315  OWLDataFactory dataFactory = rootOntology.getOWLOntologyManager()
316  .getOWLDataFactory();
317  // Set up the real parser
318  ManchesterOWLSyntaxEditorParser parser = new ManchesterOWLSyntaxEditorParser(
319  dataFactory, classExpressionString);
320  parser.setDefaultOntology(rootOntology);
321  // Specify an entity checker that wil be used to check a class
322  // expression contains the correct names.
323  OWLEntityChecker entityChecker = new ShortFormEntityChecker(bidiShortFormProvider);
324  parser.setOWLEntityChecker(entityChecker);
325  // Do the actual parsing
326  return parser.parseClassExpression();
327  }
328 }

345  public void askQuery(String classExpression) {
346  if (classExpression.length() == 0) {
347  System.out.println("No class expression specified");
348  } else {
349  try {
350  StringBuilder sb = new StringBuilder();
351  sb.append("\n--------------------------------------------------------------------------------\n");
352  sb.append("QUERY: ");
353  sb.append(classExpression);
354  sb.append("\n");
355  sb.append("--------------------------------------------------------------------------------\n\n");
356  // Ask for the subclasses, superclasses etc. of the specified
357  // class expression. Print out the results.
358  Set<OWLClass> superClasses = dlQueryEngine.getSuperClasses(
359  classExpression, true);
360  printEntities("SuperClasses", superClasses, sb);
361  Set<OWLClass> equivalentClasses = dlQueryEngine
362  .getEquivalentClasses(classExpression);
363  printEntities("EquivalentClasses", equivalentClasses, sb);
364  Set<OWLClass> subClasses = dlQueryEngine.getSubClasses(classExpression,
365  true);
366  printEntities("SubClasses", subClasses, sb);
367  Set<OWLNamedIndividual> individuals = dlQueryEngine.getInstances(
368  classExpression, true);
369  printEntities("Instances", individuals, sb);
370  // System.out.println(sb.toString());
371  } catch (ParserException e) {
372  System.out.println(e.getMessage());
373  }
374  }
375  }

377  private void printEntities(String name, Set<? extends OWLEntity> entities,
378  StringBuilder sb) {
379  sb.append(name);
380  int length = 50 - name.length();
381  for (int i = 0; i < length; i++) {
382  sb.append(".");
383  }
384  sb.append("\n\n");
385  if (!entities.isEmpty()) {
386  for (OWLEntity entity : entities) {
387  sb.append("\t");
388  sb.append(shortFormProvider.getShortForm(entity));
389  sb.append("\n");
390  }
391  } else {
392  sb.append("\t[NONE]\n");
393  }
394  sb.append("\n");
395  }
