454  private List<OWLOntologyChange> enactChangeApplication(OWLOntologyChange change) {
455  if (!isChangeApplicable(change)) {
456  return Collections.emptyList();
457  }
458  OWLOntology ont = change.getOntology();
459  if (!(ont instanceof OWLMutableOntology)) {

462  checkForOntologyIDChange(change);

465  checkForImportsChange(change);

469  @Override
470  public List<OWLOntologyChange>
471  applyChanges(List<? extends OWLOntologyChange> changes) {
472  try {
473  broadcastImpendingChanges(changes);
474  } catch (OWLOntologyChangeVetoException e) {
475  // Some listener blocked the changes.
476  broadcastOntologyChangesVetoed(changes, e);
477  return Collections.emptyList();
478  }
479  List<OWLOntologyChange> appliedChanges = new ArrayList<OWLOntologyChange>(
480  changes.size() + 2);
481  fireBeginChanges(changes.size());
482  for (OWLOntologyChange change : changes) {
483  appliedChanges.addAll(enactChangeApplication(change));
484  fireChangeApplied(change);
485  }
486  fireEndChanges();
487  broadcastChanges(appliedChanges);
488  return appliedChanges;
489  }

528  private void checkForImportsChange(OWLOntologyChange change) {
529  if (change.isImportChange()) {
530  resetImportsClosureCache();
531  if (change instanceof AddImport) {
532  OWLImportsDeclaration addImportDeclaration = ((AddImport) change)
533  .getImportDeclaration();
534  boolean found = false;
535  IRI iri = addImportDeclaration.getIRI();
536  for (OWLOntologyID id : ontologiesByID.keySet()) {
537  if (iri.equals(id.getDefaultDocumentIRI())
538  || iri.equals(id.getOntologyIRI())
539  || iri.equals(id.getVersionIRI())) {
540  found = true;
541  ontologyIDsByImportsDeclaration.put(addImportDeclaration, id);
542  }
543  }
544  if (!found) {
545  // then the import does not refer to a known IRI for
546  // ontologies; check for a document IRI
547  for (Map.Entry<OWLOntologyID, IRI> e : documentIRIsByID.entrySet()) {
548  if (e.getValue().equals(iri)) {
549  // found the ontology id corresponding to the file
550  // location
551  ontologyIDsByImportsDeclaration.put(addImportDeclaration,
552  e.getKey());
553  }
554  }
555  }
556  } else {
557  // Remove the mapping from declaration to ontology
558  ontologyIDsByImportsDeclaration.remove(((RemoveImport) change)
559  .getImportDeclaration());
560  }
561  }
562  }

564  private void checkForOntologyIDChange(OWLOntologyChange change) {
565  if (change instanceof SetOntologyID) {
566  SetOntologyID setID = (SetOntologyID) change;
567  OWLOntology existingOntology = ontologiesByID.get(((SetOntologyID) change)
568  .getNewOntologyID());
569  if (existingOntology != null
570  && !change.getOntology().equals(existingOntology)) {
571  if (!change.getOntology().getAxioms()
572  .equals(existingOntology.getAxioms())) {
573  throw new OWLOntologyRenameException(change,
574  ((SetOntologyID) change).getNewOntologyID());
575  }
576  }
577  renameOntology(setID.getOriginalOntologyID(), setID.getNewOntologyID());
578  resetImportsClosureCache();
579  }

710  protected OWLOntology loadOntology(IRI ontologyIRI, boolean allowExists,
711  OWLOntologyLoaderConfiguration configuration)
712  throws OWLOntologyCreationException {
713  OWLOntology ontByID = null;
714  // Check for matches on the ontology IRI first
715  for (OWLOntologyID nextOntologyID : ontologiesByID.keySet()) {
716  if (ontologyIRI.equals(nextOntologyID.getOntologyIRI())) {
717  ontByID = ontologiesByID.get(nextOntologyID);
718  }
719  }
720  // This method may be called using a version IRI, so also check the
721  // version IRI if necessary
722  if (ontByID == null) {
723  for (OWLOntologyID nextOntologyID : ontologiesByID.keySet()) {
724  if (ontologyIRI.equals(nextOntologyID.getVersionIRI())) {
725  ontByID = ontologiesByID.get(nextOntologyID);
726  }
727  }
728  }
729  if (ontByID != null) {
730  return ontByID;
731  }
732  OWLOntologyID id = new OWLOntologyID(ontologyIRI);
733  IRI documentIRI = getDocumentIRIFromMappers(id, true);
734  if (documentIRI != null) {
735  if (documentIRIsByID.values().contains(documentIRI) && !allowExists) {
736  throw new OWLOntologyDocumentAlreadyExistsException(documentIRI);
737  }
738  // The ontology might be being loaded, but its IRI might
739  // not have been set (as is probably the case with RDF/XML!)
740  OWLOntology ontByDocumentIRI = getOntologyByDocumentIRI(documentIRI);
741  if (ontByDocumentIRI != null) {
742  return ontByDocumentIRI;
743  }
744  } else {
745  // Nothing we can do here. We can't get a document IRI to load
746  // the ontology from.
747  throw new OntologyIRIMappingNotFoundException(ontologyIRI);
748  }
749  return loadOntology(ontologyIRI, new IRIDocumentSource(documentIRI),
750  configuration);
751  }

808  protected OWLOntology loadOntology(IRI ontologyIRI,
809  OWLOntologyDocumentSource documentSource,
810  OWLOntologyLoaderConfiguration configuration)
811  throws OWLOntologyCreationException {
812  if (loadCount != importsLoadCount) {
813  System.err
814  .println("Runtime Warning: Parsers should load imported ontologies using the makeImportLoadRequest method.");
815  }
816  fireStartedLoadingEvent(new OWLOntologyID(ontologyIRI),
817  documentSource.getDocumentIRI(), loadCount > 0);
818  loadCount++;
819  broadcastChanges = false;
820  OWLOntologyCreationException ex = null;
821  OWLOntologyID idOfLoadedOntology = new OWLOntologyID();
822  try {
823  for (OWLOntologyFactory factory : ontologyFactories) {
824  if (factory.canLoad(documentSource)) {
825  try {
826  // Note - there is no need to add the ontology here,
827  // because it will be added
828  // when the ontology is created.
829  OWLOntology ontology = factory.loadOWLOntology(documentSource,
830  this, configuration);
831  idOfLoadedOntology = ontology.getOntologyID();
832  // Store the ontology to the document IRI mapping
833  documentIRIsByID.put(ontology.getOntologyID(),
834  documentSource.getDocumentIRI());
835  return ontology;
836  } catch (OWLOntologyRenameException e) {
837  // We loaded an ontology from a document and the
838  // ontology turned out to have an IRI the same
839  // as a previously loaded ontology
840  throw new OWLOntologyAlreadyExistsException(e.getOntologyID(), e);
841  }
842  }
843  }
844  } catch (OWLOntologyCreationException e) {
845  ex = e;
846  throw e;
847  } finally {
848  loadCount--;
849  if (loadCount == 0) {
850  broadcastChanges = true;
851  // Completed loading ontology and imports
852  }
853  fireFinishedLoadingEvent(idOfLoadedOntology, documentSource.getDocumentIRI(),
854  loadCount > 0, ex);
855  }
856  throw new OWLOntologyFactoryNotFoundException(documentSource.getDocumentIRI());
857  }

1121  for (OWLOntologyChangeListener listener : new ArrayList<OWLOntologyChangeListener>(
1122  listenerMap.keySet())) {

1126  // the changes,

1129  // we iterate over a copy).

1135  // to prevent the other listeners from receiving events.
1136  strategy.broadcastChanges(listener, changes);
