148  @Override
149  public void render(OWLOntology ontology, Writer writer) throws OWLRendererException {
150  factory = ontology.getOWLOntologyManager().getOWLDataFactory();
151  exceptions.clear();
152 
153  IRI ontologyIRI = ontology.getOntologyID().getOntologyIRI();
154  if (ontologyIRI != null) {
155  final String ontURIStr = ontologyIRI.toString();
156  if (ontURIStr.endsWith("/")) {
157  defaultNamespace = ontURIStr;
158  } else {
159  defaultNamespace = ontURIStr + "#";
160  }
161  } else {
162  defaultNamespace = "urn:defaultOBONamespace:ontology"
163  + System.currentTimeMillis() + "#";
164  System.err.println("WARNING: anonymous ontology saved in OBO format. Default namespace created for it.");
165  }
166 
167  nsUtil = new NamespaceUtil();
168  defaultPrefix = nsUtil.getPrefix(defaultNamespace);
169 
170  writeHeader(ontology, writer);
171  writeStanzas(ontology, writer);
172 
173  if (!exceptions.isEmpty()) {
174  throw new OBOStorageIncompleteException(exceptions);
175  }
176  }

191  private void writeHeader(OWLOntology ontology, Writer writer) {
192  OBOTagValuePairList tvpList = new OBOTagValuePairList(OBOVocabulary.getHeaderTags());
193 
194  tvpList.setDefault(OBOVocabulary.DEFAULT_NAMESPACE, defaultPrefix);
195 
196  for (OWLAnnotation ax : ontology.getAnnotations()) {
197  if (ax.getProperty().isComment()) {
198  tvpList.addPair(OBOVocabulary.REMARK, ((OWLLiteral) ax.getValue()).getLiteral());
199  }
200  else {
201  tvpList.visit(ax);
202  }
203  }
204 
205  for (OWLImportsDeclaration importDecl : ontology.getImportsDeclarations()) {
206  tvpList.addPair(OBOVocabulary.IMPORT, importDecl.getIRI().toString());
207  }
208 
209  Map<String, String> namespace2PrefixMap = loadUsedNamespaces(ontology);
210  for (String namespace : namespace2PrefixMap.keySet()) {
211  String mapping = namespace2PrefixMap.get(namespace) + " " + namespace;
212  tvpList.addPair(OBOVocabulary.ID_SPACE, mapping);
213  }
214 
215  // overwrite the existing values for below
216  tvpList.setPair(OBOVocabulary.FORMAT_VERSION, "1.2");
217  tvpList.setPair(OBOVocabulary.DATE, getTimestampFormatter().format(new Date(System.currentTimeMillis())));
218  tvpList.setPair(OBOVocabulary.SAVED_BY, System.getProperty("user.name"));
219  tvpList.setPair(OBOVocabulary.AUTO_GENERATED_BY, VersionInfo.getVersionInfo().toString());
220 
221  tvpList.write(writer);
222  }

235  private void writeStanzas(OWLOntology ontology, Writer writer) {
236 
237  write("\n\n! ---------------------- CLASSES -------------------------\n", writer);
238 
239  final List<OWLClass> sortedClasses = new ArrayList<OWLClass>(ontology.getClassesInSignature());
240  Collections.sort(sortedClasses);
241  for (OWLClass cls : sortedClasses) {
242  writeTermStanza(cls, ontology, writer);
243  }
244 
245 
246  write("\n\n! ---------------------- PROPERTIES -------------------------\n", writer);
247 
248  final List<OWLObjectProperty> objProps = new ArrayList<OWLObjectProperty>(ontology.getObjectPropertiesInSignature());
249  Collections.sort(objProps);
250  for (OWLObjectProperty property : objProps) {
251  writeTypeDefStanza(property, ontology, writer);
252  }
253 
254  final List<OWLDataProperty> dataProps = new ArrayList<OWLDataProperty>(ontology.getDataPropertiesInSignature());
255  Collections.sort(dataProps);
256  for (OWLDataProperty property : dataProps) {
257  writeTypeDefStanza(property, ontology, writer);
258  }
259 
260 
261  write("\n\n! ---------------------- INSTANCES -------------------------\n", writer);
262 
263  final List<OWLNamedIndividual> individuals = new ArrayList<OWLNamedIndividual>(ontology.getIndividualsInSignature());
264  Collections.sort(individuals);
265  for (OWLNamedIndividual individual : individuals) {
266  writeInstanceStanza(individual, ontology, writer);
267  }
268 
269  // scan ontology for other constructs that cannot be translated
270 
271  for (OWLClassAxiom ax : ontology.getGeneralClassAxioms()) {
272  if (ax instanceof OWLSubClassOfAxiom) {
273  exceptions.add(new OBOStorageException(ax, null, "Superclass GCI found in ontology cannot be translated to OBO"));
274  }
275  if (ax instanceof OWLEquivalentClassesAxiom) {
276  exceptions.add(new OBOStorageException(ax, null, "Equivalent class GCI found in ontology cannot be translated to OBO"));
277  }
278  if (ax instanceof OWLDisjointClassesAxiom) {
279  exceptions.add(new OBOStorageException(ax, null, "Disjoint axiom contains anonymous classes - cannot be translated to OBO"));
280  }
281  }
282 
283 
284  for (SWRLRule r : ontology.getAxioms(AxiomType.SWRL_RULE)) {
285  exceptions.add(new OBOStorageException(r, null, "SWRL rules cannot be translated to OBO"));
286  }
287 
288  }

291  private void writeTermStanza(OWLClass cls, OWLOntology ontology, Writer writer) {
292  write("\n[", writer);
293  write(OBOVocabulary.TERM.getName(), writer);
294  write("]\n", writer);
295 
296  // TODO is_anonymous??
297 
298  OBOTagValuePairList tvpList = new OBOTagValuePairList(OBOVocabulary.getTermStanzaTags());
299 
300  handleEntityBase(cls, ontology, tvpList);
301 
302  relationshipHandler.setClass(cls);
303 
304  for (OWLClassExpression superCls : cls.getSuperClasses(ontology)) {
305  if (!superCls.isAnonymous()) {
306  String superclassID = getID(superCls.asOWLClass());
307  tvpList.addPair(OBOVocabulary.IS_A, superclassID);
308  // TODO handle namespace and derived?
309  }
310  else if (superCls instanceof OWLRestriction) {
311  // TODO handle datatype restrictions
312  superCls.accept(relationshipHandler);
313  }
314  else {
315  exceptions.add(new OBOStorageException(cls, superCls, "OBO format only supports named superclass or someValuesFrom restrictions"));
316  }
317  }
318 
319  final OWLClass owlThing = factory.getOWLThing();
320 
321  // if no named superclass is specified, then this must be asserted to be a subclass of owlapi:Thing
322  if (!cls.equals(owlThing) && tvpList.getValues(OBOVocabulary.IS_A).isEmpty()) {
323  tvpList.addPair(OBOVocabulary.IS_A, getID(owlThing));
324  }
325 
326  for (OWLClassExpression equiv : cls.getEquivalentClasses(ontology)) {
327  if (equiv instanceof OWLObjectIntersectionOf) {
328  handleIntersection(cls, (OWLObjectIntersectionOf) equiv, tvpList);
329  }
330  else if (equiv instanceof OWLObjectUnionOf) {
331  handleUnion(cls, (OWLObjectUnionOf) equiv, tvpList);
332  }
333  else if (equiv instanceof OWLRestriction) {
334  /* OBO equivalence must be of the form "A and p some B and ..."
335  * if this class is equiv to a restriction, put this into an intersection with owlapi:Thing as the named class
336  */
337  OWLObjectIntersectionOf intersection = factory
338  .getOWLObjectIntersectionOf(owlThing, equiv);
339  handleIntersection(cls, intersection, tvpList);
340  }
341  else {
342  // TODO handle datatype restrictions
343  exceptions.add(new OBOStorageException(cls, equiv, "Cannot answer equivalent class that is not intersection or union"));
344  }
345  }
346 
347  for (OWLClassExpression disjoint : cls.getDisjointClasses(ontology)) {
348  if (!disjoint.isAnonymous()) {
349  tvpList.addPair(OBOVocabulary.DISJOINT_FROM, getID(disjoint.asOWLClass()));
350  // TODO handle namespace and derived
351  }
352  else {
353  exceptions.add(new OBOStorageException(cls, disjoint, "Found anonymous disjoint class that cannot be represented in OBO"));
354  }
355  }
356 
357  for (OBORelationship relationship : relationshipHandler.getOBORelationships()) {
358  // TODO handle datatype restrictions
359  handleRelationship(relationship, tvpList);
360  }
361 
362  tvpList.write(writer);
363  }

366  private void handleIntersection(OWLClass cls, OWLObjectIntersectionOf intersectionOf, OBOTagValuePairList tvpList) {
367  for (OWLClassExpression op : intersectionOf.getOperands()) {
368  if (!op.isAnonymous()) {
369  tvpList.addPair(OBOVocabulary.INTERSECTION_OF, getID(op.asOWLClass()));
370  }
371  else {
372  relationshipHandler.setClass(cls);
373  op.accept(relationshipHandler);
374  Set<OBORelationship> relations = relationshipHandler.getOBORelationships();
375  if (!relations.isEmpty()) {
376  OBORelationship rel = relations.iterator().next();
377  tvpList.addPair(OBOVocabulary.INTERSECTION_OF, renderRestriction(rel));
378  }
379  else {
380  exceptions.add(new OBOStorageException(cls, op, "Found operand in intersection that cannot be represented"));
381  }
382  }
383  }
384  // TODO handle namespace
385  }

417  private void handleRelationship(OBORelationship relationship, OBOTagValuePairList tvpList) {
418  StringBuilder sb = new StringBuilder();
419  sb.append(renderRestriction(relationship));
420  sb.append("\n");
421 
422  if (relationship.getCardinality() >= 0) {
423  sb.append(OBOVocabulary.CARDINALITY.getName());
424  sb.append(":");
425  sb.append(Integer.toString(relationship.getCardinality()));
426  sb.append("\n");
427  }
428  if (relationship.getMaxCardinality() >= 0) {
429  sb.append(OBOVocabulary.MAX_CARDINALITY.getName());
430  sb.append(":");
431  sb.append(Integer.toString(relationship.getMaxCardinality()));
432  sb.append("\n");
433  }
434  if (relationship.getMinCardinality() >= 0) {
435  sb.append(OBOVocabulary.MIN_CARDINALITY.getName());
436  sb.append(":");
437  sb.append(Integer.toString(relationship.getMinCardinality()));
438  sb.append("\n");
439  }
440  tvpList.addPair(OBOVocabulary.RELATIONSHIP, sb.toString());
441  }

444  private <P extends OWLProperty> OBOTagValuePairList handleCommonTypeDefStanza(P property, OWLOntology ontology, Writer writer) {
445  write("\n[", writer);
446  write(OBOVocabulary.TYPEDEF.getName(), writer);
447  write("]\n", writer);
448 
449  OBOTagValuePairList tvpList = new OBOTagValuePairList(OBOVocabulary.getTypeDefStanzaTags());
450  handleEntityBase(property, ontology, tvpList);
451 
452  Set<OWLClassExpression> domains = property.getDomains(ontology);
453  for (OWLClassExpression domain : domains) {
454  if (!domain.isAnonymous()) {
455  tvpList.addPair(OBOVocabulary.DOMAIN, getID(domain.asOWLClass()));
456  }
457  else {
458  exceptions.add(new OBOStorageException(property, domain, "Anonymous domain that cannot be represented in OBO"));
459  }
460  }
461 
462  final Set<OWLPropertyExpression<?,?>> sp = property.getSuperProperties(ontology);
463  for (OWLPropertyExpression<?,?> superProp : sp) {
464  if (!superProp.isAnonymous()) {
465  tvpList.addPair(OBOVocabulary.IS_A, getID((OWLProperty) superProp));
466  }
467  else {
468  exceptions.add(new OBOStorageException(property, superProp, "Anonymous property in superProperty is not supported in OBO"));
469  }
470  }
471 
472  tvpList.setDefault(OBOVocabulary.IS_METADATA_TAG, "false"); // annotation properties only
473 
474  return tvpList;
475  }

478  private void writeTypeDefStanza(OWLObjectProperty property, OWLOntology ontology, Writer writer) {
479 
480  OBOTagValuePairList tvpList = handleCommonTypeDefStanza(property, ontology, writer);
481 
482  for (OWLClassExpression range : property.getRanges(ontology)) {
483  if (!range.isAnonymous()) {
484  tvpList.addPair(OBOVocabulary.RANGE, getID(range.asOWLClass()));
485  }
486  else {
487  exceptions.add(new OBOStorageException(property, range, "Anonymous range that cannot be represented in OBO"));
488  }
489  }
490 
491  if (property.isAsymmetric(ontology)) {
492  tvpList.addPair(OBOVocabulary.IS_ASYMMETRIC, "true");
493  }
494  if (property.isReflexive(ontology)) {
495  tvpList.addPair(OBOVocabulary.IS_REFLEXIVE, "true");
496  }
497  if (property.isSymmetric(ontology)) {
498  tvpList.addPair(OBOVocabulary.IS_SYMMETRIC, "true");
499  }
500  if (property.isTransitive(ontology)) {
501  tvpList.addPair(OBOVocabulary.IS_TRANSITIVE, "true");
502  }
503 
504  for (OWLObjectPropertyExpression inv : property.getInverses(ontology)) {
505  if (!inv.isAnonymous()) {
506  tvpList.addPair(OBOVocabulary.INVERSE, getID(inv.asOWLObjectProperty()));
507  }
508  else {
509  exceptions.add(new OBOStorageException(property, inv, "Anonymous property in inverse is not supported in OBO"));
510  }
511  }
512 
513  // transitive over
514  for (OWLSubPropertyChainOfAxiom ax : ontology.getAxioms(AxiomType.SUB_PROPERTY_CHAIN_OF)) {
515  if (ax.getSuperProperty().equals(property)) {
516  final List<OWLObjectPropertyExpression> chain = ax.getPropertyChain();
517  if (chain.size() == 2 && chain.get(0).equals(property) && !chain.get(1).isAnonymous()) {
518  tvpList.addPair(OBOVocabulary.TRANSITIVE_OVER, getID(chain.get(1).asOWLObjectProperty()));
519  }
520  else {
521  exceptions.add(new OBOStorageException(property, ax, "Only property chains of form 'p o q -> p' supported"));
522  }
523  }
524  }
525 
526  // TODO is RELATIONSHIP really a part of the typedef stanza?
527 
528  tvpList.write(writer);
529  }

550  private void writeInstanceStanza(OWLNamedIndividual individual, OWLOntology ontology, Writer writer) {
551  write("\n[", writer);
552  write(OBOVocabulary.INSTANCE.getName(), writer);
553  write("]\n", writer);
554 
555  OBOTagValuePairList tvpList = new OBOTagValuePairList(OBOVocabulary.getInstanceStanzaTags());
556  if (individual.isAnonymous()) {
557  tvpList.setDefault(OBOVocabulary.IS_ANONYMOUS, "true");
558  }
559  handleEntityBase(individual, ontology, tvpList);
560 
561  for (OWLClassExpression type : individual.getTypes(ontology)) {
562  if (!type.isAnonymous()) {
563  tvpList.addPair(OBOVocabulary.INSTANCE_OF, getID(type.asOWLClass()));
564  }
565  else {
566  exceptions.add(new OBOStorageException(individual, type, "Complex types cannot be represented in OBO"));
567  }
568  }
569 
570  Map<OWLObjectPropertyExpression, Set<OWLIndividual>> objPropAssertions = individual.getObjectPropertyValues(ontology);
571  for (OWLObjectPropertyExpression p : objPropAssertions.keySet()) {
572  if (!p.isAnonymous()) {
573  for (OWLIndividual ind : objPropAssertions.get(p)) {
574  if (!ind.isAnonymous()) {
575  final String rel = renderRestriction(new OBORelationship(p.asOWLObjectProperty(), ind.asOWLNamedIndividual()));
576  tvpList.addPair(OBOVocabulary.PROPERTY_VALUE, rel);
577  }
578  }
579  }
580  else {
581  exceptions.add(new OBOStorageException(individual, p, "Anonymous property in assertion is not supported in OBO"));
582  }
583  }
584 
585  Map<OWLDataPropertyExpression, Set<OWLLiteral>> dataPropAssertions = individual.getDataPropertyValues(ontology);
586  for (OWLDataPropertyExpression p : dataPropAssertions.keySet()) {
587  if (!p.isAnonymous()) {
588  for (OWLLiteral literal : dataPropAssertions.get(p)) {
589  final String rel = renderPropertyAssertion(p.asOWLDataProperty(), literal);
590  tvpList.addPair(OBOVocabulary.PROPERTY_VALUE, rel);
591  }
592  }
593  else {
594  // no anonymous data properties so this should never occur
595  exceptions.add(new OBOStorageException(individual, p, "Anonymous property in assertion is not supported in OBO"));
596  }
597  }
598 
599  tvpList.write(writer);
600  }

603  private void handleEntityBase(OWLEntity entity, OWLOntology ontology, OBOTagValuePairList tvpList) {
604  tvpList.addPair(OBOVocabulary.ID, getID(entity));
605  Set<OWLAnnotation> potentialNames = new HashSet<OWLAnnotation>();
606  for (OWLAnnotation annotation : entity.getAnnotations(ontology)) {
607  if (annotation.getProperty().getIRI().equals(OWLRDFVocabulary.RDFS_LABEL.getIRI())) {
608  potentialNames.add(annotation);
609  }
610  else if (annotation.getProperty().isComment()) {
611  tvpList.addPair(OBOVocabulary.COMMENT, ((OWLLiteral) annotation.getValue()).getLiteral());
612  }
613  else {
614  tvpList.visit(annotation);
615  }
616  }
617 
618  if (tvpList.getValues(OBOVocabulary.NAME).isEmpty()) { // one name required!!
619  if (!potentialNames.isEmpty()) {
620  OWLAnnotation firstLabel = potentialNames.iterator().next();
621  tvpList.addPair(OBOVocabulary.NAME, ((OWLLiteral) firstLabel.getValue()).getLiteral());
622  potentialNames.remove(firstLabel);
623  }
624  else {
625  tvpList.addPair(OBOVocabulary.NAME, getID(entity));
626  }
627  }
628 
629  // other labels just get rendered as label
630  for (OWLAnnotation anno : potentialNames) {
