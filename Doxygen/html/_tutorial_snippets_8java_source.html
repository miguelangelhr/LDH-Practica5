489  private static class RestrictionVisitor extends OWLClassExpressionVisitorAdapter {
490  private final Set<OWLClass> processedClasses;
491  private final Set<OWLObjectPropertyExpression> restrictedProperties;
492  private final Set<OWLOntology> onts;
493 
494  public RestrictionVisitor(Set<OWLOntology> onts) {
495  restrictedProperties = new HashSet<OWLObjectPropertyExpression>();
496  processedClasses = new HashSet<OWLClass>();
497  this.onts = onts;
498  }
499 
500  public Set<OWLObjectPropertyExpression> getRestrictedProperties() {
501  return restrictedProperties;
502  }
503 
504  @Override
505  public void visit(OWLClass desc) {
506  // avoid cycles
507  if (!processedClasses.contains(desc)) {
508  // If we are processing inherited restrictions then
509  // we recursively visit named supers.
510  processedClasses.add(desc);
511  for (OWLOntology ont : onts) {
512  for (OWLSubClassOfAxiom ax : ont.getSubClassAxiomsForSubClass(desc)) {
513  ax.getSuperClass().accept(this);
514  }
515  }
