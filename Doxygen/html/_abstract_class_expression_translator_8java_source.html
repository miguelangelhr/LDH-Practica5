233  protected boolean isResourceListStrict(IRI mainNode, TypeMatcher typeMatcher, int minSize) {
234  if(mainNode == null) {
235  return false;
236  }
237  IRI currentListNode = mainNode;
238  Set<IRI> visitedListNodes = new HashSet<IRI>();
239  int size = 0;
240  while (true) {
241  IRI firstObject = consumer.getResourceObject(currentListNode, RDF_FIRST, false);
242  if(firstObject == null) {
243  return false;
244  }
245  if(!typeMatcher.isTypeStrict(firstObject)) {
246  // Something in the list that is not of the required type
247  return false;
248  }
249  else {
250  size++;
251  }
252  IRI restObject = consumer.getResourceObject(currentListNode, RDF_REST, false);
253  if(visitedListNodes.contains(restObject)) {
254  // Cycle - Non-terminating
255  return false;
256  }
257  if(restObject == null) {
258  // Not terminated properly
259  return false;
260  }
261  if(restObject.equals(RDF_NIL.getIRI())) {
262  // Terminated properly
263  return size >= minSize;
264  }
265  // Carry on
266  visitedListNodes.add(restObject);
267  currentListNode = restObject;
268  }
269  }
