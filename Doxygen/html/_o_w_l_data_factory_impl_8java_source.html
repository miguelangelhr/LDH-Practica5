169  private void checkNull(Object o, String name, boolean emptyAllowed) {
170  if (o == null) {
171  throw new IllegalArgumentException(name + " cannot be null");
172  }
173  if (o instanceof Iterable) {
174  int i = 0;
175  for (Object t : (Iterable<?>) o) {
176  if (t == null) {
177  throw new IllegalArgumentException(name + " element " + i + " cannot be null");
178  }
179  i++;
180  }
181  if (!emptyAllowed && i == 0) {
182  throw new IllegalArgumentException(name + " cannot be empty");
183  }
184  }
185  }
