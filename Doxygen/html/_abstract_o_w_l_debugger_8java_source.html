86  private void mergeImportsClosure() {
87  OWLOntology originalOntology = ontology;
88  try {
89  ontology = owlOntologyManager.createOntology(IRI.create("http://debugger.semanticweb.org/", "ontolog" + System.nanoTime()));
90  }
91  catch (OWLOntologyCreationException e) {
92  throw new OWLRuntimeException(e);
93  }
94  List<OWLOntologyChange> changes = new ArrayList<OWLOntologyChange>();
95  for (OWLOntology ont : owlOntologyManager.getImportsClosure(originalOntology)) {
96  for (OWLAxiom ax : ont.getLogicalAxioms()) {
97  changes.add(new AddAxiom(ontology, ax));
98  }
99  }
100  owlOntologyManager.applyChanges(changes);
101  }

152  public void constructHittingSetTree(Set<OWLAxiom> mups, Set<Set<OWLAxiom>> allMups, Set<Set<OWLAxiom>> satPaths, Set<OWLAxiom> currentPathContents) throws OWLException {
153 
154  // We go through the current mups, axiom by axiom, and extend the tree
155  // with edges for each axiom
156  for (OWLAxiom axiom : mups) {
157  // Remove the current axiom from the ontology
158  owlOntologyManager.applyChanges(Arrays.asList(new RemoveAxiom(ontology, axiom)));
159  currentPathContents.add(axiom);
160 
161  boolean earlyTermination = false;
162  // Early path termination. If our path contents are the superset of
163  // the contents of a path then we can terminate here.
164  for (Set<OWLAxiom> satPath : satPaths) {
165  if (satPath.containsAll(currentPathContents)) {
166  earlyTermination = true;
167  break;
168  }
169  }
170 
171  if (!earlyTermination) {
172  // Generate a new node - i.e. a new justification set
173 // generateSOSAxioms();
174  Set<OWLAxiom> newMUPS = getSOSForIncosistentClass(getCurrentClass());
175 
176  if (!newMUPS.isEmpty()) {
177  // We have a new justification set, and a new node
178  if (!allMups.contains(newMUPS)) {
179  // Entirely new justification set
180  allMups.add(newMUPS);
181  constructHittingSetTree(newMUPS, allMups, satPaths, currentPathContents);
182  }
183  }
184  else {
185  // End of current path - add it to the list of paths
186  satPaths.add(new HashSet<OWLAxiom>(currentPathContents));
187  }
188  }
189  // Back track - go one level up the tree and run for the next axiom
190  currentPathContents.remove(axiom);
191  // Done with the axiom that was removed. Add it back in
192  owlOntologyManager.applyChanges(Arrays.asList(new AddAxiom(ontology, axiom)));
193  }
194  }
