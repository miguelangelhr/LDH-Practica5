121  public void parse(InputSource source, RDFConsumer consumer) throws SAXException,
122  IOException {
123  String systemID = source.getSystemId();
124  try {
125  m_documentLocator = s_nullDocumentLocator;
126  if (systemID != null) {
127  m_baseIRI = IRI.create(new URI(source.getSystemId()));
128  } else {
129  throw new SAXException(
130  "Supplied InputSource object myst have systemId property set, which is needed for IRI resolution.");
131  }
132  m_consumer = consumer;
133  m_consumer.startModel(m_baseIRI.toString());
134  SAXParser parser = s_parserFactory.newSAXParser();
135  parser.parse(source, this);
136  m_consumer.endModel();
137  } catch (ParserConfigurationException e) {
138  throw new SAXException("Parser coniguration exception", e);
139  } catch (URISyntaxException e) {
140  throw new SAXException("Invalid SystemID '" + systemID
141  + "'of the supplied input source.");
142  } finally {
143  m_state = null;
144  m_states.clear();
145  m_documentLocator = null;
146  m_baseIRIs.clear();
147  }
148  }

224  @Override
225  public void processingInstruction(String target, String data) throws SAXException {
226  if ("include-rdf".equals(target)) {
227  Map<String, String> arguments = parseStringArguments(data);
228  if (arguments.size() > 2) {
229  throw new RDFParserException(
230  "Incorrect number of arguments for 'include-rdf' processing instruction.",
231  m_documentLocator);
232  }
233  String logicalIRI = arguments.get("logicalIRI");
234  String physicalIRI = arguments.get("physicalIRI");
235  if (physicalIRI != null) {
236  physicalIRI = resolveIRI(physicalIRI);
237  }
238  m_consumer.includeModel(logicalIRI, physicalIRI);
239  } else if ("model-attribute".equals(target)) {
240  Map<String, String> arguments = parseStringArguments(data);
241  if (arguments.size() != 2) {
242  throw new RDFParserException(
243  "Incorrect number of arguments for 'model-attribute' processing instruction.",
244  m_documentLocator);
245  }
246  String key = arguments.get("key");
247  if (key == null) {
248  throw new RDFParserException(
249  "Mising the 'key' argument for 'model-attribute' processing instruction.",
250  m_documentLocator);
251  }
252  String value = arguments.get("value");
253  if (value == null) {
254  throw new RDFParserException(
255  "Mising the 'value' argument for 'model-attribute' processing instruction.",
256  m_documentLocator);
257  }
258  m_consumer.addModelAttribte(key, value);
259  }
260  }

301  private IRI resolveFromDelegate(IRI iri, String value) {
302  if (NodeID.isAnonymousNodeIRI(value)) {
303  return IRI.create(value, null);
304  }
305  // cache the delegate URI if not there already
306  if (!m_baseURICache.containsKey(m_baseIRI)) {
307  m_baseURICache.put(m_baseIRI, m_baseIRI.toURI());
308  }
309  // get hold of the delegate URI
310  URI delegateURI = m_baseURICache.get(m_baseIRI);
311  // resolve against delegate
312  return IRI.create(delegateURI.resolve(value));
313  }

355  protected String resolveIRI(String uri) throws SAXException {
356  if (uri.length() == 0) {
357  // MH - Fix for resolving a "This document" reference against base
358  // IRIs.
359  // XXX namespace?
360  String base = m_baseIRI.toString();
361  int hashIndex = base.indexOf("#");
362  if (hashIndex != -1) {
363  return base.substring(0, hashIndex);
364  } else {
365  return base;
366  }
367  } else {
368  try {
369  String resolved = resolvedIRIs.get(uri);
370  if (resolved != null) {
371  return resolved;
372  } else {
373  IRI theIRI = resolveFromDelegate(m_baseIRI, uri);
374  String u = theIRI.toString();
375  uriCache.put(u, theIRI);
376  resolvedIRIs.put(uri, u);
377  return u;
378  }
379  } catch (IllegalArgumentException e) {
380  RDFParserException exception = new RDFParserException("IRI '" + uri
381  + "' cannot be resolved against curent base IRI "
382  + m_baseIRI.toString(), m_documentLocator);
383  exception.initCause(e);
384  throw exception;
385  }
386  }

407  String result = null;

409  if (value != null) {

417  m_documentLocator);

514  protected void propertyAttributes(String subjectIRI, Attributes atts,
515  ReificationManager reificationManager) throws SAXException {
516  int length = atts.getLength();
517  for (int i = 0; i < length; i++) {
518  String nsIRI = atts.getURI(i);
519  String localName = atts.getLocalName(i);
520  if (!XMLNS.equals(nsIRI)
521  && !XMLLANG.equals(localName)
522  && !(RDFNS.equals(nsIRI) && (ATTR_ID.equals(localName)
523  || ATTR_NODE_ID.equals(localName)
524  || ATTR_ABOUT.equals(localName) || ELT_TYPE.equals(localName)
525  || ATTR_RESOURCE.equals(localName)
526  || ATTR_PARSE_TYPE.equals(localName)
527  || ATTR_ABOUT_EACH.equals(localName)
528  || ATTR_ABOUT_EACH_PREFIX.equals(localName) || ATTR_BAG_ID
529  .equals(localName)))) {
530  String value = atts.getValue(i);
531  String reificationID = reificationManager.getReificationID(null);
532  statementWithLiteralValue(subjectIRI, nsIRI + localName, value, null,
533  reificationID);
534  } else if (RDFNS.equals(nsIRI) && ELT_TYPE.equals(localName)) {
535  String value = resolveIRI(atts.getValue(i));
536  String reificationID = reificationManager.getReificationID(null);
537  statementWithResourceValue(subjectIRI, nsIRI + localName, value,
538  reificationID);
539  }
540  }

602  protected Map<String, String> parseStringArguments(String string) throws SAXException {
603  try {
604  StreamTokenizer tokenizer = new StreamTokenizer(new StringReader(string));
605  Map<String, String> result = new HashMap<String, String>();
606  tokenizer.nextToken();
607  while (tokenizer.ttype != StreamTokenizer.TT_EOF) {
608  if (tokenizer.ttype != StreamTokenizer.TT_WORD) {
609  throw new RDFParserException(
610  "Invalid processing instruction argument.", m_documentLocator);
611  }
612  String name = tokenizer.sval;
613  if ('=' != tokenizer.nextToken()) {
614  throw new RDFParserException("Expecting token =", m_documentLocator);
615  }
616  tokenizer.nextToken();
617  if (tokenizer.ttype != '\"' && tokenizer.ttype != '\'') {
618  throw new RDFParserException(
619  "Invalid processing instruction argument.", m_documentLocator);
620  }
621  String value = tokenizer.sval;
622  result.put(name, value);
623  tokenizer.nextToken();
624  }
625  return result;
626  } catch (IOException e) {
627  RDFParserException exception = new RDFParserException("I/O error",
628  m_documentLocator);
629  exception.initCause(e);
630  throw exception;
631  }
632  }
