116  protected InputStream getInputStream(IRI documentIRI,
117  OWLOntologyLoaderConfiguration config) throws IOException {
118  String requestType = getRequestTypes();
119  URL originalURL = documentIRI.toURI().toURL();
120  String originalProtocol = originalURL.getProtocol();
121  URLConnection conn = originalURL.openConnection();
122  conn.addRequestProperty("Accept", requestType);
123  if (IOProperties.getInstance().isConnectionAcceptHTTPCompression()) {
124  conn.setRequestProperty("Accept-Encoding", "gzip, deflate");
125  }
126  conn.setConnectTimeout(IOProperties.getInstance().getConnectionTimeout());
127  if (conn instanceof HttpURLConnection && config.isFollowRedirects()) {
128  // follow redirects to HTTPS
129  HttpURLConnection con = (HttpURLConnection) conn;
130  con.connect();
131  int responseCode = con.getResponseCode();
132  // redirect
133  if (responseCode == HttpURLConnection.HTTP_MOVED_TEMP
134  || responseCode == HttpURLConnection.HTTP_MOVED_PERM
135  || responseCode == HttpURLConnection.HTTP_SEE_OTHER) {
136  String location = con.getHeaderField("Location");
137  URL newURL = new URL(location);
138  String newProtocol = newURL.getProtocol();
139  if (!originalProtocol.equals(newProtocol)) {
140  // then different protocols: redirect won't follow
141  // automatically
142  conn = newURL.openConnection();
143  conn.addRequestProperty("Accept", requestType);
144  if (IOProperties.getInstance().isConnectionAcceptHTTPCompression()) {
145  conn.setRequestProperty("Accept-Encoding", "gzip, deflate");
146  }
147  conn.setConnectTimeout(IOProperties.getInstance()
148  .getConnectionTimeout());
149  }
150  }
151  }
152  String contentEncoding = conn.getContentEncoding();
153  InputStream is = getInputStreamFromContentEncoding(conn, contentEncoding);
154  if (isZipName(documentIRI, conn)) {
155  ZipInputStream zis = new ZipInputStream(is);
156  ZipEntry entry = zis.getNextEntry();
157  while (!couldBeOntology(entry)) {
158  ZipEntry nextEntry = zis.getNextEntry();
159  if (nextEntry != null) {
160  entry = nextEntry;
161  } else {
162  break;
163  }
164  }
165  is = new BufferedInputStream(zis);
166  }
167  return is;
168  }

170  private boolean couldBeOntology(ZipEntry zipEntry) {
171  String name = zipEntry.getName();
172  Matcher matcher = ZIP_ENTRY_ONTOLOGY_NAME_PATTERN.matcher(name);
173  return matcher.matches();
174  }

176  private InputStream getInputStreamFromContentEncoding(URLConnection conn,
177  String contentEncoding) throws IOException {
178  InputStream is;
179  if ("gzip".equals(contentEncoding)) {
180  logger.fine("URL connection input stream is compressed using gzip");
181  is = new BufferedInputStream(new GZIPInputStream(conn.getInputStream()));
182  } else if ("deflate".equals(contentEncoding)) {
183  logger.fine("URL connection input stream is compressed using deflate");
184  is = new BufferedInputStream(new InflaterInputStream(conn.getInputStream(),
185  new Inflater(true)));
186  } else {
187  is = new BufferedInputStream(conn.getInputStream());
188  }
189  return is;
190  }

192  private boolean isZipName(IRI documentIRI, URLConnection connection) {
193  if (isZipFileName(documentIRI.toString())) {
194  return true;
195  } else {
196  String fileName = getFileNameFromContentDisposition(connection);
197  return fileName != null && isZipFileName(fileName);
198  }
199  }
200 
201  private String getFileNameFromContentDisposition(URLConnection connection) {
202  String contentDispositionHeaderValue = connection
203  .getHeaderField(CONTENT_DISPOSITION_HEADER);
204  if (contentDispositionHeaderValue != null) {
205  Matcher matcher = CONTENT_DISPOSITION_FILE_NAME_PATTERN
206  .matcher(contentDispositionHeaderValue);
207  if (matcher.matches()) {
208  return matcher.group(CONTENT_DISPOSITION_FILE_NAME_PATTERN_GROUP);
209  }
210  }
211  return null;
212  }
213 
214  private boolean isZipFileName(String fileName) {
215  return fileName.toLowerCase(Locale.getDefault()).endsWith(ZIP_FILE_EXTENSION);
216  }

218  protected InputSource getInputSource(OWLOntologyDocumentSource documentSource,
219  OWLOntologyLoaderConfiguration config)
220  throws IOException {
221  InputSource is;
222  if (documentSource.isReaderAvailable()) {
223  is = new InputSource(documentSource.getReader());
224  } else if (documentSource.isInputStreamAvailable()) {
225  is = new InputSource(documentSource.getInputStream());
226  } else {
227  is = new InputSource(getInputStream(documentSource.getDocumentIRI(), config));
228  }
229  is.setSystemId(documentSource.getDocumentIRI().toString());
230  return is;
231  }
