82  public void addClosureAxioms(OWLClass clazz) {
83  /* Get the class axioms */
84  Set<OWLSubClassOfAxiom> axioms = ontology.getAxioms(AxiomType.SUBCLASS_OF);
85  /* Collect those that assert superclasses of the class */
86  SubClassCollector collector = new SubClassCollector(clazz);
87  for (OWLClassAxiom axiom : axioms) {
88  axiom.accept(collector);
89  }
90  Map<OWLObjectPropertyExpression, Set<OWLClassExpression>> restrictions = new HashMap<OWLObjectPropertyExpression, Set<OWLClassExpression>>();
91  /* For each axiom.... */
92  for (OWLSubClassOfAxiom axiom : collector.getAxioms()) {
93  /* Get the superclass */
94  OWLClassExpression superClass = axiom.getSuperClass();
95  /* Collect any existentials */
96  ExistentialCollector ec = new ExistentialCollector(restrictions);
97  superClass.accept(ec);
98  }
99  /* For any existentials.... */
100  for (OWLObjectPropertyExpression prop : restrictions.keySet()) {
101  System.out.println("prop: " + prop);
102  Set<OWLClassExpression> fillers = restrictions.get(prop);
103  for (OWLClassExpression filler : fillers) {
104  System.out.println("------> " + filler);
105  }
106  /* Create a union of the fillers */
107  OWLClassExpression union = factory.getOWLObjectUnionOf(fillers);
108  /* Create a universal restriction */
109  OWLClassExpression universal = factory.getOWLObjectAllValuesFrom(prop, union);
110  /* Create a new axiom */
111  OWLAxiom newAxiom = factory.getOWLSubClassOfAxiom(clazz, universal);
112  /* Now add the axiom to the ontology */
113  AddAxiom addAxiom = new AddAxiom(ontology, newAxiom);
114  /* Use the manager to apply the change */
115  manager.applyChange(addAxiom);
116  }
117  }
