126  private void ExpandBuff(boolean wrapAround) {
127  char[] newbuffer = new char[bufsize + 2048];
128  int newbufline[] = new int[bufsize + 2048];
129  int newbufcolumn[] = new int[bufsize + 2048];
130  if (wrapAround) {
131  System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
132  System.arraycopy(buffer, 0, newbuffer, bufsize - tokenBegin, bufpos);
133  buffer = newbuffer;
134  System.arraycopy(bufline, tokenBegin, newbufline, 0, bufsize - tokenBegin);
135  System.arraycopy(bufline, 0, newbufline, bufsize - tokenBegin, bufpos);
136  bufline = newbufline;
137  System.arraycopy(bufcolumn, tokenBegin, newbufcolumn, 0, bufsize
138  - tokenBegin);
139  System.arraycopy(bufcolumn, 0, newbufcolumn, bufsize - tokenBegin, bufpos);
140  bufcolumn = newbufcolumn;
141  bufpos += bufsize - tokenBegin;
142  } else {
143  System.arraycopy(buffer, tokenBegin, newbuffer, 0, bufsize - tokenBegin);
144  buffer = newbuffer;

153  available = bufsize += 2048;
154  tokenBegin = 0;

157  private void FillBuff() throws IOException {
158  int i;
159  if (maxNextCharInd == 4096) {
160  maxNextCharInd = nextCharInd = 0;
161  }
162  try {
163  if ((i = inputStream.read(nextCharBuf, maxNextCharInd, 4096 - maxNextCharInd)) == -1) {
164  inputStream.close();
165  throw new IOException();
166  } else {
167  maxNextCharInd += i;
168  }
169  // this treats the BOM problem; when the parser is regenerated,
170  // this needs to be applied again
171  if (beginning && nextCharBuf[0] == '\uFEFF') {
172  nextCharInd++;
173  beginning = false;
174  }
175  return;
176  } catch (IOException e) {
177  if (bufpos != 0) {
178  --bufpos;
179  backup(0);
180  } else {
181  bufline[bufpos] = line;
182  bufcolumn[bufpos] = column;
183  }
184  throw e;
185  }
186  }

211  private void AdjustBuffSize() {
212  if (available == bufsize) {
213  if (tokenBegin > 2048) {
214  bufpos = 0;

228  private void UpdateLineColumn(char c) {
229  column++;
230  if (prevCharIsLF) {
231  prevCharIsLF = false;
232  line += column = 1;
233  } else if (prevCharIsCR) {
234  prevCharIsCR = false;
235  if (c == '\n') {
236  prevCharIsLF = true;
237  } else {
238  line += column = 1;
239  }
240  }
241  switch (c) {
242  case '\r':
243  prevCharIsCR = true;
244  break;
245  case '\n':
246  prevCharIsLF = true;
247  break;
248  case '\t':
249  column--;
250  column += tabSize - column % tabSize;
251  break;
252  default:
253  break;
254  }
255  bufline[bufpos] = line;
256  bufcolumn[bufpos] = column;
257  }

263  public char readChar() throws IOException {
264  if (inBuf > 0) {
265  --inBuf;
266  if (++bufpos == bufsize) {
267  bufpos = 0;
268  }
269  return buffer[bufpos];
270  }
271  char c;
272  if (++bufpos == available) {
273  AdjustBuffSize();
274  }
275  if ((buffer[bufpos] = c = ReadByte()) == '\\') {
276  UpdateLineColumn(c);
277  int backSlashCnt = 1;
278  for (;;) // Read all the backslashes
279  {
280  if (++bufpos == available) {
281  AdjustBuffSize();
282  }
283  try {
284  if ((buffer[bufpos] = c = ReadByte()) != '\\') {
285  UpdateLineColumn(c);
286  // found a non-backslash char.
287  if (c == 'u' && (backSlashCnt & 1) == 1) {
288  if (--bufpos < 0) {
289  bufpos = bufsize - 1;
290  }
291  break;
292  }
293  backup(backSlashCnt);
294  return '\\';
295  }
296  } catch (IOException e) {
297  // We are returning one backslash so we should only backup
298  // (count-1)
299  if (backSlashCnt > 1) {
300  backup(backSlashCnt - 1);
301  }
302  return '\\';
303  }
304  UpdateLineColumn(c);
305  backSlashCnt++;
306  }
307  // Here, we have seen an odd number of backslash's followed by a 'u'
308  try {
309  while ((c = ReadByte()) == 'u') {
310  ++column;
311  }
312  buffer[bufpos] = c = (char) (hexval(c) << 12 | hexval(ReadByte()) << 8
313  | hexval(ReadByte()) << 4 | hexval(ReadByte()));
314  column += 4;
315  } catch (IOException e) {
316  throw new OWLRuntimeException("Invalid escape character at line " + line
317  + " column " + column + ".", e);
318  }
319  if (backSlashCnt == 1) {
320  return c;
321  } else {
322  backup(backSlashCnt - 1);
323  return '\\';
324  }
325  } else {
326  UpdateLineColumn(c);
327  return c;
328  }
329  }
