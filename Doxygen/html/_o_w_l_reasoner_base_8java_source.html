91  protected OWLReasonerBase(OWLOntology rootOntology, OWLReasonerConfiguration configuration, BufferingMode bufferingMode) {
92  this.rootOntology = rootOntology;
93  this.bufferingMode = bufferingMode;
94  this.configuration = configuration;
95  timeOut = configuration.getTimeOut();
96  manager = rootOntology.getOWLOntologyManager();
97  manager.addOntologyChangeListener(ontologyChangeListener);
98  reasonerAxioms = new HashSet<OWLAxiom>();
99  for (OWLOntology ont : rootOntology.getImportsClosure()) {
100  for (OWLAxiom ax : ont.getLogicalAxioms()) {
101  reasonerAxioms.add(ax.getAxiomWithoutAnnotations());
102  }
103  for (OWLAxiom ax : ont.getAxioms(AxiomType.DECLARATION)) {
104  reasonerAxioms.add(ax.getAxiomWithoutAnnotations());
105  }
106  }
107  }

168  @Override
169  public void flush() {
170  // Process the changes
171  final Set<OWLAxiom> added = new HashSet<OWLAxiom>();
172  final Set<OWLAxiom> removed = new HashSet<OWLAxiom>();
173  computeDiff(added, removed);
174  reasonerAxioms.removeAll(removed);
175  reasonerAxioms.addAll(added);
176  rawChanges.clear();
177  if (!added.isEmpty() || !removed.isEmpty()) {
178  handleChanges(added, removed);
179  }
180  }

191  private void computeDiff(Set<OWLAxiom> added, Set<OWLAxiom> removed) {
192  if (rawChanges.isEmpty()) {
193  return;
194  }
195  for (OWLOntology ont : rootOntology.getImportsClosure()) {
196  for (OWLAxiom ax : ont.getLogicalAxioms()) {
197  if (!reasonerAxioms.contains(ax.getAxiomWithoutAnnotations())) {
198  added.add(ax);
199  }
200  }
201  for (OWLAxiom ax : ont.getAxioms(AxiomType.DECLARATION)) {
202  if (!reasonerAxioms.contains(ax.getAxiomWithoutAnnotations())) {
203  added.add(ax);
204  }
205  }
206  }
207  for (OWLAxiom ax : reasonerAxioms) {
208  if (!rootOntology.containsAxiomIgnoreAnnotations(ax, true)) {
209  removed.add(ax);
210  }
211  }
212  }
