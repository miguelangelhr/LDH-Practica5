72  @Override
73  public void visit(OWLAnnotation annotation) {
74  /*
75  * If it's a label, grab it as the result. Note that if there are
76  * multiple labels, the last one will be used.
77  */
78  if (annotation.getProperty().isLabel()) {
79  OWLLiteral c = (OWLLiteral) annotation.getValue();
80  result = c.getLiteral();
81  }
82  }
