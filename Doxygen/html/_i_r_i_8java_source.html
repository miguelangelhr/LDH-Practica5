77  public boolean isAbsolute() {
78  int colonIndex = prefix.indexOf(':');
79  if (colonIndex == -1) {
80  return false;
81  }
82  for (int i = 0; i < colonIndex; i++) {
83  char ch = prefix.charAt(i);
84  if (!Character.isLetter(ch) && !Character.isDigit(ch) && ch != '.' && ch != '+' && ch != '-') {
85  return false;
86  }
87  }
88  return true;
89  }

219  public static IRI create(String prefix, String suffix) {
220  if(prefix == null) {
221  return create(suffix);
222  } else if(suffix == null) {
223  // suffix set deliberately to null is used only in blank node
224  // management
225  // this is not great but blank nodes should be changed to not refer
226  // to IRIs at all
227  // XXX address blank node issues with iris
228  return create(prefix);
229  } else {
230  int index = XMLUtils.getNCNameSuffixIndex(prefix);
231  int test = XMLUtils.getNCNameSuffixIndex(suffix);
232  if (index == -1 && test == 0) {
233  // the prefix does not contain an ncname character and there is
234  // no illegal character in the suffix
235  // the split is therefore correct
236  return new IRI(prefix, suffix);
237  }
238  // otherwise the split is wrong; we could obtain the right split by
239  // using index and test, but it's just as easy to use the other
240  // constructor
241  return create(prefix + suffix);
242  }
243  }
