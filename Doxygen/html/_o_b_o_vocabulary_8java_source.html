187  public static IRI ID2IRI(String oboId, IDSpaceManager idSpaceManager) {
188  Matcher matcher = OBO_ID_PATTERN.matcher(oboId);
189  if (matcher.matches()) {
190  String idSpace = matcher.group(2);
191  String localId = matcher.group(3);
192  StringBuilder sb = new StringBuilder();
193  String iriPrefix = idSpaceManager.getIRIPrefix(idSpace);
194  sb.append(iriPrefix);
195  if (idSpace != null) {
196  sb.append(idSpace);
197  sb.append("_");
198  }
199  sb.append(localId);
200  return IRI.create(sb.toString());
201  } else {
202  return IRI.create(oboId);
203  }
204  }
205 // Format of Foundry-compliant URIs
206 
207  // FOUNDRY_OBO_URI ::= "http://purl.obolibrary.org/obo/" IDSPACE "_" LOCALID
208  // Format of OBO legacy URIs
209  // Those are found in documents that were natively authored using the OBO
210  // format and which were converted using the NCBOoboInOwl script before this
211  // policy was put in place.
212  // LEGACY_OBO_URI ::= "http://purl.org/obo/owl/" IDSPACE "#" IDSPACE "_"
213  // LOCALID
214  public static String IRI2ID(IRI oboIRI) {
215  Matcher matcher = OBO_IRI_PATTERN.matcher(oboIRI.toString());
216  if (matcher.matches()) {
217  String idSpace = matcher.group(3);
218  String localId = matcher.group(4);
219  StringBuilder sb = new StringBuilder();
220  if (idSpace != null) {
221  sb.append(idSpace);
222  sb.append(":");
223  }
224  sb.append(localId);
225  return sb.toString();
226  } else {
227  throw new RuntimeException("Not an OBO IRI");
228  }
229  }
