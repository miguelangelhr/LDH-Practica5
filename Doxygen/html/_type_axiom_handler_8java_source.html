101  @Override
102  public void handleTriple(IRI subject, IRI predicate, IRI object) throws UnloadableImportException {
103 
104 
105 
106  IRI annotatedSource = getObjectOfSourceTriple(subject);
107  IRI annotatedProperty = getObjectOfPropertyTriple(subject);
108  IRI annotatedTarget = getObjectOfTargetTriple(subject);
109  OWLLiteral annotatedTargetLiteral = null;
110  if (annotatedTarget == null) {
111  annotatedTargetLiteral = getTargetLiteral(subject);
112  }
113 
114  // check that other conditions are not invalid
115  if(annotatedSource!=null&&annotatedProperty!=null) {
116  consumeTriple(subject, predicate, object);
117  Set<OWLAnnotation> annotations = getConsumer().translateAnnotations(subject);
118  getConsumer().setPendingAnnotations(annotations);
119  if (annotatedTarget != null) {
120  getConsumer().handle(annotatedSource, annotatedProperty, annotatedTarget);
121  }
122  else if(annotatedTargetLiteral != null) {
123  getConsumer().handle(annotatedSource, annotatedProperty, annotatedTargetLiteral);
124  }
125  if (!annotations.isEmpty()) {
126  OWLAxiom ax = getConsumer().getLastAddedAxiom();
127  getConsumer().removeAxiom(ax.getAxiomWithoutAnnotations());
128  }
129  }
130  }
