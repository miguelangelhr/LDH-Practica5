89  public OWLObjectDuplicator(OWLDataFactory dataFactory, Map<IRI, IRI> iriReplacementMap) {
90  this.dataFactory = dataFactory;
91  replacementMap = new HashMap<OWLEntity, IRI>();
92  for (Map.Entry<IRI, IRI> e: iriReplacementMap.entrySet()) {
93  IRI iri=e.getKey();
94  IRI repIRI = e.getValue();
95  replacementMap.put(dataFactory.getOWLClass(iri), repIRI);
96  replacementMap.put(dataFactory.getOWLObjectProperty(iri), repIRI);
97  replacementMap.put(dataFactory.getOWLDataProperty(iri), repIRI);
98  replacementMap.put(dataFactory.getOWLNamedIndividual(iri), repIRI);
99  replacementMap.put(dataFactory.getOWLDatatype(iri), repIRI);
100  replacementMap.put(dataFactory.getOWLAnnotationProperty(iri), repIRI);
101  }
102  }
