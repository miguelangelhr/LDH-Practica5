184  @Override
185  public String getPrefixIRI(IRI iri) {
186  String ns = iri.getNamespace();
187  for (String prefixName : prefix2NamespaceMap.keySet()) {
188  String prefix = prefix2NamespaceMap.get(prefixName);
189  if (ns.equals(prefix)) {
190  String ncNameSuffix = iri.getFragment();
191  if (ncNameSuffix == null) {
192  ncNameSuffix = "";
193  }
194  return prefixName + ncNameSuffix;
195  }
196  }
197  return null;
198  }

211  @Override
212  public IRI getIRI(String curie) {
213  if (curie.startsWith("<")) {
214  return IRI.create(curie.substring(1, curie.length() - 1));
215  }
216  int sep = curie.indexOf(':');
217  if (sep == -1) {
218  if (getDefaultPrefix() != null) {
219  return IRI.create(getDefaultPrefix() + curie);
220  } else {
221  return IRI.create(curie);
222  }
223  } else {
224  String prefixName = curie.substring(0, sep + 1);
225  if (!containsPrefixMapping(prefixName)) {
226  throw new RuntimeException("Prefix not registered for prefix name: "
227  + prefixName);
228  }
229  String prefix = getPrefix(prefixName);
230  String localName = curie.substring(sep + 1);
231  return IRI.create(prefix, localName);
232  }
233  }

254  // XXX not in the interface
255  public void setPrefix(String prefixName, String prefix) {
256  if (prefix == null) {
257  throw new IllegalArgumentException("prefix cannot be null");
258  }
259  if (prefixName == null) {
260  throw new IllegalArgumentException("prefixName cannot be null");
261  }
262  if (!prefixName.endsWith(":")) {
263  throw new IllegalArgumentException("Prefix names must end with a colon (:)");
264  }
265  prefix2NamespaceMap.put(prefixName, prefix);

276  if (e.getValue().equals(namespace)) {
