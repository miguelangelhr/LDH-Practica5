185  protected boolean isNonNegativeIntegerStrict(IRI mainNode, OWLRDFVocabulary predicate) {
186  OWLLiteral literal = consumer.getLiteralObject(mainNode, predicate, false);
187  if (literal == null) {
188  return false;

191  OWL2Datatype nni = OWL2Datatype.XSD_NON_NEGATIVE_INTEGER;

303  protected boolean isResourceListStrict(IRI mainNode, TypeMatcher typeMatcher, int minSize) {
304  if (mainNode == null) {
305  return false;
306  }
307  IRI currentListNode = mainNode;
308  Set<IRI> visitedListNodes = new HashSet<IRI>();
309  int size = 0;
310  while (true) {
311  IRI firstObject = consumer.getResourceObject(currentListNode, RDF_FIRST, false);
312  if (firstObject == null) {
313  return false;
314  }
315  if (!typeMatcher.isTypeStrict(firstObject)) {
316  // Something in the list that is not of the required type
317  return false;
318  }
319  else {
320  size++;
321  }
322  IRI restObject = consumer.getResourceObject(currentListNode, RDF_REST, false);
323  if (visitedListNodes.contains(restObject)) {
324  // Cycle - Non-terminating
325  return false;
326  }
327  if (restObject == null) {
328  // Not terminated properly
329  return false;
330  }
331  if (restObject.equals(RDF_NIL.getIRI())) {
332  // Terminated properly
333  return size >= minSize;
334  }
335  // Carry on
336  visitedListNodes.add(restObject);
337  currentListNode = restObject;
338  }
339  }
