160  @Override
161  public OWLLiteral getOWLLiteral(String lexicalValue, OWLDatatype datatype) {
162  OWLLiteral literal;
163  if (datatype.isRDFPlainLiteral()) {
164  int sep = lexicalValue.lastIndexOf('@');
165  if (sep != -1) {
166  String lex = lexicalValue.substring(0, sep);
167  String lang = lexicalValue.substring(sep + 1);
168  literal = getBasicLiteral(lex, lang, getRDFPlainLiteral());
169  } else {
170  literal = getBasicLiteral(lexicalValue, datatype);
171  }
172  } else {
173  // check the four special cases
174  try {
175  if (datatype.isBoolean()) {
176  literal = getOWLLiteral(isBooleanTrueValue(lexicalValue.trim()));
177  } else if (datatype.isFloat()) {
178  if (lexicalValue.trim().equals("-0.0")) {
179  // according to some W3C test, this needs to be
180  // different from 0.0; Java floats disagree
181  if (negativeFloatZero == null) {
182  negativeFloatZero = getBasicLiteral("-0.0", XSD_FLOAT);
183  }
184  literal = negativeFloatZero;
185  } else {
186  try {
187  float f = Float.parseFloat(lexicalValue);
188  literal = getOWLLiteral(f);
189  } catch (NumberFormatException e) {
190  literal = getBasicLiteral(lexicalValue, datatype);
191  }
192  }
193  } else if (datatype.isDouble()) {
194  literal = getOWLLiteral(Double.parseDouble(lexicalValue));
195  } else if (datatype.isInteger()) {
196  // again, some W3C tests require padding zeroes to make
197  // literals different
198  if (lexicalValue.trim().charAt(0) == '0') {
199  literal = getBasicLiteral(lexicalValue, getIntegerOWLDatatype());
200  } else {
201  try {
202  // this is fine for values that can be parsed as
203  // ints - not all values are
204  literal = getOWLLiteral(Integer.parseInt(lexicalValue));
205  } catch (NumberFormatException ex) {
206  // try as a big decimal
207  literal = getBasicLiteral(lexicalValue, datatype);
208  }
209  }
210  } else {
211  literal = getBasicLiteral(lexicalValue, datatype);
212  }
213  } catch (NumberFormatException e) {
214  // some literal is malformed, i.e., wrong format
215  literal = getBasicLiteral(lexicalValue, datatype);
216  }
217  }
218  return literal;
219  }
