384  @Override
385  public final void visit(OWLOntology ontology1) {
386  reset();
387  for (final OWLClass eachClass : ontology1.getClassesInSignature()) {
388  if (ignoreDeclarations) {
389  if (ontology1.getAxioms(eachClass).size() == 1
390  && ontology1.getDeclarationAxioms(eachClass).size() == 1) {
391  continue;
392  }
393  }
394  final boolean primitive = !eachClass.isDefined(ontology1);// !eachClass.getSuperClasses(ontology).isEmpty();
395  writeOpenBracket();
396 
397 
398  if (primitive) { //there is no equivalentclasses axiom!
399  write(DEFINE_PRIMITIVE_CONCEPT);
400  write(eachClass);
401  writeSpace();
402  flatten(eachClass.getSuperClasses(ontology1), KRSSVocabulary.AND);
403  writeCloseBracket();
404  writeln();
405  for (OWLClassExpression description : eachClass
406  .getEquivalentClasses(ontology1)) {
407  writeOpenBracket();
408  write(eachClass);
409  write(EQUIVALENT);
410  write(description);
411  writeCloseBracket();
412  writeln();
413  }
414  } else {
415  writeOpenBracket();
416  write(DEFINE_CONCEPT);
417  write(eachClass);
418  flatten(eachClass.getEquivalentClasses(ontology1), KRSSVocabulary.AND);
419  writeCloseBracket();
420  writeln();
421  for (OWLClassExpression description : eachClass
422  .getSuperClasses(ontology1)) {
423  writeOpenBracket();
424  write(eachClass);
425  write(IMPLIES);
426  write(description);
427  writeCloseBracket();
428  writeln();
429  }
430  }
431  }
432  /* for (final OWLClassAxiom axiom : ontology.getGeneralClassAxioms()) {
433  axiom.accept(this);
434  }
435  */
436  for (final OWLObjectProperty property : sort(ontology1
437  .getObjectPropertiesInSignature())) {
438  if (ignoreDeclarations) {
439  if (ontology1.getAxioms(property).size() == 1
440  && ontology1.getDeclarationAxioms(property).size() == 1) {
441  continue;
442  }
443  }
444  writeOpenBracket();
445  final Set<OWLObjectPropertyExpression> properties = property
446  .getEquivalentProperties(ontology1);
447  final boolean isPrimitive = properties.isEmpty();
448 
449  if (isPrimitive) {
450  write(DEFINE_PRIMITIVE_ROLE);
451  write(property);
452  Set<OWLObjectPropertyExpression> superProperties = property
453  .getSuperProperties(ontology1);
454  if (superProperties.size() == 1) {
455  writeSpace();
456  write(PARENT_ATTR);
457  writeSpace();
458  write(superProperties.iterator().next());
459  } else if (superProperties.size() > 1) {
460  writeSpace();
461  write(PARENTS_ATTR);
462  writeSpace();
463  flattenProperties(superProperties, null);
464  } else {
465  //right/left identity?
466  //we only allow for either right or left identity axiom, otherwise it is
467  //expressed via role-inclusion axioms
468 
469  Set<OWLSubPropertyChainOfAxiom> chainAxioms = getPropertyChainSubPropertyAxiomsFor(property);
470  if (chainAxioms.size() == 1) {
471  OWLSubPropertyChainOfAxiom axiom = chainAxioms.iterator().next();
472  if (isLeftIdentityAxiom(axiom, property)) {
473  leftRightIdentityUsed.add(axiom);
474  writeSpace();
475  write(LEFTIDENTITY_ATTR);
476  write(axiom.getPropertyChain().get(0));
477 
478  } else if (isRightIdentityAxiom(axiom, property)) {
479  leftRightIdentityUsed.add(axiom);
480  writeSpace();
481  write(RIGHTIDENTITY_ATTR);
482  write(axiom.getPropertyChain().get(1));
483  }
484  }
485  }
486  } else {
487  if (properties.isEmpty()) {
488  write(DEFINE_PRIMITIVE_ROLE);
489  write(property);
490  writeSpace();
491  } else {
492  write(DEFINE_ROLE);
493  write(property);
494  OWLObjectPropertyExpression expr = properties.iterator().next();
495  write(expr);
496  properties.remove(expr);
497  writeSpace();
498  }
499  }
500 
501  if (property.isTransitive(ontology1)) {
502  writeSpace();
503  write(TRANSITIVE_ATTR);
504  writeSpace();
505  write(TRUE);
506  }
507  if (property.isSymmetric(ontology1)) {
508  writeSpace();
509  write(SYMMETRIC_ATTR);
510  writeSpace();
511  write(TRUE);
512  }
513  if (property.isReflexive(ontology1)) {
514  writeSpace();
515  write(REFLEXIVE_ATTR);
516  writeSpace();
517  write(TRUE);
518  }
519  final Iterator<OWLObjectPropertyExpression> inverses = property.getInverses(
520  ontology1).iterator();
521  if (inverses.hasNext()) {
522  writeSpace();
523  write(INVERSE_ATTR);
524  write(inverses.next());
525  }
526  Set<OWLClassExpression> desc = property.getDomains(ontology1);
527  if (!desc.isEmpty()) {
528  writeSpace();
529  write(DOMAIN_ATTR);
530  flatten(desc, KRSSVocabulary.AND);
531  }
532  desc = property.getRanges(ontology1);
533  if (!desc.isEmpty()) {
534  writeSpace();
535  write(RANGE_ATTR);
536  flatten(desc, KRSSVocabulary.AND);
537  }
538  writeCloseBracket();
539  writeln();
540  while (inverses.hasNext()) {
541  writeOpenBracket();
542  write(INVERSE);
543  write(property);
544  write(inverses.next());
545  writeOpenBracket();
546  writeln();
547  }
548  for (OWLObjectPropertyExpression expr : properties) {
549  writeOpenBracket();
550  write(ROLES_EQUIVALENT);
551  write(property);
552  write(expr);
553  writeCloseBracket();
554  writeln();
555  }
556  }
557  for (final OWLNamedIndividual individual : sort(ontology1
558  .getIndividualsInSignature())) {
559  if (ignoreDeclarations) {
560  if (ontology1.getAxioms(individual).size() == 1
561  && ontology1.getDeclarationAxioms(individual).size() == 1)
562  {
563  continue;
564  }
565  }
566  writeOpenBracket();
567  write(DEFINE_INDIVIDUAL);
568  write(individual);
569  writeCloseBracket();
570  writeln();
571  }
572  for (final OWLAxiom axiom : ontology1.getAxioms()) {
573  axiom.accept(this);
574  }
575  try {
576  writer.flush();
577  } catch (IOException io) {
578  io.printStackTrace();
579  }
580

851  protected boolean isLeftIdentityAxiom(OWLSubPropertyChainOfAxiom axiom, OWLObjectProperty property) {
852  if (axiom.getSuperProperty().equals(property)) {
853  Iterator<OWLObjectPropertyExpression> chain = axiom.getPropertyChain().iterator();
854  if (chain.hasNext()) {
855  if (chain.next() instanceof OWLObjectProperty) {
856  if (chain.hasNext() && chain.next().equals(property)) {
857  return !chain.hasNext();
858  }
859  }
860  }
861  }
862  return false;
863  }
864 
865  protected boolean isRightIdentityAxiom(OWLSubPropertyChainOfAxiom axiom, OWLObjectProperty property) {
866  if (axiom.getSuperProperty().equals(property)) {
867  Iterator<OWLObjectPropertyExpression> chain = axiom.getPropertyChain().iterator();
868  if (chain.hasNext()) {
869  if (chain.next().equals(property)) {
870  if (chain.hasNext()) {
871  chain.next();
872  return !chain.hasNext();
873  }
874  }
875  }
876  }
877  return false;
878  }

880  protected Set<OWLSubPropertyChainOfAxiom> getPropertyChainSubPropertyAxiomsFor(OWLPropertyExpression<?,?> property) {
881  Set<OWLSubPropertyChainOfAxiom> axioms = new HashSet<OWLSubPropertyChainOfAxiom>();
882  for (OWLSubPropertyChainOfAxiom axiom : ontology.getAxioms(AxiomType.SUB_PROPERTY_CHAIN_OF)) {
883  if (axiom.getSuperProperty().equals(property)) {
884  axioms.add(axiom);
885  }
886  }
