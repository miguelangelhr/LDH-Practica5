140  public void fillOntology(OWLOntologyManager manager, OWLOntology ontology) throws OWLOntologyChangeException {
141  List<OWLOntologyChange> changes = new ArrayList<OWLOntologyChange>();
142  for (InferredAxiomGenerator<? extends OWLAxiom> axiomGenerator : axiomGenerators) {
143  for (OWLAxiom ax : axiomGenerator.createAxioms(manager, reasoner)) {
144  changes.add(new AddAxiom(ontology, ax));
145  }
146  }
147  manager.applyChanges(changes);
