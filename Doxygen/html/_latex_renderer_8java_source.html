88  private void writeEntitySection(OWLEntity entity, LatexWriter w) {
89  w.write("\\subsubsection*{");
90  w.write(escapeName(shortFormProvider.getShortForm(entity)));

98  @Override
99  public void render(OWLOntology ontology, Writer writer) throws OWLRendererException {
100  try {
101 
102 
103  LatexWriter w = new LatexWriter(writer);
104  w.write("\\documentclass{article}\n");
105  w.write("\\parskip 0pt\n");
106  w.write("\\parindent 0pt\n");
107  w.write("\\oddsidemargin 0cm\n");
108  w.write("\\textwidth 19cm\n");
109 
110  w.write("\\begin{document}\n\n");
111 
112  LatexObjectVisitor renderer = new LatexObjectVisitor(w, ontology
113  .getOWLOntologyManager().getOWLDataFactory());
114 
115  Collection<OWLClass> clses = sortEntities(ontology.getClassesInSignature());
116  if (!clses.isEmpty()) {
117  w.write("\\subsection*{Classes}\n\n");
118  }
119  for (OWLClass cls : clses) {
120  writeEntitySection(cls, w);
121  for (OWLAxiom ax : sortAxioms(ontology.getAxioms(cls))) {
122  renderer.setSubject(cls);
123  ax.accept(renderer);
124  w.write("\n\n");
125  }
126  }
127 
128  w.write("\\section*{Object properties}");
129 
130  for (OWLObjectProperty prop : sortEntities(ontology.getObjectPropertiesInSignature())) {
131  writeEntitySection(prop, w);
132  for (OWLAxiom ax : sortAxioms(ontology.getAxioms(prop))) {
133  ax.accept(renderer);
134  w.write("\n\n");
135  }
136  }
137 
138  w.write("\\section*{Data properties}");
139 
140  for (OWLDataProperty prop : sortEntities(ontology.getDataPropertiesInSignature())) {
141  writeEntitySection(prop, w);
142  for (OWLAxiom ax : sortAxioms(ontology.getAxioms(prop))) {
143  ax.accept(renderer);
144  w.write("\n\n");
145  }
146  }
147 
148  w.write("\\section*{Individuals}");
149 
150  for (OWLNamedIndividual ind : sortEntities(ontology.getIndividualsInSignature())) {
151  writeEntitySection(ind, w);
152  for (OWLAxiom ax : sortAxioms(ontology.getAxioms(ind))) {
153  ax.accept(renderer);
154  w.write("\n\n");
155  }
156  }
157 
158  w.write("\\section*{Datatypes}");
159  for(OWLDatatype type:sortEntities(ontology.getDatatypesInSignature())) {
160  writeEntitySection(type, w);
161  for(OWLAxiom ax:sortAxioms(ontology.getAxioms(type))) {
162  ax.accept(renderer);
163  w.write("\n\n");
164  }
165  }
166  writer.write("\\end{document}\n");
167  writer.flush();
168  }
169  catch (IOException e) {
170  throw new LatexRendererIOException(e);
171  }
172  }

174  private <T extends OWLEntity> Collection<T> sortEntities(Set<T> entites) {
175  List<T> list = new ArrayList<T>(entites);
176  OWLEntityComparator entityComparator = new OWLEntityComparator(shortFormProvider);
177  Collections.sort(list, entityComparator);
178  return list;
179  }
180 
181  private Collection<OWLAxiom> sortAxioms(Set<? extends OWLAxiom> axioms) {
182  List<OWLAxiom> list = new ArrayList<OWLAxiom>(axioms);
183  Collections.sort(list, new OWLAxiomComparator());
184  return list;
185  }

195  public int compare(OWLAxiom o1, OWLAxiom o2) {
