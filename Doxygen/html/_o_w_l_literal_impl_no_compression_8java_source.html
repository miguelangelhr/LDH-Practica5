86  public OWLLiteralImplNoCompression(byte[] bytes, String lang, OWLDatatype datatype) {
87  super();
88  literal = new byte[bytes.length];
89  System.arraycopy(bytes, 0, literal, 0, bytes.length);
90  OWLDatatype rdfplainlit = RDF_PLAIN_LITERAL;
91  if (lang == null || lang.length() == 0) {
92  this.lang = "";
93  if (datatype == null) {
94  this.datatype = rdfplainlit;
95  } else {
96  this.datatype = datatype;
97  }
98  } else {
99  if (datatype != null && !datatype.isRDFPlainLiteral()) {
100  // ERROR: attempting to build a literal with a language tag and
101  // type different from plain literal
102  throw new OWLRuntimeException("Error: cannot build a literal with type: "
103  + datatype.getIRI() + " and language: " + lang);
104  }
105  this.lang = lang;
106  this.datatype = rdfplainlit;
107  }
108  hashcode = getHashCode();
109  }

227  private int getHashCode() {
228  int hashCode = 277;
229  hashCode = hashCode * 37 + getDatatype().hashCode();
230  hashCode = hashCode * 37;
231  try {
232  if (isInteger()) {
233  hashCode += parseInteger() * 65536;
234  } else if (isDouble()) {
235  hashCode += (int) parseDouble() * 65536;
236  } else if (isFloat()) {
237  hashCode += (int) parseFloat() * 65536;
238  } else if (isBoolean()) {
239  hashCode += parseBoolean() ? 65536 : 0;
240  } else {
241  hashCode += getLiteral().hashCode() * 65536;
242  }
243  } catch (NumberFormatException e) {
244  // it is possible that a literal does not have a value that's valid
245  // for its datatype; not very useful for a consistent ontology but
246  // some W3C reasoner tests use them
247  hashCode += getLiteral().hashCode() * 65536;
248  }
249  if (hasLang()) {
250  hashCode = hashCode * 37 + getLang().hashCode();
251  }
252  return hashCode;
253  }
