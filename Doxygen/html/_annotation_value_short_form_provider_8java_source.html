176  @Override
177  public String getShortForm(OWLEntity entity) {
178 
179  for (OWLAnnotationProperty prop : annotationProperties) {
180  // visit the properties in order of preference
181  AnnotationLanguageFilter checker = new AnnotationLanguageFilter(prop, preferredLanguageMap.get(prop));
182 
183  for (OWLOntology ontology : ontologySetProvider.getOntologies()) {
184  for (OWLAnnotationAssertionAxiom ax : entity.getAnnotationAssertionAxioms(ontology)) {
185  ax.accept(checker);
186  }
187  }
188  if (checker.getMatch() != null) {
189  return getRendering(checker.getMatch());
190  }
191  }
192 
193  return alternateShortFormProvider.getShortForm(entity);
194  }

264  @Override
265  public void visit(OWLAnnotationAssertionAxiom anno) {
266  if (lastLangMatchIndex > 0 &&
267  anno.getProperty().equals(prop)) {
268  // a perfect match - no need to carry on search
269  anno.getValue().accept(this);
270  }

273  @Override
274  public void visit(OWLLiteral node) {
275  if (preferredLanguages == null || preferredLanguages.isEmpty()) {
276  // if there are no languages just match the first thing
277  lastLangMatchIndex = 0;
278  candidateValue = node;
279  }
280  else {
281  final int index = preferredLanguages.indexOf(node.getLang());
282  if (index >= 0 && index < lastLangMatchIndex) {
283  lastLangMatchIndex = index;
284  candidateValue = node;
285  }
286  }
287  }
