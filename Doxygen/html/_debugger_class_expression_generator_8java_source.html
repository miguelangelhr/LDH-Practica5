117  // A and not (B)

190  public void visit(OWLSubObjectPropertyOfAxiom axiom) {
191  // subProp some {a} subClassOf supProp some {a}
192  OWLIndividual ind = dataFactory.getOWLNamedIndividual(IRI
193 .create("http://debugger.com#", "A" + System.nanoTime()));
194  OWLClassExpression sub = dataFactory.getOWLObjectHasValue(axiom.getSubProperty(),
195  ind);

197  axiom.getSuperProperty(), ind);
198  OWLAxiom ax = dataFactory.getOWLSubClassOfAxiom(sub, sup);
199  ax.accept(this);

231  @Override
232  public void visit(OWLEquivalentClassesAxiom axiom) {
233  if (axiom.getClassExpressions().size() == 2
234  && axiom.getClassExpressions().contains(dataFactory.getOWLNothing())) {
235  for (OWLClassExpression c : axiom.getClassExpressions()) {
236  if (!c.isOWLNothing()) {
237  desc = c;
238  return;
239  }
240  }
241  }
242  // (C and not D) or (not C and D)
243  Set<OWLClassExpression> clses = axiom.getClassExpressions();
244  Iterator<OWLClassExpression> it = clses.iterator();
245  OWLClassExpression descC = it.next();
246  OWLClassExpression notC = dataFactory.getOWLObjectComplementOf(descC);
247  OWLClassExpression descD = it.next();
248  OWLClassExpression notD = dataFactory.getOWLObjectComplementOf(descD);
249  OWLObjectIntersectionOf left = dataFactory
250  .getOWLObjectIntersectionOf(CollectionFactory.createSet(descC, notD));
251  OWLObjectIntersectionOf right = dataFactory
252  .getOWLObjectIntersectionOf(CollectionFactory.createSet(notC, descD));
253  desc = dataFactory.getOWLObjectUnionOf(CollectionFactory.createSet(left, right));
254  }
