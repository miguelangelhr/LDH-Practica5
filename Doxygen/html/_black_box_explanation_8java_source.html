159 
160  if (!getDefinitionTracker().isDefined(unsatClass)) {
161  return Collections.emptySet();

190  private int expandAxioms() {
191  /*
192  * We expand the axiom set using axioms that define entities that are
193  * already referenced in the existing set of axioms. If this fails to
194  * expand the axiom set we expand using axioms that reference the
195  * entities in the axioms that have already been expanded.
196  */
197 
198  // Keep track of the number of axioms that have been added
199  int axiomsAdded = 0;
200 
201  int remainingSpace = expansionLimit;
202  for (OWLAxiom ax : new ArrayList<OWLAxiom>(debuggingAxioms)) {
203  if (expandedWithDefiningAxioms.contains(ax)) {
204  // Skip if already done
205  continue;
206  }
207  // Collect the entities that have been used in the axiom
208  for (OWLEntity curObj : ax.getSignature()) {
209  if (!objectsExpandedWithDefiningAxioms.contains(curObj)) {
210  int added = expandWithDefiningAxioms(curObj, remainingSpace);
211  axiomsAdded += added;
212  remainingSpace = remainingSpace - added;
213  if (remainingSpace == 0) {
214  expansionLimit *= expansionFactor;
215  return axiomsAdded;
216  }
217  // Flag that we have completely expanded all defining axioms
218  // for this particular entity
219  objectsExpandedWithDefiningAxioms.add(curObj);
220  }
221  }
222 
223  // Flag that we've completely expanded this particular axiom
224  expandedWithDefiningAxioms.add(ax);
225  }
226 
227  if (axiomsAdded > 0) {
228  return axiomsAdded;
229  }
230 
231  // No axioms added at this point. Start adding axioms that reference
232  // entities contained in the current set of debugging axioms
233  for (OWLAxiom ax : new ArrayList<OWLAxiom>(debuggingAxioms)) {
234  if (expandedWithReferencingAxioms.contains(ax)) {
235  // Skip - already done this one
236  continue;
237  }
238  // Keep track of the number of axioms that have been added
239  for (OWLEntity curObj : ax.getSignature()) {
240  if (!objectsExpandedWithReferencingAxioms.contains(curObj)) {
241  int added = expandWithReferencingAxioms(curObj, expansionLimit);
242  axiomsAdded += added;
243  remainingSpace -= added;
244  if (remainingSpace == 0) {
245  expansionLimit *= expansionFactor;
246  return axiomsAdded;
247  }
248  objectsExpandedWithReferencingAxioms.add(curObj);
249  }
250  }
251  expandedWithReferencingAxioms.add(ax);
252  }
253  return axiomsAdded;
254  }

261  private int expandWithDefiningAxioms(OWLEntity obj, int limit) {
262  Set<OWLAxiom> expansionAxioms = new HashSet<OWLAxiom>();
263  for (OWLOntology ont : getOntology().getImportsClosure()) {
264  boolean referenceFound = false;
265  if (obj instanceof OWLClass) {
266  referenceFound = expansionAxioms.addAll(ont.getAxioms((OWLClass) obj));
267  }
268  else if (obj instanceof OWLObjectProperty) {
269  referenceFound = expansionAxioms.addAll(ont.getAxioms((OWLObjectProperty) obj));
270  }
271  else if (obj instanceof OWLDataProperty) {
272  referenceFound = expansionAxioms.addAll(ont.getAxioms((OWLDataProperty) obj));
273  }
274  else if (obj instanceof OWLIndividual) {
275  referenceFound = expansionAxioms.addAll(ont.getAxioms((OWLIndividual) obj));
276  }
277  if (!referenceFound) {
278  expansionAxioms.add(owlOntologyManager.getOWLDataFactory().getOWLDeclarationAxiom(obj));
279  }
280  }
281  expansionAxioms.removeAll(debuggingAxioms);
282  return addMax(expansionAxioms, debuggingAxioms, limit);
283  }

290  private int expandWithReferencingAxioms(OWLEntity obj, int limit) {
291  Set<OWLAxiom> expansionAxioms = new HashSet<OWLAxiom>();
292  // First expand by getting the defining axioms - if this doesn't
293  // return any axioms, then get the axioms that reference the entity
294  for (OWLOntology ont : getOntology().getImportsClosure()) {
295  expansionAxioms.addAll(ont.getReferencingAxioms(obj));
296  }
297  expansionAxioms.removeAll(debuggingAxioms);
298  return addMax(expansionAxioms, debuggingAxioms, limit);
299  }

312  private static <N extends OWLAxiom> int addMax(Set<N> source, Set<N> dest, int limit) {
313  int count = 0;
314  for (N obj : source) {
315  if (count == limit) {
316  break;
317  }
318  if (!(obj instanceof OWLAnnotationAxiom)) {
319  if (dest.add(obj)) {
320  count++;
321  }
322  }
323  }
324  return count;
325  }

335  private void performFastPruning(OWLClassExpression unsatClass) throws OWLException {
336  Set<OWLAxiom> axiomWindow = new HashSet<OWLAxiom>();
337  Object[] axioms = debuggingAxioms.toArray();
338  if (log.isLoggable(Level.FINE)) {
339  log.info("Fast pruning: ");
340  }
341  if (log.isLoggable(Level.FINE)) {
342  log.fine(" - Window size: " + fastPruningWindowSize);
343  }
344  int windowCount = debuggingAxioms.size() / fastPruningWindowSize;
345  for (int currentWindow = 0; currentWindow < windowCount; currentWindow++) {
346  axiomWindow.clear();
347  int startIndex = currentWindow * fastPruningWindowSize;
348  int endIndex = startIndex + fastPruningWindowSize;
349  for (int axiomIndex = startIndex; axiomIndex < endIndex; axiomIndex++) {
350  OWLAxiom currentAxiom = (OWLAxiom) axioms[axiomIndex];
351  axiomWindow.add(currentAxiom);
352  debuggingAxioms.remove(currentAxiom);
353  }
354  if (isSatisfiable(unsatClass)) {
355  debuggingAxioms.addAll(axiomWindow);
356  }
357  }
358  // Add any left over axioms
359  axiomWindow.clear();
360  int remainingAxiomsCount = debuggingAxioms.size() % fastPruningWindowSize;
361  if (remainingAxiomsCount > 0) {
362  int fragmentIndex = windowCount * fastPruningWindowSize;
363  while (fragmentIndex < axioms.length) {
364  OWLAxiom curAxiom = (OWLAxiom) axioms[fragmentIndex];
365  axiomWindow.add(curAxiom);
366  debuggingAxioms.remove(curAxiom);
367  fragmentIndex++;
368  }
369  if (isSatisfiable(unsatClass)) {
370  debuggingAxioms.addAll(axiomWindow);
371  }
372  }
373  if (log.isLoggable(Level.FINE)) {
374  log.fine(" - End of fast pruning");
375  }
376  }

379  private void performSlowPruning(OWLClassExpression unsatClass) throws OWLException {
380  // Simply remove axioms one at a time. If the class
381  // being debugged turns satisfiable then we know we have
382  // an SOS axiom.
383  List<OWLAxiom> axiomsCopy = new ArrayList<OWLAxiom>(debuggingAxioms);
384  for (OWLAxiom ax : axiomsCopy) {
385  debuggingAxioms.remove(ax);
386  if (isSatisfiable(unsatClass)) {
387  // Affects satisfiability, so add back in
388  debuggingAxioms.add(ax);
389  }
390  }
391  }

406  private boolean isSatisfiable(OWLClassExpression unsatClass) throws OWLException {
407  createDebuggingOntology();
408 
409  OWLReasoner reasoner = getReasonerFactory().createNonBufferingReasoner(debuggingOntology);
410 
411  if (OntologyUtils.containsUnreferencedEntity(debuggingOntology, unsatClass)) {
412  reasoner.dispose();
413  return true;
414  }
415  satTestCount++;
416  boolean sat = reasoner.isSatisfiable(unsatClass);
417  reasoner.dispose();
418  return sat;
419  }

440  private void expandUntilUnsatisfiable(OWLClassExpression unsatClass) throws OWLException {
441  // Perform the initial expansion - this will cause
442  // the debugging axioms set to be expanded to the
443  // defining axioms for the class being debugged
444  resetSatisfiabilityTestCounter();
445 
446  if (!unsatClass.isAnonymous()) {
447  expandWithDefiningAxioms((OWLClass) unsatClass, expansionLimit);
448  }
449  else {
450  OWLClass owlThing = owlOntologyManager.getOWLDataFactory().getOWLThing();
451  OWLSubClassOfAxiom axiom = owlOntologyManager.getOWLDataFactory().getOWLSubClassOfAxiom(unsatClass, owlThing);
452  debuggingAxioms.add(axiom);
453  expandAxioms();
454  debuggingAxioms.remove(axiom);
455  }
456 
457  if (log.isLoggable(Level.FINE)) {
458  log.fine("Initial axiom count: " + debuggingAxioms.size());
459  }
460 
461  int totalAdded = 0;
462  int expansionCount = 0;
463  while (isSatisfiable(unsatClass)) {
464 
465  if (log.isLoggable(Level.FINE)) {
466  log.fine("Expanding axioms (expansion " + expansionCount + ")");
467  }
468 
469  expansionCount++;
470  int numberAdded = expandAxioms();
471  totalAdded += numberAdded;
472 
473  if (log.isLoggable(Level.FINE)) {
474  log.fine(" ... expanded by " + numberAdded);
475  }
476 
477  if (numberAdded == 0) {
478 
479  if (log.isLoggable(Level.FINE)) {
480  log.fine("ERROR! Cannot find SOS axioms!");
481  }
482 
483  debuggingAxioms.clear();
484  return;
485  }
486  }
487 
488  if (log.isLoggable(Level.FINE)) {
489  log.fine("Total number of axioms added: " + totalAdded);
490  }
491  }

494  protected void pruneUntilMinimal(OWLClassExpression unsatClass) throws OWLException {
495  if (log.isLoggable(Level.FINE)) {
496  log.fine("FOUND CLASH! Pruning " + debuggingAxioms.size() + " axioms...");
497  }
498 
499  resetSatisfiabilityTestCounter();
500  if (log.isLoggable(Level.FINE)) {
501  log.fine("Fast pruning...");
502  }
503 
504  // fastPruningWindowSize = 0;
505  if (performRepeatedFastPruning) {
506  // Base the initial fast pruning window size on the number of axioms
507  fastPruningWindowSize = debuggingAxioms.size() / 6;
508  if (fastPruningWindowSize < DEFAULT_FAST_PRUNING_WINDOW_SIZE) {
509  fastPruningWindowSize = DEFAULT_FAST_PRUNING_WINDOW_SIZE;
510  }
511  if (log.isLoggable(Level.FINE)) {
512  log.fine(" Initial fast pruning window size: " + fastPruningWindowSize);
513  }
514  int fastPruningCounter = 0;
515  while (fastPruningWindowSize != 1) {
516  if (log.isLoggable(Level.FINE)) {
517  log.fine(" Round: " + fastPruningCounter + " (axioms to prune: " + debuggingAxioms.size() + ")");
518  }
519  fastPruningCounter++;
520  performFastPruning(unsatClass);
521  fastPruningWindowSize = fastPruningWindowSize / 3;
522  if (fastPruningWindowSize < 1) {
523  fastPruningWindowSize = 1;
524  }
525  }
526  if (log.isLoggable(Level.FINE)) {
527  log.fine("... end of fast pruning. Axioms remaining: " + debuggingAxioms.size());
528  log.fine("Performed " + satTestCount + " satisfiability tests during fast pruning");
529  }
530  }
531  else {
532  fastPruningWindowSize = DEFAULT_FAST_PRUNING_WINDOW_SIZE;
533  performFastPruning(unsatClass);
534  if (log.isLoggable(Level.FINE)) {
535  log.info("... end of fast pruning. Axioms remaining: " + debuggingAxioms.size());
536  log.info("Performed " + satTestCount + " satisfiability tests during fast pruning");
537  }
538  }
539 
540  int totalSatTests = satTestCount;
541 
542  resetSatisfiabilityTestCounter();
543  if (log.isLoggable(Level.FINE)) {
544  log.fine("Slow pruning...");
545  }
546 
547  performSlowPruning(unsatClass);
548 
549  if (log.isLoggable(Level.FINE)) {
550  log.fine("... end of slow pruning");
551  log.fine("Performed " + satTestCount + " satisfiability tests during slow pruning");
552  }
553  totalSatTests += satTestCount;
554  if (log.isLoggable(Level.FINE)) {
555  log.fine("Total number of satisfiability tests performed: " + totalSatTests);
556  }
557  }
