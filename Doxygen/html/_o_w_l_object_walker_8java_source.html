294  private class StructureWalker implements OWLObjectVisitor {
295 
296  private final Set<OWLObject> visited = new HashSet<OWLObject>();
297 
298  public StructureWalker() {}
299 
300  private void process(OWLObject object) {
301  if (!visitDuplicates) {
302  if (!visited.contains(object)) {
303  visited.add(object);
304  object.accept(visitor);
305  }
306  }
307  else {
308  object.accept(visitor);
309  }
310  }

319  OWLObjectWalker.this.ontology = ontologyToVisit;
320  OWLObjectWalker.this.ax = null;
321  process(ontologyToVisit);
322  for (OWLAnnotation anno : ontologyToVisit.getAnnotations()) {
323  anno.accept(this);
324  }
325  for (OWLAxiom a : ontologyToVisit.getAxioms()) {
326  a.accept(this);
327  }
328  }
