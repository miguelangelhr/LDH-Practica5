105  public RDFXMLRenderer(OWLOntology ontology, Writer w, OWLOntologyFormat format) {
106  super(ontology, format);
107  this.format = format;
108  qnameManager = new RDFXMLNamespaceManager(ontology, format);
109  String defaultNamespace = qnameManager.getDefaultNamespace();
110  String base;
111  if (defaultNamespace.endsWith("#")) {
112  base = defaultNamespace.substring(0, defaultNamespace.length() - 1);
113  } else {
114  base = defaultNamespace;
115  }
116  writer = new RDFXMLWriter(XMLWriterFactory.getInstance().createXMLWriter(w,
117  qnameManager, base));
118  prettyPrintedTypes = new HashSet<IRI>();
119  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_CLASS.getIRI());
120  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_OBJECT_PROPERTY.getIRI());
121  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_DATA_PROPERTY.getIRI());
122  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_ANNOTATION_PROPERTY.getIRI());
123  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_RESTRICTION.getIRI());
124  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_THING.getIRI());
125  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_NOTHING.getIRI());
126  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_ONTOLOGY.getIRI());
127  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_ANNOTATION_PROPERTY.getIRI());
128  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_NAMED_INDIVIDUAL.getIRI());
129  prettyPrintedTypes.add(OWLRDFVocabulary.RDFS_DATATYPE.getIRI());
130  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_AXIOM.getIRI());
131  prettyPrintedTypes.add(OWLRDFVocabulary.OWL_ANNOTATION.getIRI());
132  }

198  @Override
199  public void render(RDFResourceNode node) throws IOException {
200  if (pending.contains(node)) {
201  return;
202  }
203  pending.add(node);
204  RDFTriple candidatePrettyPrintTypeTriple = null;
205  final List<RDFTriple> triples = getGraph().getSortedTriplesForSubject(node, true);
206  for (RDFTriple triple : triples) {
207  IRI propertyIRI = triple.getProperty().getIRI();
208  if (propertyIRI.equals(OWLRDFVocabulary.RDF_TYPE.getIRI())
209  && !triple.getObject().isAnonymous()) {
210  if (OWLRDFVocabulary.BUILT_IN_VOCABULARY_IRIS.contains(triple.getObject()
211  .getIRI())) {
212  if (prettyPrintedTypes.contains(triple.getObject().getIRI())) {
213  candidatePrettyPrintTypeTriple = triple;
214  }
215  }
216  }
217  }
218  if (candidatePrettyPrintTypeTriple == null) {
219  writer.writeStartElement(RDF_DESCRIPTION.getIRI());
220  } else {
221  writer.writeStartElement(candidatePrettyPrintTypeTriple.getObject().getIRI());
222  }
223  if (!node.isAnonymous()) {
224  writer.writeAboutAttribute(node.getIRI());
225  }
226  // XXX this call looks like it should be made, but only when the node id
227  // is necessary
228  // else {
229  // writer.writeNodeIDAttribute(node);
230  // }
231  for (RDFTriple triple : triples) {
232  if (candidatePrettyPrintTypeTriple != null
233  && candidatePrettyPrintTypeTriple.equals(triple)) {
234  continue;
235  }
236  writer.writeStartElement(triple.getProperty().getIRI());
237  RDFNode objectNode = triple.getObject();
238  if (!objectNode.isLiteral()) {
239  RDFResourceNode objectRes = (RDFResourceNode) objectNode;
240  if (objectRes.isAnonymous()) {
241  // Special rendering for lists
242  if (isObjectList(objectRes)) {
243  writer.writeParseTypeAttribute();
244  List<RDFNode> list = new ArrayList<RDFNode>();
245  toJavaList(objectRes, list);
246  for (RDFNode n : list) {
247  if (n.isAnonymous()) {
248  render((RDFResourceNode) n);
249  } else {
250  if (n.isLiteral()) {
251  RDFLiteralNode litNode = (RDFLiteralNode) n;
252  writer.writeStartElement(OWLRDFVocabulary.RDFS_LITERAL
253  .getIRI());
254  if (litNode.getDatatype() != null) {
255  writer.writeDatatypeAttribute(litNode
256  .getDatatype());
257  } else if (litNode.getLang() != null) {
258  writer.writeLangAttribute(litNode.getLang());
259  }
260  writer.writeTextContent(litNode.getLiteral());
261  writer.writeEndElement();
262  } else {
263  writer.writeStartElement(RDF_DESCRIPTION.getIRI());
264  writer.writeAboutAttribute(n.getIRI());
265  writer.writeEndElement();
266  }
267  }
268  }
269  } else {
270  render(objectRes);
271  }
272  } else {
273  writer.writeResourceAttribute(objectRes.getIRI());
274  }
275  } else {
276  RDFLiteralNode rdfLiteralNode = (RDFLiteralNode) objectNode;
277  if (rdfLiteralNode.getDatatype() != null) {
278  writer.writeDatatypeAttribute(rdfLiteralNode.getDatatype());
279  } else if (rdfLiteralNode.getLang() != null) {
280  writer.writeLangAttribute(rdfLiteralNode.getLang());
281  }
282  writer.writeTextContent(rdfLiteralNode.getLiteral());
283  }
284  writer.writeEndElement();
285  }
286  writer.writeEndElement();
287  pending.remove(node);
288  }
