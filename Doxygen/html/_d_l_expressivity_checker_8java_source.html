215  private void pruneConstructs() {
216  if (constructs.contains(AL)) {
217  // AL + U + E can be represented using ALC
218  if (constructs.contains(C)) {
219  // Remove existential because this can be represented
220  // with AL + Neg
221  constructs.remove(E);
222  // Remove out union (intersection + negation (demorgan))
223  constructs.remove(U);
224  } else if (constructs.contains(E) && constructs.contains(U)) {
225  // Simplify to ALC
226  constructs.add(AL);
227  constructs.add(C);
228  constructs.remove(E);
229  constructs.remove(U);
230  }
231  }
232 
233  if (constructs.contains(N) || constructs.contains(Q)) {
234  constructs.remove(F);
235  }
236  if (constructs.contains(Q)) {
237  constructs.remove(N);
238  }
239  if (constructs.contains(AL) && constructs.contains(C) && constructs.contains(TRAN)) {
240  constructs.remove(AL);
241  constructs.remove(C);
242  constructs.remove(TRAN);
243  constructs.add(S);
244  }
245  if (constructs.contains(R)) {
246  constructs.remove(H);
247  }
248  }

251  private List<Construct> getOrderedConstructs() {
252  constructs.clear();
253  constructs.add(AL);
254  for (OWLOntology ont : ontologies) {
255  for (OWLAxiom ax : ont.getLogicalAxioms()) {
256  ax.accept(this);
257  }
258  }
259  pruneConstructs();
260  List<Construct> cons = new ArrayList<Construct>(constructs);
261  Collections.sort(cons, new ConstructComparator());
262  return cons;
263  }

401  private boolean isAtomic(OWLClassExpression classExpression) {
402  if (classExpression.isAnonymous()) {
403  return false;
404  } else {
405  for (OWLOntology ont : ontologies) {
406  if (!ont.getAxioms((OWLClass) classExpression).isEmpty()) {
407  return false;
408  }
409  }
410  return true;
411  }
