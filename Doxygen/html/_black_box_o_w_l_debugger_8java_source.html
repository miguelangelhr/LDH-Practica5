148  private OWLClass setupDebuggingClass(OWLClassExpression cls) throws OWLException {
149  if (!cls.isAnonymous()) {
150  return (OWLClass) cls;
151  } else {
152  // The class is anonymous, so we need to assign it a name
153  OWLClass curCls = owlOntologyManager.getOWLDataFactory().getOWLClass(
154  createIRI());
155  Set<OWLClassExpression> operands = new HashSet<OWLClassExpression>();
156  operands.add(curCls);
157  operands.add(cls);
158  temporaryAxioms.add(owlOntologyManager.getOWLDataFactory()
159  .getOWLEquivalentClassesAxiom(operands));
160  for (OWLAxiom ax : temporaryAxioms) {
161  owlOntologyManager.applyChanges(Arrays.asList(new AddAxiom(
162  getOWLOntology(), ax)));
163  }
164  return curCls;

182  // /////////////////////////////////////////////////////////////////////////////////////////
183  //
184  // Expansion
185  //
186  // /////////////////////////////////////////////////////////////////////////////////////////
187  private int expandAxioms() throws OWLException {
188  // We expand the axiom set using axioms that define entities that are
189  // already
190  // referenced in the existing set of axioms. If this fails to expand the
191  // axiom
192  // set we expand using axioms that reference the entities in the axioms
193  // that have
194  // already been expanded.
195  // Keep track of the number of axioms that have been added
196  int axiomsAdded = 0;
197  int remainingSpace = expansionLimit;
198  for (OWLAxiom ax : new ArrayList<OWLAxiom>(debuggingAxioms)) {
199  if (expandedWithDefiningAxioms.contains(ax)) {
200  // Skip if already done
201  continue;
202  }
203  // Collect the entities that have been used in the axiom
204  for (OWLEntity curObj : ax.getSignature()) {
205  if (!objectsExpandedWithDefiningAxioms.contains(curObj)) {
206  int added = expandWithDefiningAxioms(curObj, remainingSpace);
207  axiomsAdded += added;
208  remainingSpace = remainingSpace - added;
209  if (remainingSpace == 0) {
210  expansionLimit *= expansionFactor;
211  return axiomsAdded;
212  }
213  // Flag that we have completely expanded all defining axioms
214  // for this particular entity
215  objectsExpandedWithDefiningAxioms.add(curObj);
216  }
217  }
218  // Flag that we've completely expanded this particular axiom
219  expandedWithDefiningAxioms.add(ax);
220  }
221  if (axiomsAdded > 0) {
222  return axiomsAdded;
223  }
224  // No axioms added at this point. Start adding axioms that reference
225  // entities contained in the current set of debugging axioms
226  for (OWLAxiom ax : new ArrayList<OWLAxiom>(debuggingAxioms)) {
227  if (expandedWithReferencingAxioms.contains(ax)) {
228  // Skip - already done this one
229  continue;
230  }
231  // Keep track of the number of axioms that have been added
232  for (OWLEntity curObj : ax.getSignature()) {
233  if (!objectsExpandedWithReferencingAxioms.contains(curObj)) {
234  int added = expandWithReferencingAxioms(curObj, expansionLimit);
235  axiomsAdded += added;
236  remainingSpace -= added;
237  if (remainingSpace == 0) {
238  expansionLimit *= expansionFactor;
239  return axiomsAdded;
240  }
241  objectsExpandedWithReferencingAxioms.add(curObj);
242  }
243  }
244  expandedWithReferencingAxioms.add(ax);
245  }
246  return axiomsAdded;
247  }

251  private int expandWithDefiningAxioms(OWLEntity obj, int limit) throws OWLException {
252  Set<OWLAxiom> expansionAxioms = new HashSet<OWLAxiom>();
253  for (OWLOntology ont : owlOntologyManager.getImportsClosure(getOWLOntology())) {
254  if (obj instanceof OWLClass) {
255  expansionAxioms.addAll(ont.getAxioms((OWLClass) obj));
256  } else if (obj instanceof OWLObjectProperty) {
257  expansionAxioms.addAll(ont.getAxioms((OWLObjectProperty) obj));
258  } else if (obj instanceof OWLDataProperty) {
259  expansionAxioms.addAll(ont.getAxioms((OWLDataProperty) obj));
260  } else if (obj instanceof OWLIndividual) {
261  expansionAxioms.addAll(ont.getAxioms((OWLIndividual) obj));
262  }
263  }
264  expansionAxioms.removeAll(debuggingAxioms);
265  return addMax(expansionAxioms, debuggingAxioms, limit);
266  }

270  private int expandWithReferencingAxioms(OWLEntity obj, int limit) throws OWLException {
271  Set<OWLAxiom> expansionAxioms = new HashSet<OWLAxiom>();
272  // First expand by getting the defining axioms - if this doesn't
273  // return any axioms, then get the axioms that reference the entity
274  for (OWLOntology ont : owlOntologyManager.getImportsClosure(getOWLOntology())) {
275  expansionAxioms.addAll(ont.getReferencingAxioms(obj));
276  }
277  expansionAxioms.removeAll(debuggingAxioms);
278  return addMax(expansionAxioms, debuggingAxioms, limit);
279  }

292  private static <N extends OWLAxiom> int addMax(Set<N> source, Set<N> dest, int limit) {
293  int count = 0;
294  for (N obj : source) {
295  if (count == limit) {
296  break;
297  }
298  if (!(obj instanceof OWLAnnotationAxiom)) {
299  if (dest.add(obj)) {
300  count++;
301  }
302  }
303  }
304  return count;
305  }

307  // /////////////////////////////////////////////////////////////////////////////////////////
308  //
309  // Contraction/Pruning - Fast pruning is performed and then slow pruning is
310  // performed.
311  //
312  // /////////////////////////////////////////////////////////////////////////////////////////
313  private void performFastPruning() throws OWLException {
314  logger.setLevel(Level.INFO);
315  Set<OWLAxiom> axiomWindow = new HashSet<OWLAxiom>();
316  Object[] axioms = debuggingAxioms.toArray();
317  if (logger.isLoggable(Level.INFO)) {
318  logger.info("Fast pruning: ");
319  }
320  if (logger.isLoggable(Level.INFO)) {
321  logger.info(" - Window size: " + fastPruningWindowSize);
322  }
323  int windowCount = debuggingAxioms.size() / fastPruningWindowSize;
324  for (int currentWindow = 0; currentWindow < windowCount; currentWindow++) {
325  axiomWindow.clear();
326  int startIndex = currentWindow * fastPruningWindowSize;
327  int endIndex = startIndex + fastPruningWindowSize;
328  for (int axiomIndex = startIndex; axiomIndex < endIndex; axiomIndex++) {
329  OWLAxiom currentAxiom = (OWLAxiom) axioms[axiomIndex];
330  axiomWindow.add(currentAxiom);
331  debuggingAxioms.remove(currentAxiom);
332  }
333  if (isSatisfiable()) {
334  debuggingAxioms.addAll(axiomWindow);
335  }
336  }
337  // Add any left over axioms
338  axiomWindow.clear();
339  int remainingAxiomsCount = debuggingAxioms.size() % fastPruningWindowSize;
340  if (remainingAxiomsCount > 0) {
341  int fragmentIndex = windowCount * fastPruningWindowSize;
342  while (fragmentIndex < axioms.length) {
343  OWLAxiom curAxiom = (OWLAxiom) axioms[fragmentIndex];
344  axiomWindow.add(curAxiom);
345  debuggingAxioms.remove(curAxiom);
346  fragmentIndex++;
347  }
348  if (isSatisfiable()) {
349  debuggingAxioms.addAll(axiomWindow);
350  }
351  }
352  if (logger.isLoggable(Level.INFO)) {
353  logger.info(" - End of fast pruning");
354  }
355  }

357  private void performSlowPruning() throws OWLException {
358  // Simply remove axioms one at a time. If the class
359  // being debugged turns satisfiable then we know we have
360  // an SOS axoiom.
361  List<OWLAxiom> axiomsCopy = new ArrayList<OWLAxiom>(debuggingAxioms);
362  for (OWLAxiom ax : axiomsCopy) {
363  debuggingAxioms.remove(ax);
364  if (isSatisfiable()) {
365  // Affects satisfiability, so add back in
366  debuggingAxioms.add(ax);
367  }
368  }
369  }

380  private boolean isSatisfiable() throws OWLException {
381  createDebuggingOntology();
382  OWLReasoner reasoner = reasonerFactory
383  .createNonBufferingReasoner(debuggingOntology);
384  satTestCount++;
385  boolean sat = reasoner.isSatisfiable(currentClass);
386  reasoner.dispose();
387  return sat;
388  }

390  private void createDebuggingOntology() throws OWLException {
391  if (debuggingOntology != null) {
392  owlOntologyManager.removeOntology(debuggingOntology);
393  }
394  IRI iri = createIRI();
395  SimpleIRIMapper mapper = new SimpleIRIMapper(iri, iri);
396  owlOntologyManager.addIRIMapper(mapper);
397  debuggingOntology = owlOntologyManager.createOntology(iri);
398  owlOntologyManager.removeIRIMapper(mapper);
399  List<OWLOntologyChange> changes = new ArrayList<OWLOntologyChange>();
400  for (OWLAxiom ax : debuggingAxioms) {
401  changes.add(new AddAxiom(debuggingOntology, ax));
402  }
403  for (OWLAxiom ax : temporaryAxioms) {
404  changes.add(new AddAxiom(debuggingOntology, ax));
405  }
406  // Ensure the ontology contains the signature of the class which is
407  // being debugged
408  OWLDataFactory factory = owlOntologyManager.getOWLDataFactory();
409  OWLAxiom ax = factory.getOWLSubClassOfAxiom(currentClass, factory.getOWLThing());
410  changes.add(new AddAxiom(debuggingOntology, ax));
411  owlOntologyManager.applyChanges(changes);
412  }

418  private void generateSOSAxioms() throws OWLException {
419  // Perform the initial expansion - this will cause
420  // the debugging axioms set to be expanded to the
421  // defining axioms for the class being debugged
422  resetSatisfiabilityTestCounter();
423  expandWithDefiningAxioms(currentClass, expansionLimit);
424  if (logger.isLoggable(Level.INFO)) {
425  logger.info("Initial axiom count: " + debuggingAxioms.size());
426  }
427  int totalAdded = 0;
428  int expansionCount = 0;
429  while (isSatisfiable()) {
430  if (logger.isLoggable(Level.INFO)) {
431  logger.info("Expanding axioms (expansion " + expansionCount + ")");
432  }
433  expansionCount++;
434  int numberAdded = expandAxioms();
435  totalAdded += numberAdded;
436  if (logger.isLoggable(Level.INFO)) {
437  logger.info(" ... expanded by " + numberAdded);
438  }
439  if (numberAdded == 0) {
440  if (logger.isLoggable(Level.INFO)) {
441  logger.info("ERROR! Cannot find SOS axioms!");
442  }
443  debuggingAxioms.clear();
444  return;
445  }
446  }
447  if (logger.isLoggable(Level.INFO)) {
448  logger.info("Total number of axioms added: " + totalAdded);
449  }
450  if (logger.isLoggable(Level.INFO)) {
451  logger.info("FOUND CLASH! Prunning " + debuggingAxioms.size() + " axioms...");
452  }
453  resetSatisfiabilityTestCounter();
454  if (logger.isLoggable(Level.INFO)) {
455  logger.info("Fast pruning...");
456  }
457  if (performRepeatedFastPruning) {
458  // Base the initial fast pruning window size on the number of axioms
459  fastPruningWindowSize = debuggingAxioms.size() / 10;
460  if (fastPruningWindowSize < DEFAULT_FAST_PRUNING_WINDOW_SIZE) {
461  fastPruningWindowSize = DEFAULT_FAST_PRUNING_WINDOW_SIZE;
462  }
463  if (logger.isLoggable(Level.INFO)) {
464  logger.info(" Initial fast prunung window size: "
465  + fastPruningWindowSize);
466  }
467  int fastPruningCounter = 0;
468  while (fastPruningWindowSize != 1) {
469  if (logger.isLoggable(Level.INFO)) {
470  logger.info(" Round: " + fastPruningCounter
471  + " (axioms to prune: " + debuggingAxioms.size() + ")");
472  }
473  fastPruningCounter++;
474  performFastPruning();
475  fastPruningWindowSize = fastPruningWindowSize / 3;
476  if (fastPruningWindowSize < 1) {
477  fastPruningWindowSize = 1;
478  }
479  }
480  if (logger.isLoggable(Level.INFO)) {
481  logger.info("... end of fast pruning. Axioms remaining: "
482  + debuggingAxioms.size());
483  logger.info("Performed " + satTestCount
484  + " satisfiability tests during fast pruning");
485  }
486  } else {
487  fastPruningWindowSize = DEFAULT_FAST_PRUNING_WINDOW_SIZE;
488  performFastPruning();
489  if (logger.isLoggable(Level.INFO)) {
490  logger.info("... end of fast pruning. Axioms remaining: "
491  + debuggingAxioms.size());
492  logger.info("Performed " + satTestCount
493  + " satisfiability tests during fast pruning");
494  }
495  }
496  int totalSatTests = satTestCount;
497  resetSatisfiabilityTestCounter();
498  if (logger.isLoggable(Level.INFO)) {
499  logger.info("Slow pruning...");
500  }
501  performSlowPruning();
502  if (logger.isLoggable(Level.INFO)) {
503  logger.info("... end of slow pruning");
504  logger.info("Performed " + satTestCount
505  + " satisfiability tests during slow pruning");
506  }
507  totalSatTests += satTestCount;
508  if (logger.isLoggable(Level.INFO)) {
509  logger.info("Total number of satisfiability tests performed: "
510  + totalSatTests);
511  }
512  }
