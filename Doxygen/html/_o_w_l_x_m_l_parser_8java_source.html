79  public OWLOntologyFormat parse(OWLOntologyDocumentSource documentSource, OWLOntology ontology, OWLOntologyLoaderConfiguration configuration) throws OWLParserException, IOException, OWLOntologyChangeException, UnloadableImportException {
80  InputSource isrc = null;
81  try {
82  System.setProperty("entityExpansionLimit", "100000000");
83  OWLXMLOntologyFormat format = new OWLXMLOntologyFormat();
84  SAXParserFactory factory = SAXParserFactory.newInstance();
85  factory.setNamespaceAware(true);
86  SAXParser parser = factory.newSAXParser();
87  isrc = getInputSource(documentSource, configuration);
88  OWLXMLParserHandler handler = new OWLXMLParserHandler(ontology, configuration);
89  parser.parse(isrc, handler);
90  Map<String, String> prefix2NamespaceMap = handler.getPrefixName2PrefixMap();
91  for(String prefix : prefix2NamespaceMap.keySet()) {
92  format.setPrefix(prefix, prefix2NamespaceMap.get(prefix));
93  }
94  return format;
95  }
96  catch (ParserConfigurationException e) {
97  // What the hell should be do here? In serious trouble if this happens
98  throw new OWLRuntimeException(e);
99  }
100  catch (TranslatedOWLParserException e) {
101  throw e.getParserException();
102  }
103  catch (TranslatedUnloadableImportException e) {
104  throw e.getUnloadableImportException();
105  }
106  catch (SAXException e) {
107  // General exception
108  throw new OWLParserSAXException(e);
109  } finally {
110  if (isrc != null && isrc.getByteStream() != null) {
111  isrc.getByteStream().close();
112  } else if (isrc != null && isrc.getCharacterStream() != null) {
113  isrc.getCharacterStream().close();
114  }
115  }
116  }
