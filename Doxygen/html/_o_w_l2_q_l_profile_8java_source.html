140  @Override
141  public OWLProfileReport checkOntology(OWLOntology ontology) {
142  OWL2DLProfile profile = new OWL2DLProfile();
143  OWLProfileReport report = profile.checkOntology(ontology);
144  Set<OWLProfileViolation> violations = new HashSet<OWLProfileViolation>();
145  violations.addAll(report.getViolations());
146 
147  OWLOntologyWalker walker = new OWLOntologyWalker(ontology.getImportsClosure());
148  OWL2QLObjectVisitor visitor = new OWL2QLObjectVisitor(walker);
149  walker.walkStructure(visitor);
150  violations.addAll(visitor.getProfileViolations());
151  return new OWLProfileReport(this, violations);
152  }
