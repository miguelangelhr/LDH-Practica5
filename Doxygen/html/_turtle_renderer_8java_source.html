96  public TurtleRenderer(OWLOntology ontology, Writer writer, OWLOntologyFormat format) {
97  super(ontology, format);
98  this.format = format;
99  this.writer = new PrintWriter(writer);
100  pending = new HashSet<RDFResourceNode>();
101  pm = new DefaultPrefixManager();
102  if (!ontology.isAnonymous()) {
103  pm.setDefaultPrefix(ontology.getOntologyID().getOntologyIRI() + "#");
104  }
105  if (format instanceof PrefixOWLOntologyFormat) {
106  PrefixOWLOntologyFormat prefixFormat = (PrefixOWLOntologyFormat) format;
107  for (String prefixName : prefixFormat.getPrefixNames()) {
108  pm.setPrefix(prefixName, prefixFormat.getPrefix(prefixName));
109  }
110  }
111  base = "";
112  }

114  private void writeNamespaces() {
115  for (String prefixName : pm.getPrefixName2PrefixMap().keySet()) {
116  String prefix = pm.getPrefix(prefixName);
117  write("@prefix ");
118  write(prefixName);
119  write(" ");
120  writeAsURI(prefix);
121  write(" .");
122  writeNewLine();
123  }
124  }

140  private void write(String s) {
141  int newLineIndex = s.indexOf('\n');
142  if (newLineIndex != -1) {
143  lastNewLineIndex = bufferLength + newLineIndex;
144  }
145  writer.write(s);
146  bufferLength += s.length();
147  }

157  private void writeAsURI(String s) {
158  write("<");
159  if (s.startsWith(base)) {
160  write(s.substring(base.length()));
161  } else {
162  write(s);
163  }
164  write(">");

167  private void write(IRI iri) {
168  if (iri.equals(getOntology().getOntologyID().getOntologyIRI())) {
169  writeAsURI(iri.toString());
170  } else {
171  String name = pm.getPrefixIRI(iri);
172  if (name == null) {
173  // No QName!
174  writeAsURI(iri.toString());
175  } else {
176  if (name.indexOf(':') != -1) {
177  write(name);
178  } else {
179  write(":");
180  write(name);
181  }
182  }
183  }
184  }

186  private void writeNewLine() {
187  write("\n");

189  if (!tabs.isEmpty()) {
190  tabIndent = tabs.peek();
191  }
192  for (int i = 1; i < tabIndent; i++) {
193  write(" ");

205  private void write(RDFLiteralNode node) {
206  if (node.getDatatype() != null) {
207  if (node.getDatatype().equals(XSDVocabulary.INTEGER.getIRI())) {
208  write(node.getLiteral());
209  } else if (node.getDatatype().equals(XSDVocabulary.DECIMAL.getIRI())) {
210  write(node.getLiteral());
211  } else {
212  writeStringLiteral(node.getLiteral());
213  write("^^");
214  write(node.getDatatype());
215  }
216  } else {
217  writeStringLiteral(node.getLiteral());
218  if (node.getLang() != null) {
219  write("@");
220  write(node.getLang());
221  }
222  }
223  }

225  private void writeStringLiteral(String literal) {
226  String escapedLiteral = EscapeUtils.escapeString(literal);
227  if (escapedLiteral.indexOf('\n') != -1) {
228  write("\"\"\"");
229  write(escapedLiteral);
230  write("\"\"\"");
231  } else {
232  write("\"");
233  write(escapedLiteral);
234  write("\"");
235  }
236  }

238  private void write(RDFResourceNode node) {
239  if (!node.isAnonymous()) {
240  write(node.getIRI());
241  } else {
242  pushTab();
243  if (!isObjectList(node)) {
244  render(node);
245  } else {
246  // List - special syntax
247  List<RDFNode> list = new ArrayList<RDFNode>();
248  toJavaList(node, list);
249  pushTab();
250  write("(");
251  write(" ");
252  pushTab();
253  for (Iterator<RDFNode> it = list.iterator(); it.hasNext();) {
254  write(it.next());
255  if (it.hasNext()) {
256  writeNewLine();
257  }
258  }
259  popTab();
260  writeNewLine();
261  write(")");
262  popTab();
263  }
264  popTab();
265  }

359  @Override
360  public void render(RDFResourceNode node) {
361  level++;
362  List<RDFTriple> triples = getGraph().getSortedTriplesForSubject(node, true);
363  if (pending.contains(node)) {
364  // We essentially remove all structure sharing during parsing - any
365  // cycles therefore indicate a bug!
366  triples = new ArrayList<RDFTriple>();
367  }
368  pending.add(node);
369  RDFResourceNode lastSubject = null;
370  RDFResourceNode lastPredicate = null;
371  boolean first = true;
372  for (RDFTriple triple : triples) {
373  RDFResourceNode subj = triple.getSubject();
374  RDFResourceNode pred = triple.getProperty();
375  if (lastSubject != null && (subj.equals(lastSubject) || subj.isAnonymous())) {
376  if (lastPredicate != null && pred.equals(lastPredicate)) {
377  // Only the object differs from previous triple
378  // Just write the object
379  write(" ,");
380  writeNewLine();
381  write(triple.getObject());
382  } else {
383  // The predicate, object differ from previous triple
384  // Just write the predicate and object
385  write(" ;");
386  popTab();
387  if (!subj.isAnonymous()) {
388  writeNewLine();
389  }
390  writeNewLine();
391  write(triple.getProperty());
392  write(" ");
393  pushTab();
394  write(triple.getObject());
395  }
396  } else {
397  if (!first) {
398  popTab();
399  popTab();
400  writeNewLine();
401  }
402  // Subject, preficate and object are different from last triple
403  if (!node.isAnonymous()) {
404  write(triple.getSubject());
405  write(" ");
406  } else {
407  pushTab();
408  write("[");
409  write(" ");
410  }
411  pushTab();
412  write(triple.getProperty());
413  write(" ");
414  pushTab();
415  write(triple.getObject());
416  }
417  lastSubject = subj;
418  lastPredicate = pred;
419  first = false;
420  }
421  if (node.isAnonymous()) {
422  popTab();
423  popTab();
424  if (triples.isEmpty()) {
425  write("[ ");
426  } else {
427  writeNewLine();
428  }
429  write("]");
430  popTab();
431  } else {
432  popTab();
433  popTab();
434  }
435  if (level == 1 && !triples.isEmpty()) {
436  write(" .\n");
437  }
438  writer.flush();
439  pending.remove(node);
440  level--;
441  }
