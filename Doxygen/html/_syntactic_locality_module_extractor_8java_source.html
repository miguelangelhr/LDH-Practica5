283  boolean[] extractLogicalAxioms(boolean[] subOnt, Set<OWLEntity> signature,
284  LocalityClass localityClass, boolean verbose) {
285  boolean[] mod = ontologyAxiomSet.getSubset(false);
286  boolean[] q2 = ontologyAxiomSet.cloneSubset(subOnt);
287  SyntacticLocalityEvaluator sle = new SyntacticLocalityEvaluator(localityClass);
288  boolean change = true;
289  int loopNumber = 0;
290  while (change) {
291  change = false;
292  loopNumber++;
293  if (verbose) {
294  System.out.println(" Loop " + loopNumber);
295  }
296  for (int i = 0; i < q2.length; i = i + 1) {
297  if (q2[i]) {
298  if (!sle.isLocal(ontologyAxiomSet.getAxiom(i), signature)) {
299  if (verbose) {
300  System.out.println(" Non-local axiom: "

302  .toString()));
303  }
304  mod[i] = true;
305  q2[i] = false;
306  int oldSize = signature.size();
307  signature.addAll(ontologyAxiomSet.getAxiom(i).getSignature());
308  // only triggering a change when the signature has

311  change = true;
312  if (verbose) {
313  System.out.print(" New signature: ");

315  System.out.print(minusOntologyURI(ent.toString())

317  }

319  }
320  }
321  } else {
322  if (verbose) {

350  LocalityClass localityClass, boolean verbose) {
351  HashSet<OWLAxiom> mod = new HashSet<OWLAxiom>();
352  HashSet<OWLAxiom> q2 = new HashSet<OWLAxiom>(subOnt);
353  SyntacticLocalityEvaluator sle = new SyntacticLocalityEvaluator(localityClass);
354  boolean change = true;
355  int loopNumber = 0;
356  while (change) {
357  change = false;
358  loopNumber++;
359  if (verbose) {
360  System.out.println(" Loop " + loopNumber);
361  }
362  HashSet<OWLAxiom> q2remove = new HashSet<OWLAxiom>();
363  for (OWLAxiom ax : q2) {
364  if (!sle.isLocal(ax, signature)) {
365  if (verbose) {
366  System.out.println(" Non-local axiom: "
367  + minusOntologyURI(ax.toString()));
368  }
369  mod.add(ax);
370  q2remove.add(ax);
371  int oldSize = signature.size();
372  signature.addAll(ax.getSignature());
373  // only triggering a change when the signature has changed
374  // doesn't improve performance
375  if (signature.size() > oldSize) {
376  change = true;
377  if (verbose) {
378  System.out.print(" New signature: ");
379  for (OWLEntity ent : signature) {
380  System.out.print(minusOntologyURI(ent.toString()) + ", ");
381  }
382  System.out.println();
383  }
384  }
385  } else {
386  if (verbose) {
387  System.out.println(" Local axiom: "
388  + minusOntologyURI(ax.toString()));
389  }
390  }
391  }
392  q2.removeAll(q2remove);
393  }

411  Set<OWLAxiom> enrichedModule = new HashSet<OWLAxiom>(module);
412  if (verbose) {
413  System.out
414  .println("\nEnriching with declaration axioms, annotation axioms, same/different individual axioms ...");
415  }
416  // Adding all entity declaration axioms
417  // Adding all entity annotation axioms
418  if (ontology != null) {
419  for (OWLEntity entity : sig) {
420  Set<OWLDeclarationAxiom> declarationAxioms = ontology
421  .getDeclarationAxioms(entity);
422  enrichedModule.addAll(declarationAxioms);
423  if (verbose) {
424  for (OWLDeclarationAxiom declarationAxiom : declarationAxioms) {
425  System.out.println(" Added entity declaration axiom: "

427  }
428  }
429  Set<OWLAnnotationAssertionAxiom> entityAnnotationAxioms = entity
430  .getAnnotationAssertionAxioms(ontology);
431  enrichedModule.addAll(entityAnnotationAxioms);
432  if (verbose) {
433  for (OWLAnnotationAssertionAxiom entityAnnotationAxiom : entityAnnotationAxioms) {
434  System.out.println(" Added entity annotation axiom: "
435  + minusOntologyURI(entityAnnotationAxiom.toString()));
436  }
437  }
438  }
439  }
440  // Adding all same-individuals axioms
441  // Adding all different-individuals axioms
442  if (ontology != null) {
443  for (OWLEntity entity : sig) {
444  if (OWLNamedIndividual.class.isAssignableFrom(entity.getClass())) {
445  OWLIndividual individual = (OWLIndividual) entity;
446  Set<OWLSameIndividualAxiom> sameIndividualAxioms = ontology
447  .getSameIndividualAxioms(individual);
448  enrichedModule.addAll(sameIndividualAxioms);

450  for (OWLSameIndividualAxiom sameIndividualAxiom : sameIndividualAxioms) {

453  }
454  }

456  .getDifferentIndividualAxioms(individual);

458  if (verbose) {
459  for (OWLDifferentIndividualsAxiom differentIndividualsAxiom : differentIndividualAxioms) {

497  Set<OWLClass> SuperOrSubClasses(int superOrSubClassLevel, boolean superVsSub,
498  OWLReasoner reasoner, Set<OWLClass> classesInSig) {
499  Set<OWLClass> superOrSubClasses = new HashSet<OWLClass>();
500  if (superOrSubClassLevel < 0) {
501  for (OWLClassExpression ent : classesInSig) {
502  NodeSet<OWLClass> nodes;
503  if (superVsSub) {
504  nodes = reasoner.getSuperClasses(ent, false);
505  } else {
506  nodes = reasoner.getSubClasses(ent, false);
507  }
508  superOrSubClasses.addAll(nodes.getFlattened());
509  }
510  } else if (superOrSubClassLevel > 0) {
511  Queue<OWLClass> toBeSuClassedNow;
512  Queue<OWLClass> toBeSuClassedNext = new LinkedList<OWLClass>(classesInSig);
513  Queue<OWLClass> suClassesToBeAdded = new LinkedList<OWLClass>();
514  for (int i = 0; i < superOrSubClassLevel; i++) {
515  toBeSuClassedNow = toBeSuClassedNext;
516  toBeSuClassedNext = new LinkedList<OWLClass>();
517  for (OWLClassExpression ce : toBeSuClassedNow) {
518  Set<OWLClass> suClasses;
519  if (superVsSub) {
520  suClasses = reasoner.getSuperClasses(ce, true).getFlattened();
521  } else {
522  suClasses = reasoner.getSubClasses(ce, true).getFlattened();
523  }
524  for (OWLClass suClass : suClasses) {
525  if (!classesInSig.contains(suClass)
526  && !suClassesToBeAdded.contains(suClass)) {
527  toBeSuClassedNext.add(suClass);
528  suClassesToBeAdded.add(suClass);
529  }
530  }
531  }
532  }
533  superOrSubClasses.addAll(suClassesToBeAdded);
534  }
535  return superOrSubClasses;
536  }

538  Set<OWLEntity> enrichSignature(Set<OWLEntity> sig, int superClassLevel,
539  int subClassLevel, OWLReasoner reasoner) {
540  Set<OWLEntity> enrichedSig = new HashSet<OWLEntity>(sig);
541  Set<OWLClass> classesInSig = new HashSet<OWLClass>();
542  for (OWLEntity ent : sig) {
543  if (OWLClass.class.isAssignableFrom(ent.getClass())) {
544  classesInSig.add((OWLClass) ent);
545  }
546  }
547  if (superClassLevel != 0) {
548  enrichedSig.addAll(SuperOrSubClasses(superClassLevel, true, reasoner,
549  classesInSig));
550  }
551  if (subClassLevel != 0) {
552  enrichedSig.addAll(SuperOrSubClasses(subClassLevel, false, reasoner,
553  classesInSig));
554  }
555  return enrichedSig;
556  }

604  public Set<OWLAxiom> extract(Set<OWLEntity> sig, int superClassLevel,
605  int subClassLevel, OWLReasoner reasoner, boolean verbose) {
606  Set<OWLEntity> enrichedSig = enrichSignature(sig, superClassLevel, subClassLevel,
607  reasoner);
608  switch (moduleType) {
609  case TOP: {
610  return extractUnnestedModule(enrichedSig, LocalityClass.TOP_TOP, verbose);
611  }
612  case BOT: {
613  return extractUnnestedModule(enrichedSig, LocalityClass.BOTTOM_BOTTOM,
614  verbose);
615  }
616  case STAR: {
617  boolean[] subOnt = ontologyAxiomSet.getSubset(true);
618  boolean nextStepNecessary = true;
619  boolean inFirstStep = true;
620  LocalityClass localityClass = LocalityClass.BOTTOM_BOTTOM;
621  Set<OWLEntity> seedSig = new HashSet<OWLEntity>(enrichedSig);
622  while (nextStepNecessary) {
623  outputSignature("\nExtracting " + localityClass
624  + " module for the following seed signature: ", enrichedSig,
625  verbose);
626  int previousModuleSize = ontologyAxiomSet.subsetCardinality(subOnt);
627  seedSig = new HashSet<OWLEntity>(enrichedSig);
628  subOnt = extractLogicalAxioms(subOnt, seedSig, localityClass, verbose);
629  if (ontologyAxiomSet.subsetCardinality(subOnt) == previousModuleSize
630  && !inFirstStep) {
631  nextStepNecessary = false;
632  }
633  inFirstStep = false;
634  if (localityClass == LocalityClass.BOTTOM_BOTTOM) {
635  localityClass = LocalityClass.TOP_TOP;
636  } else {
637  localityClass = LocalityClass.BOTTOM_BOTTOM;
638  }
639  }
640  Set<OWLAxiom> moduleAsSet = ontologyAxiomSet.toSet(subOnt);
641  return enrich(moduleAsSet, seedSig, verbose);
642  }
643  default:
644  throw new RuntimeException("Unsupported module type: " + moduleType);
645  }
646  }
