69  public IRI getIRIFromAttribute(String localName, String value)
70  throws OWLParserException {
71  if (localName.equals(OWLXMLVocabulary.IRI_ATTRIBUTE.getShortName())) {
72  return getIRI(value);
73  } else if (localName.equals(OWLXMLVocabulary.ABBREVIATED_IRI_ATTRIBUTE
74  .getShortName())) {
75  return getAbbreviatedIRI(value);
76  } else if (localName.equals("URI")) {
77  // Legacy
78  return getIRI(value);
79  }
80  throw new OWLXMLParserAttributeNotFoundException(getLineNumber(),
81  getColumnNumber(), OWLXMLVocabulary.IRI_ATTRIBUTE.getShortName());
82  }

84  public IRI getIRIFromElement(String elementLocalName, String textContent)
85  throws OWLParserException {
86  if (elementLocalName.equals(OWLXMLVocabulary.IRI_ELEMENT.getShortName())) {
87  return handler.getIRI(textContent.trim());
88  } else if (elementLocalName.equals(OWLXMLVocabulary.ABBREVIATED_IRI_ELEMENT
89  .getShortName())) {
90  return handler.getAbbreviatedIRI(textContent.trim());
91  }
92  throw new OWLXMLParserException(elementLocalName + " is not an IRI element",
93  getLineNumber(), getColumnNumber());
94  }
