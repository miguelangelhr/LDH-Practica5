85  public boolean put(Key key, Value value) {

98  protected Collection<Value> createCollection() {
99  Collection<Value> toReturn;
100  if (useSets) {
101  if (threadSafe) {
102  toReturn = CollectionFactory.createSyncSet();
103  } else {
104  toReturn = CollectionFactory.createSet();
105  }
106  } else {
107  if (threadSafe) {
108  toReturn = Collections.synchronizedList(new ArrayList<Value>());

112  }

178  public boolean remove(Key key, Value value) {
179  Collection<Value> c = this.map.get(key);
180  if (c != null) {
181  boolean toReturn = c.remove(value);
182  // if false, no change was actually made - skip the rest
183  if (!toReturn) {
