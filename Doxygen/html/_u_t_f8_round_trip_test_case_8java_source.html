57 @SuppressWarnings("javadoc")
58 public class UTF8RoundTripTestCase {
59  @Test
60  public void testRoundTrip() throws Exception {
61  String NS = "http://protege.org/ontologies/UTF8RoundTrip.owl";
62  OWLClass C = Class(IRI(NS + "#C"));
63  /*
64  * The two unicode characters entered here are valid and can be found in
65  * the code chart http://www.unicode.org/charts/PDF/U4E00.pdf. It has
66  * been said that they are chinese and they do look the part. In UTF-8
67  * these characters are encoded as \u8655 --> \350\231\225 \u65b9 -->
68  * \346\226\271 where the right hand side is in octal. (I chose octal
69  * because this is how emacs represents it with find-file-literally).
70  */
71  String CHINESE = "Rx\u8655\u65b9";
72  System.setProperty("file.encoding", "UTF-8");
73  OWLOntology ontology = createOriginalOntology(NS, C, CHINESE);
74  checkOntology(ontology, C, CHINESE);
75  OWLOntology newOntology = roundTrip(ontology);
76  checkOntology(newOntology, C, CHINESE);
77  }
78 
79  private static OWLOntology createOriginalOntology(String NS, OWLClass C,
80  String CHINESE) throws OWLOntologyCreationException {
81  OWLOntologyManager manager = Factory.getManager();
82  OWLOntology ontology = manager.createOntology(IRI(NS));
83  OWLAxiom annotationAxiom = AnnotationAssertion(RDFSLabel(), C.getIRI(),
84  Literal(CHINESE));
85  manager.addAxiom(ontology, annotationAxiom);
86  return ontology;
87  }
88 
89  private static boolean
90  checkOntology(OWLOntology ontology, OWLClass C, String CHINESE) {
91  for (OWLAnnotation annotation : C.getAnnotations(ontology)) {
92  String value = ((OWLLiteral) annotation.getValue()).getLiteral();
93  return CHINESE.equals(value);
94  }
95  return false;
96  }
97 
98  private static OWLOntology roundTrip(OWLOntology ontology)
99  throws
100  OWLOntologyStorageException, OWLOntologyCreationException {
101  OWLOntologyManager oldManager = ontology.getOWLOntologyManager();
102  StringDocumentTarget target = new StringDocumentTarget();
103  oldManager.saveOntology(ontology, new RDFXMLOntologyFormat(), target);
104  OWLOntologyManager newManager = Factory.getManager();
105  OWLOntology newOntology;
106  newOntology = newManager
107  .loadOntologyFromOntologyDocument(new StringDocumentSource(target
108  .toString()));
109  return newOntology;
110  }
