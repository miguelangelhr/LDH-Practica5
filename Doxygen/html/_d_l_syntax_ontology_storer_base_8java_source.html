80  @Override
81  protected void
82  storeOntology(OWLOntology ontology, Writer w, OWLOntologyFormat format)
83  throws OWLOntologyStorageException {
84 
85  ont = ontology;
86  PrintWriter writer = new PrintWriter(w);
87  beginWritingOntology(ontology, writer);
88  for (OWLObjectProperty prop : new TreeSet<OWLObjectProperty>(ontology.getObjectPropertiesInSignature())) {
89  write(prop, ontology.getAxioms(prop), writer);
90  }
91  for (OWLDataProperty prop : new TreeSet<OWLDataProperty>(ontology.getDataPropertiesInSignature())) {
92  write(prop, ontology.getAxioms(prop), writer);
93  }
94  for (OWLClass cls : new TreeSet<OWLClass>(ontology.getClassesInSignature())) {
95  write(cls, ontology.getAxioms(cls), writer);
96  }
97  for (OWLNamedIndividual ind : new TreeSet<OWLNamedIndividual>(ontology.getIndividualsInSignature())) {
98  write(ind, ontology.getAxioms(ind), writer);
99  }
100  beginWritingGeneralAxioms(ontology.getGeneralClassAxioms(), writer);
101  for (OWLAxiom ax : ontology.getGeneralClassAxioms()) {
102  beginWritingAxiom(ax, writer);
103  writeAxiom(null, ax, writer);
104  endWritingAxiom(ax, writer);
105  }
106  endWritingGeneralAxioms(ontology.getGeneralClassAxioms(), writer);
107  endWritingOntology(ontology, writer);
108  writer.flush();
109  }

111  private void write(OWLEntity entity, Set<? extends OWLAxiom> axioms, PrintWriter writer) {
112  beginWritingAxioms(entity, axioms, writer);
113  for (OWLAxiom ax : new TreeSet<OWLAxiom>(axioms)) {
114  beginWritingAxiom(ax, writer);
115  writeAxiom(entity, ax, writer);
116  endWritingAxiom(ax, writer);
117  }
118  Set<OWLAxiom> usages = new TreeSet<OWLAxiom>(ont.getReferencingAxioms(entity));
119  usages.removeAll(axioms);
120  beginWritingUsage(entity, usages, writer);
121  for (OWLAxiom usage : usages) {
122  if (!axioms.contains(usage)) {
123  beginWritingAxiom(usage, writer);
124  writeAxiom(entity, usage, writer);
125  endWritingAxiom(usage, writer);
126  }
127  }
128  endWritingUsage(entity, usages, writer);
129  endWritingAxioms(entity, axioms, writer);
130  }
