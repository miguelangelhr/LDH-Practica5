437  protected void flattenProperties(final Set<OWLObjectPropertyExpression> properties, KRSSVocabulary junctor) {
438  if (properties.isEmpty()) {
439  return;
440  }
441  if (properties.size() == 1) {
442  write(properties.iterator().next());
443  return;
444  }
445 
446  if (junctor != null) {
447  writeOpenBracket();
448  write(junctor);
449  }
450  List<OWLObjectPropertyExpression> props = sort(properties);
451  write(props.get(0));
452  final int size = props.size();
453  final int indent = getIndent();
454  for (int i = 1; i < size; i++) {
455  writeln();
456  writeIndent(indent);
457  write(props.get(i));
458  }
459  if (junctor != null) {
460  writeCloseBracket();
461  }
462  }

464  protected void flatten(final Set<OWLClassExpression> description, KRSSVocabulary junctor) {
465  if (description.isEmpty()) {
466  return;
467  }
468  if (description.size() == 1) {
469  write(description.iterator().next());
470  return;
471  }
472  writeOpenBracket();
473  write(junctor);
474  List<OWLClassExpression> descs = sort(description);
475  write(descs.get(0));
476  final int size = descs.size();
477  final int indent = getIndent();
478  for (int i = 1; i < size; i++) {
479  writeln();
480  writeIndent(indent);
481  write(descs.get(i));
482  }
483  writeCloseBracket();
484  }

486  @Override
487  public void visit(OWLOntology ontology1) {
488  Set<OWLClass> classes = ontology1.getClassesInSignature();
489  classes.remove(ontology1.getOWLOntologyManager().getOWLDataFactory()
490  .getOWLThing());
491  classes.remove(ontology1.getOWLOntologyManager().getOWLDataFactory()
492  .getOWLNothing());
493 
494  for (final OWLClass eachClass : sort(classes)) {
495  final boolean primitive = !eachClass.isDefined(ontology1);
496  if (primitive) {
497  writeOpenBracket();
498  write(DEFINE_PRIMITIVE_CONCEPT);
499  write(eachClass);
500  writeSpace();
501  flatten(eachClass.getSuperClasses(ontology1), KRSSVocabulary.AND);
502  writeCloseBracket();
503  writeln();
504  } else {
505  writeOpenBracket();
506  write(DEFINE_CONCEPT);
507  write(eachClass);
508  flatten(eachClass.getEquivalentClasses(ontology1), KRSSVocabulary.AND);
509  writeCloseBracket();
510  writeln();
511  }
512  }
513 
514  for (final OWLObjectProperty property : sort(ontology1
515  .getObjectPropertiesInSignature())) {
516  writeOpenBracket();
517  Set<OWLObjectPropertyExpression> properties = property
518  .getEquivalentProperties(ontology1);
519  boolean isDefined = !properties.isEmpty();
520 
521  if (isDefined) {
522  write(DEFINE_ROLE);
523  write(property);
524  //choose randomly one property (KRSS restriction)
525  properties.remove(property);
526  if (!properties.isEmpty()) {
527  write(properties.iterator().next());
528  }
529  } else {
530  write(DEFINE_PRIMITIVE_ROLE);
531  write(property);
532  writeSpace();
533  properties = property.getSuperProperties(ontology1);
534  if (!properties.isEmpty()) {
535  write(properties.iterator().next());
536  }
537  }
538  writeCloseBracket();
539  writeln();
540  }
541 
542  for (final OWLAxiom axiom : ontology1.getAxioms()) {
543  axiom.accept(this);
544  }
545 
546  try {
547  writer.flush();
548  } catch (IOException io) {
549  io.printStackTrace();
550  }
551 
552  }
