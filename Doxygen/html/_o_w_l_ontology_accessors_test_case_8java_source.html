91 @SuppressWarnings("javadoc")
92 public class OWLOntologyAccessorsTestCase extends AbstractOWLAPITestCase {
93  private void performAxiomTests(OWLOntology ont, OWLAxiom... axioms) {
94  assertEquals(ont.getAxiomCount(), axioms.length);
95  for (OWLAxiom ax : axioms) {
96  assertTrue(ont.getAxioms().contains(ax));
97  if (ax.isLogicalAxiom()) {
98  assertTrue(ont.getLogicalAxioms().contains(ax));
99  }
100  assertEquals(ont.getLogicalAxiomCount(), axioms.length);
101  AxiomType<?> axiomType = ax.getAxiomType();
102  assertTrue(ont.getAxioms(axiomType).contains(ax));
103  assertTrue(ont.getAxioms(axiomType, true).contains(ax));
104  assertEquals(ont.getAxiomCount(axiomType), axioms.length);
105  assertEquals(ont.getAxiomCount(axiomType, true), axioms.length);
106  for (OWLEntity entity : ax.getSignature()) {
107  assertTrue(ont.getReferencingAxioms(entity).contains(ax));
108  assertTrue(ont.getSignature().contains(entity));
109  }
110  }
111  }

113  @Test
114  public void testSubClassOfAxiomAccessors() {
115  OWLOntology ont = getOWLOntology("ont");
116  OWLClass clsA = Class(getIRI("A"));
117  OWLClass clsB = Class(getIRI("B"));
118  OWLObjectProperty prop = ObjectProperty(getIRI("prop"));
119  OWLOntologyManager man = ont.getOWLOntologyManager();
120  OWLSubClassOfAxiom ax = SubClassOf(clsA, clsB);
121  man.addAxiom(ont, ax);
122  OWLSubClassOfAxiom ax2 = SubClassOf(clsA, ObjectSomeValuesFrom(prop, clsB));
123  man.addAxiom(ont, ax2);
124  performAxiomTests(ont, ax, ax2);
125  assertTrue(ont.getSubClassAxiomsForSubClass(clsA).contains(ax));
126  assertTrue(ont.getSubClassAxiomsForSuperClass(clsB).contains(ax));
127  assertTrue(ont.getAxioms(clsA).contains(ax));
128  assertTrue(clsA.getSuperClasses(ont).contains(clsB));
129  assertTrue(clsB.getSubClasses(ont).contains(clsA));
130  }
