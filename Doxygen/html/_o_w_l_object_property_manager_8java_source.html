180  public Set<OWLObjectPropertyExpression> getCompositeProperties() {
181  if (compositeDirty) {
182  compositeProperties.clear();
183  compositeProperties.add(man.getOWLDataFactory().getOWLTopObjectProperty());
184  compositeProperties.add(man.getOWLDataFactory().getOWLBottomObjectProperty());
185  // We only depend on:
186  // 1) Property chain axioms
187  // 2) Transitive property axioms

189  for (OWLTransitiveObjectPropertyAxiom ax : ont.getAxioms(AxiomType.TRANSITIVE_OBJECT_PROPERTY)) {
190  markComposite(ax.getProperty());

192  markComposite(namedInv);
193  }
194  }
195  for (OWLSubPropertyChainOfAxiom ax : ont.getAxioms(AxiomType.SUB_PROPERTY_CHAIN_OF)) {
196  markComposite(ax.getSuperProperty());
197  for (OWLObjectPropertyExpression namedInv : ax.getSuperProperty().getInverses(ontology)) {
198  markComposite(namedInv);
199  }
200  }
201  }
202  compositeDirty = false;

225  public Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> getPropertyHierarchy() {
226  if (hierarchyDirty) {
227  Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> map = new HashMap<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>>();
228  // Examine: SubObjectPropertyOf
229  // EquivalentObjectProperties
230  // InverseObjectProperties
231  // SymmetricObjectProperty
232 
233  for (OWLOntology ont : getOntologies()) {
234  for (OWLSubObjectPropertyOfAxiom ax : ont.getAxioms(AxiomType.SUB_OBJECT_PROPERTY)) {
235  getKeyValue(ax.getSubProperty().getSimplified(), map).add(ax.getSuperProperty().getSimplified());
236  getKeyValue(ax.getSubProperty().getInverseProperty().getSimplified(),
237  map).add(ax.getSuperProperty().getInverseProperty().getSimplified());
238  }
239  for (OWLEquivalentObjectPropertiesAxiom ax : ont.getAxioms(AxiomType.EQUIVALENT_OBJECT_PROPERTIES)) {
240  // Do pairwise
241  for (OWLObjectPropertyExpression propA : ax.getProperties()) {
242  for (OWLObjectPropertyExpression propB : ax.getProperties()) {
243  if (!propA.equals(propB)) {
244  getKeyValue(propA.getSimplified(), map).add(propB.getSimplified());
245  getKeyValue(propB.getSimplified(), map).add(propA.getSimplified());
246  getKeyValue(propA.getInverseProperty().getSimplified(),
247  map).add(propB.getInverseProperty().getSimplified());
248  getKeyValue(propB.getInverseProperty().getSimplified(),
249  map).add(propA.getInverseProperty().getSimplified());
250  }
251  }
252  }
253  }
254  for (OWLInverseObjectPropertiesAxiom ax : ont.getAxioms(AxiomType.INVERSE_OBJECT_PROPERTIES)) {
255  getKeyValue(ax.getFirstProperty().getSimplified(),
256  map).add(ax.getSecondProperty().getInverseProperty().getSimplified());
257  getKeyValue(ax.getSecondProperty().getInverseProperty().getSimplified(), map).add(ax.getFirstProperty().getSimplified());
258  getKeyValue(ax.getFirstProperty().getInverseProperty().getSimplified(),
259  map).add(ax.getSecondProperty().getSimplified());
260  getKeyValue(ax.getSecondProperty().getSimplified(), map).add(ax.getFirstProperty().getInverseProperty().getSimplified());
261  }
262  for (OWLSymmetricObjectPropertyAxiom ax : ont.getAxioms(AxiomType.SYMMETRIC_OBJECT_PROPERTY)) {
263  getKeyValue(ax.getProperty().getSimplified(),
264  map).add(ax.getProperty().getInverseProperty().getSimplified());
265  getKeyValue(ax.getProperty().getInverseProperty().getSimplified(),
266  map).add(ax.getProperty().getSimplified());
267  }
268  }
269  hierarchy.clear();
270  hierarchy.putAll(map);
271  hierarchyDirty = false;
272  }
273  return hierarchy;
274  }

332  public Set<OWLObjectPropertyExpression> getNonSimpleProperties() {
333  if (simpleDirty) {
334  nonSimpleProperties.clear();
335  Set<OWLObjectPropertyExpression> props = getReferencedProperties();
336  Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> reflexiveTransitiveClosureMap = getHierarchyReflexiveTransitiveClosure();
337  for (OWLObjectPropertyExpression prop : getReferencedProperties()) {
338  if (isComposite(prop)) {
339  // Supers are not simple
340  Set<OWLObjectPropertyExpression> rtc = reflexiveTransitiveClosureMap.get(prop);
341  props.removeAll(rtc);
342  nonSimpleProperties.add(prop);
343  nonSimpleProperties.addAll(rtc);
344  }
345  }
346  for (OWLObjectPropertyExpression prop : new ArrayList<OWLObjectPropertyExpression>(nonSimpleProperties)) {
347  nonSimpleProperties.add(prop.getInverseProperty().getSimplified());
348  }
349  simpleDirty = false;
350  }

357  public Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> getPropertyPartialOrdering() {
358  if (partialOrderingDirty) {
359  partialOrdering.clear();
360  Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> map = new HashMap<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>>(getPropertyHierarchy());
361  for (OWLOntology ont : getOntologies()) {
362  for (OWLSubPropertyChainOfAxiom ax : ont.getAxioms(AxiomType.SUB_PROPERTY_CHAIN_OF)) {
363  for (OWLObjectPropertyExpression prop : ax.getPropertyChain()) {
364  Set<OWLObjectPropertyExpression> sups = map.get(prop.getSimplified());
365  if (sups == null) {
366  sups = new HashSet<OWLObjectPropertyExpression>();
367  map.put(prop, sups);
368  }
369  sups.add(ax.getSuperProperty().getSimplified());
370  Set<OWLObjectPropertyExpression> supsInv = map.get(prop.getInverseProperty().getSimplified());
371  if (supsInv == null) {
372  supsInv = new HashSet<OWLObjectPropertyExpression>();
373  map.put(prop.getInverseProperty().getSimplified(), supsInv);
374  }
375  supsInv.add(ax.getSuperProperty().getInverseProperty().getSimplified());
376  }
377  }
378  }
379  Map<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>> ordering = new HashMap<OWLObjectPropertyExpression, Set<OWLObjectPropertyExpression>>();
380  for (OWLObjectPropertyExpression prop : getReferencedProperties()) {
381  Set<OWLObjectPropertyExpression> processed = new HashSet<OWLObjectPropertyExpression>();
382  Set<OWLObjectPropertyExpression> rtc = new HashSet<OWLObjectPropertyExpression>();
383  getReflexiveTransitiveClosure(prop, map, rtc, processed);

421  public static Collection<Set<OWLObjectPropertyExpression>> getEquivalentObjectProperties(Set<OWLOntology> ontologies) {
422  Set<Set<OWLObjectPropertyExpression>> result = new HashSet<Set<OWLObjectPropertyExpression>>();
423 
424  Set<OWLObjectPropertyExpression> processed = new HashSet<OWLObjectPropertyExpression>();
425  Set<OWLObjectPropertyExpression> properties = new HashSet<OWLObjectPropertyExpression>();
426  for (OWLOntology ont : ontologies) {
427  properties.addAll(ont.getObjectPropertiesInSignature());
428  }
429 
430 
431  for (OWLObjectPropertyExpression prop : properties) {
432  if (processed.contains(prop)) {
433  continue;
434  }
435  tarjan(ontologies,
436  prop,
437  0,
438  new Stack<OWLObjectPropertyExpression>(),
439  new HashMap<OWLObjectPropertyExpression, Integer>(),
440  new HashMap<OWLObjectPropertyExpression, Integer>(),
441  result,
442  processed,
443  new HashSet<OWLObjectPropertyExpression>());
444  }
445  // Get maximal
446  List<Set<OWLObjectPropertyExpression>> equivs = new ArrayList<Set<OWLObjectPropertyExpression>>(result);
447  Collections.sort(equivs, new SetSizeComparator());
448  for (int i = 0; i < equivs.size(); i++) {
449  Set<OWLObjectPropertyExpression> first = equivs.get(i);
450  for (int j = i; j < equivs.size(); j++) {
451  Set<OWLObjectPropertyExpression> second = equivs.get(j);
452  if (second.size() > first.size() && second.containsAll(first)) {
453  equivs.remove(i);
454  break;
455  }
456  }
457  }
458  return equivs;
459  }

480  @SuppressWarnings("boxing")
481  public static void tarjan(Set<OWLOntology> ontologies, OWLObjectPropertyExpression prop, int index,
482  Stack<OWLObjectPropertyExpression> stack,
483  Map<OWLObjectPropertyExpression, Integer> indexMap,
484  Map<OWLObjectPropertyExpression, Integer> lowlinkMap,
485  Set<Set<OWLObjectPropertyExpression>> result,
486  Set<OWLObjectPropertyExpression> processed,
487  Set<OWLObjectPropertyExpression> stackProps) {
488  processed.add(prop);
489  indexMap.put(prop, index);
490  lowlinkMap.put(prop, index);
491  index = index + 1;
492  stack.push(prop);
493  stackProps.add(prop);
494  for (OWLOntology ont : ontologies) {
495  for (OWLSubObjectPropertyOfAxiom ax : ont.getObjectSubPropertyAxiomsForSubProperty(prop)) {
496  if (ax.getSubProperty().equals(prop)) {
497  OWLObjectPropertyExpression supProp = ax.getSuperProperty();
498  if (!indexMap.containsKey(supProp)) {
499  tarjan(ontologies, supProp, index, stack, indexMap, lowlinkMap, result, processed, stackProps);
500  lowlinkMap.put(prop, Math.min(lowlinkMap.get(prop), lowlinkMap.get(supProp)));
501  } else if (stackProps.contains(supProp)) {
502  lowlinkMap.put(prop, Math.min(lowlinkMap.get(prop), indexMap.get(supProp)));
503  }
504  }
505  }
506  }
507  if (lowlinkMap.get(prop).equals(indexMap.get(prop))) {
508  Set<OWLObjectPropertyExpression> scc = new HashSet<OWLObjectPropertyExpression>();
509  OWLObjectPropertyExpression propPrime;
510  while (true) {
511  propPrime = stack.pop();
512  stackProps.remove(propPrime);
513  scc.add(propPrime);
514  if (propPrime.equals(prop)) {
515  break;
516  }
517  }
518  if (scc.size() > 1) {
519  result.add(scc);
520  }
521  }
522  }
