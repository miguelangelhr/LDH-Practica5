161  private class BottomReplacer extends OWLAxiomVisitorAdapter implements
162  OWLAxiomVisitor, OWLClassExpressionVisitor {
163  private OWLAxiom newAxiom;
164  private OWLClassExpression newClassExpression;
165  private Set<? extends OWLEntity> signature;

167  public BottomReplacer() {}

169  public OWLAxiom getResult() {
170  return newAxiom;

173  public OWLAxiom replaceBottom(OWLAxiom axiom, Set<? extends OWLEntity> sig) {
174  reset(sig);
175  axiom.accept(this);
176  return getResult();

179  // Takes an OWLClassExpression and a signature replaces by bottom the
180  // entities not in the signature
181  public OWLClassExpression replaceBottom(OWLClassExpression desc) {
182  newClassExpression = null;
183  desc.accept(this);
184  if (newClassExpression == null) {
185  throw new RuntimeException("Unsupported class expression " + desc);
186  }
187  return newClassExpression;
188  }
189 
190  public Set<OWLClassExpression> replaceBottom(
191  Set<OWLClassExpression> classExpressions) {
192  Set<OWLClassExpression> result = new HashSet<OWLClassExpression>();
193  for (OWLClassExpression desc : classExpressions) {
194  result.add(replaceBottom(desc));
195  }
196  return result;
197  }
198 
199  public void reset(Set<? extends OWLEntity> s) {
200  signature = s;
201  newAxiom = null;
202  }

364  @Override
365  public boolean isLocal(OWLAxiom axiom, Set<? extends OWLEntity> signature) {
366  if (log.isLoggable(Level.FINE)) {
367  log.fine("Replacing axiom by Bottom");
368  }
369  OWLAxiom newAxiom = bottomReplacer.replaceBottom(axiom, signature);
370  if (log.isLoggable(Level.FINE)) {
371  log.fine("DONE Replacing axiom by Bottom. Success: " + (newAxiom != null));
372  }
373  return newAxiom != null && axiomVisitor.isLocal(newAxiom);
