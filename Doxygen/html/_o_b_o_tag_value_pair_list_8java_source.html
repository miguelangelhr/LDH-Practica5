104  public void addPair(IRI tag, String value) {
105  boolean found = false;
106  for (OBOVocabulary obo : vocab) {
107  if (tag.equals(obo.getIRI())) {
108  addPair(obo.getName(), value, knownTVPs);
109  found = true;
110  break;
111  }
112  }
113  if (!found) {
114  final String name = iriSFP.getShortForm(tag);
115  addPair(name, value, unknownTVPs);
116  }
117  }

155  public void write(Writer w) {
156  writer = w;
157 
158  // write tags out in order
159  for (OBOVocabulary tag : vocab) {
160  Set<String> values = knownTVPs.get(tag.getName());
161  if (values == null) {
162  String def = defaults.get(tag.getIRI());
163  if (def != null) {
164  values = Collections.singleton(def);
165  }
166  }
167  if (values != null) {
168  for (String value : values) {
169  writeTagValuePair(tag, value);
170  }
171  }
172  }
173 
174  // write additional tags in no specified order
175  for (String unknownTag : unknownTVPs.keySet()) {
176  for (String value : unknownTVPs.get(unknownTag)) {
177  writeTagValuePair(unknownTag, value);
178  }
179  }
180  }
