302  // an OWLAxiomImpl will implement this interface with <OWLAnnotation > parameter; this will avoid creating a defensive copy of the annotation set
303  if (ax instanceof CollectionContainer) {
304  ((CollectionContainer<OWLAnnotation>) ax).accept(annotationVisitor);

306  // default behavior: iterate over the annotations outside the axiom
