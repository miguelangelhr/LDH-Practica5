131  @Override
132  public OWLProfileReport checkOntology(OWLOntology ontology) {
133  OWL2DLProfile profile = new OWL2DLProfile();
134  OWLProfileReport report = profile.checkOntology(ontology);
135  Set<OWLProfileViolation> violations = new HashSet<OWLProfileViolation>();
136  violations.addAll(report.getViolations());
137  OWLOntologyWalker ontologyWalker = new OWLOntologyWalker(
138  ontology.getImportsClosure());
139  OWL2ELProfileObjectVisitor visitor = new OWL2ELProfileObjectVisitor(
140  ontologyWalker, ontology.getOWLOntologyManager());
141  ontologyWalker.walkStructure(visitor);
142  violations.addAll(visitor.getProfileViolations());
143  return new OWLProfileReport(this, violations);
144  }

365  @Override
366  public Object visit(OWLSubPropertyChainOfAxiom axiom) {
367  Set<OWLObjectPropertyRangeAxiom> rangeAxioms = getCurrentOntology()
368  .getAxioms(AxiomType.OBJECT_PROPERTY_RANGE, true);
369  if (rangeAxioms.isEmpty()) {
370  return Boolean.FALSE;
371  }
372  // Do we have a range restriction imposed on our super property?
373  for (OWLObjectPropertyRangeAxiom rngAx : rangeAxioms) {
374  if (getPropertyManager().isSubPropertyOf(axiom.getSuperProperty(),
375  rngAx.getProperty())) {
376  // Imposed range restriction!
377  OWLClassExpression imposedRange = rngAx.getRange();
378  // There must be an axiom that imposes a range on the last
379  // prop in the chain
380  List<OWLObjectPropertyExpression> chain = axiom.getPropertyChain();
381  if (!chain.isEmpty()) {
382  OWLObjectPropertyExpression lastProperty = chain
383  .get(chain.size() - 1);
384  boolean rngPresent = false;
385  for (OWLOntology ont : getCurrentOntology().getImportsClosure()) {
386  for (OWLObjectPropertyRangeAxiom lastPropRngAx : ont
387  .getObjectPropertyRangeAxioms(lastProperty)) {
388  if (lastPropRngAx.getRange().equals(imposedRange)) {
389  // We're o.k.
390  rngPresent = true;
391  break;
392  }
393  }
394  }
395  if (!rngPresent) {
396  profileViolations
397  .add(new LastPropertyInChainNotInImposedRange(
398  getCurrentOntology(), axiom, rngAx));
399  }
400  }
401  }
402  }
403  return null;
404  }
