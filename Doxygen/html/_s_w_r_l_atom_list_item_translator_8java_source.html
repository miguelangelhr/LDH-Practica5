83  @Override
84  public SWRLAtom translate(IRI firstObject) {
85  if (consumer.isSWRLBuiltInAtom(firstObject)) {
86  IRI builtInIRI = consumer.getResourceObject(firstObject, BUILT_IN.getIRI(), true);
87  IRI mainIRI = consumer.getResourceObject(firstObject, ARGUMENTS.getIRI(), true);
88  OptimisedListTranslator<SWRLDArgument> listTranslator = new OptimisedListTranslator<SWRLDArgument>(
89  consumer,
90  new SWRLAtomDObjectListItemTranslator());
91  List<SWRLDArgument> args = listTranslator.translateList(mainIRI);
92  return dataFactory.getSWRLBuiltInAtom(builtInIRI, args);
93  }
94  else if (consumer.isSWRLClassAtom(firstObject)) {
95  // C(?x) or C(ind)
96  SWRLIArgument iObject = translateSWRLAtomIObject(firstObject, ARGUMENT_1.getIRI());
97  IRI classIRI = consumer.getResourceObject(firstObject, CLASS_PREDICATE.getIRI(), true);
98  if (classIRI == null) {
99  throw new OWLRuntimeException(
100  "Don't know how to translate SWRL Atom: class IRI is null "
101  + firstObject);
102  }
103  OWLClassExpression desc = consumer.translateClassExpression(classIRI);
104  return dataFactory.getSWRLClassAtom(desc, iObject);
105  }
106  else if (consumer.isSWRLDataRangeAtom(firstObject)) {
107  // DR(?x) or DR(val)
108  SWRLDArgument dObject = translateSWRLAtomDObject(firstObject, ARGUMENT_1.getIRI());
109  IRI dataRangeIRI = consumer.getResourceObject(firstObject, DATA_RANGE.getIRI(), true);
110  if (dataRangeIRI == null) {
111  throw new OWLRuntimeException(
112  "Don't know how to translate SWRL Atom: data range IRI is null "
113  + firstObject);
114  }
115  OWLDataRange dataRange = consumer.translateDataRange(dataRangeIRI);
116  return dataFactory.getSWRLDataRangeAtom(dataRange, dObject);
117  }
118  else if (consumer.isSWRLDataValuedPropertyAtom(firstObject)) {
119  SWRLIArgument arg1 = translateSWRLAtomIObject(firstObject, ARGUMENT_1.getIRI());
120  SWRLDArgument arg2 = translateSWRLAtomDObject(firstObject, ARGUMENT_2.getIRI());
121  IRI dataPropertyIRI = consumer.getResourceObject(firstObject, PROPERTY_PREDICATE.getIRI(), true);
122  if (dataPropertyIRI == null) {
123  throw new OWLRuntimeException(
124  "Don't know how to translate SWRL Atom: data property IRI is null "
125  + firstObject);
126  }
127  OWLDataPropertyExpression prop = consumer.translateDataPropertyExpression(dataPropertyIRI);
128  return dataFactory.getSWRLDataPropertyAtom(prop, arg1, arg2);
129  }
130  else if (consumer.isSWRLIndividualPropertyAtom(firstObject)) {
131  SWRLIArgument arg1 = translateSWRLAtomIObject(firstObject, ARGUMENT_1.getIRI());
132  SWRLIArgument arg2 = translateSWRLAtomIObject(firstObject, ARGUMENT_2.getIRI());
133  IRI objectPropertyIRI = consumer.getResourceObject(firstObject, PROPERTY_PREDICATE.getIRI(), true);
134  if (objectPropertyIRI == null) {
135  throw new OWLRuntimeException(
136  "Don't know how to translate SWRL Atom: object property IRI is null "
137  + firstObject);
138  }
139  OWLObjectPropertyExpression prop = consumer.translateObjectPropertyExpression(objectPropertyIRI);
140  return dataFactory.getSWRLObjectPropertyAtom(prop, arg1, arg2);
141  }
142  else if (consumer.isSWRLSameAsAtom(firstObject)) {
143  SWRLIArgument arg1 = translateSWRLAtomIObject(firstObject, ARGUMENT_1.getIRI());
144  SWRLIArgument arg2 = translateSWRLAtomIObject(firstObject, ARGUMENT_2.getIRI());
145  return dataFactory.getSWRLSameIndividualAtom(arg1, arg2);
146  }
147  else if (consumer.isSWRLDifferentFromAtom(firstObject)) {
148  SWRLIArgument arg1 = translateSWRLAtomIObject(firstObject, ARGUMENT_1.getIRI());
149  SWRLIArgument arg2 = translateSWRLAtomIObject(firstObject, ARGUMENT_2.getIRI());
150  return dataFactory.getSWRLDifferentIndividualsAtom(arg1, arg2);
151  }
152  throw new OWLRuntimeException("Don't know how to translate SWRL Atom: "
153  + firstObject);
154  }

163  private SWRLIArgument translateSWRLAtomIObject(IRI mainIRI, IRI argPredicateIRI) {
164  IRI argIRI = consumer.getResourceObject(mainIRI, argPredicateIRI, true);
165  if (argIRI != null) {
166  if (consumer.isSWRLVariable(argIRI)) {
167  return dataFactory.getSWRLVariable(argIRI);
168  }
169  else {
170  return dataFactory.getSWRLIndividualArgument(consumer.getOWLIndividual(argIRI));
171  }
172  }
173  else {
174  throw new OWLRuntimeException("Cannot translate SWRL Atom I-Object for "
175  + argPredicateIRI + " Triple not found.");
176  }
177  }
178 
179 
180  private SWRLDArgument translateSWRLAtomDObject(IRI mainIRI, IRI argPredicateIRI) {
181  IRI argIRI = consumer.getResourceObject(mainIRI, argPredicateIRI, true);
182  if (argIRI != null) {
183  // Must be a variable -- double check
184  if (!consumer.isSWRLVariable(argIRI)) {
185  logger.info("Expected SWRL variable for SWRL Data Object: " + argIRI + "(possibly untyped)");
186  }
187  return dataFactory.getSWRLVariable(argIRI);
188  }
189  else {
190  // Must be a literal
191  OWLLiteral con = consumer.getLiteralObject(mainIRI, argPredicateIRI, true);
192  if (con != null) {
193  return dataFactory.getSWRLLiteralArgument(con);
194  }
195  }
196  throw new IllegalStateException("Could not translate SWRL Atom D-Object");
197  }
