109  public JustificationMap(OWLClassExpression desc, Set<OWLAxiom> axioms) {
110  this.axioms = axioms;
111  this.desc = desc;
112  rootAxioms = new HashSet<OWLAxiom>();
113  map = new HashMap<OWLAxiom, Set<OWLAxiom>>();
114  usedAxioms = new HashSet<OWLAxiom>();
115  axiomsByRHS = new HashMap<OWLEntity, Set<OWLAxiom>>();
116  axiomsByLHS = new HashMap<OWLEntity, Set<OWLAxiom>>();
117  createMap();
118  }

120  private void createMap() {
121  for (OWLAxiom ax : axioms) {
122  OWLAxiomPartExtractor extractor = new OWLAxiomPartExtractor();
123  ax.accept(extractor);
124  Set<OWLEntity> rhscollected = new HashSet<OWLEntity>();
125  OWLEntityCollector rhsCollector = new OWLEntityCollector(rhscollected);
126  for (OWLObject rhsObject : extractor.getRHS()) {
127  rhsObject.accept(rhsCollector);
128  }
129  for (OWLEntity rhsEntity : rhscollected) {
130  index(rhsEntity, axiomsByRHS, ax);
131  }
132  Set<OWLEntity> lhscollected = new HashSet<OWLEntity>();
133  OWLEntityCollector lhsCollector = new OWLEntityCollector(lhscollected);
134  for (OWLObject lhsObject : extractor.getLHS()) {
135  lhsObject.accept(lhsCollector);
136  }
137  for (OWLEntity lhsEntity : lhscollected) {
138  index(lhsEntity, axiomsByLHS, ax);
139  }
140  }
141  buildChildren(desc);
142  }

144  private Set<OWLAxiom> getAxiomsByLHS(OWLEntity lhs) {
145  Set<OWLAxiom> axiomSet = axiomsByLHS.get(lhs);
146  if (axiomSet != null) {
147  Set<OWLAxiom> ts = new TreeSet<OWLAxiom>(new OWLAxiomComparator());
148  ts.addAll(axiomSet);
149  return ts;
150  } else {
151  return Collections.emptySet();
152  }
153  }

155  private void buildChildren(OWLClassExpression seed) {
156  // Return the axioms that have the entity on the LHS
157  Set<OWLAxiom> result = new HashSet<OWLAxiom>();
158  for (OWLEntity ent : seed.getSignature()) {
159  Set<OWLAxiom> axs = getAxiomsByLHS(ent);
160  for (OWLAxiom ax : axs) {
161  result.add(ax);
162  usedAxioms.add(ax);
163  }
164  }
165  rootAxioms.addAll(result);
166  buildChildren(result);
167  }
168 
169  private void buildChildren(Set<OWLAxiom> axiomSet) {
170  List<Set<OWLAxiom>> axiomChildren = new ArrayList<Set<OWLAxiom>>();
171  for (OWLAxiom ax : axiomSet) {
172  Set<OWLAxiom> children = build(ax);
173  for (OWLAxiom childAx : children) {
174  index(ax, map, childAx);
175  }
176  axiomChildren.add(children);
177  }
178  for (Set<OWLAxiom> children : axiomChildren) {
179  buildChildren(children);
180  }
181  }

183  private Set<OWLAxiom> build(OWLAxiom parentAxiom) {
184  usedAxioms.add(parentAxiom);
185  OWLAxiomPartExtractor extractor = new OWLAxiomPartExtractor();
186  parentAxiom.accept(extractor);
187  Set<OWLAxiom> result = new HashSet<OWLAxiom>();
188  for (OWLObject obj : extractor.getRHS()) {
189  for (OWLEntity ent : obj.getSignature()) {
190  Set<OWLAxiom> axs = getAxiomsByLHS(ent);
191  for (OWLAxiom ax : axs) {
192  if (!usedAxioms.contains(ax)) {
193  result.add(ax);
194  usedAxioms.add(ax);
195  }
196  }
197  }
198  }
199  return result;
200  }

202  private static <K, V> void index(K key, Map<K, Set<V>> map, V value) {
203  Set<V> values = map.get(key);
204  if (values == null) {
205  values = new HashSet<V>();
206  map.put(key, values);
207  }
208  values.add(value);
209  }

431  private static class OWLAxiomComparator extends OWLAxiomVisitorAdapter implements
432  Comparator<OWLAxiom>, Serializable {
433  private static final long serialVersionUID = 30406L;
434 
435  public OWLAxiomComparator() {}

437  @Override
438  public void visit(OWLSubClassOfAxiom axiom) {
439  result = 0;
440  }
441 
442  @Override
443  public void visit(OWLEquivalentClassesAxiom axiom) {
444  result = 1;

452  private int result;
453 
454  @Override
455  public int compare(OWLAxiom o1, OWLAxiom o2) {
456  result = 0;
457  o1.accept(this);
458  int result1 = result;
459  o2.accept(this);
460  int result2 = result;
461  int diff = result2 - result1;
462  if (diff != 0) {
463  return diff;
464  } else {
465  return -1;
466  }
467  }
