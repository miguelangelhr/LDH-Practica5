115  @Override
116  public void setDocumentLocator(Locator locator) {
117  super.setDocumentLocator(locator);
118  this.locator = locator;
119  URI base = null;
120  try {
121  String systemId = locator.getSystemId();
122  if (systemId != null) {
123  base = new URI(systemId);
124  }
125  } catch (URISyntaxException e) {}
126  bases.push(base);
127  }
