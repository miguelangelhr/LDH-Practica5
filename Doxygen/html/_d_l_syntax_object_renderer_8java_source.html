166  protected void write(Collection<? extends OWLObject> objects, DLSyntax delim, boolean nest) {
167  if (objects.size() == 2) {
168  Iterator<? extends OWLObject> it = objects.iterator();
169  OWLObject o1 = it.next();
170  OWLObject o2 = it.next();
171  if (isFocusedObject(o1) || !isFocusedObject(o2)) {
172  writeObject(o1, nest);
173  writeSpace();
174  write(delim);
175  writeSpace();
176  writeObject(o2, nest);
177  } else {
178  writeObject(o2, nest);
179  writeSpace();
180  write(delim);
181  writeSpace();
182  writeObject(o1, nest);
183  }
184  } else {
185  for (Iterator<? extends OWLObject> it = objects.iterator(); it.hasNext();) {
186  OWLObject o = it.next();
187  writeObject(o, nest);
188  if (it.hasNext()) {
189  writeSpace();
190  write(delim);
191  writeSpace();
192  }
193  }
194  }
195 
196  }

522  @Override
523  public void visit(OWLInverseObjectPropertiesAxiom axiom) {
524  OWLObject o1 = axiom.getFirstProperty();

526

541  write(INVERSE);
