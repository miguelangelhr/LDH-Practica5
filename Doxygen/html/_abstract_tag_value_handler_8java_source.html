163  protected OWLClassExpression getOWLClassOrRestriction(String termList) {
164  StringTokenizer tok = new StringTokenizer(termList, " ", false);
165  String id0 = null;
166  String id1 = null;
167  id0 = tok.nextToken();
168  if (tok.hasMoreTokens()) {
169  id1 = tok.nextToken();
170  }
171  if (id1 == null) {
172  return getDataFactory().getOWLClass(getIRIFromOBOId(id0));
173  } else {
174  IRI propertyIRI = getConsumer().getRelationIRIFromSymbolicIdOrOBOId(id0);
175  OWLObjectProperty prop = getDataFactory().getOWLObjectProperty(propertyIRI);
176  OWLClass filler = getDataFactory().getOWLClass(getIRIFromOBOId(id1));
177  return getDataFactory().getOWLObjectSomeValuesFrom(prop, filler);
178  }
179  }
