211  // Get hold of an ontology manager

222  manager.saveOntology(pizzaOntology, IRI.create(file.toURI()));
223  // By default ontologies are saved in the format from which they were
224  // loaded. In this case the ontology was loaded from an rdf/xml file We
225  // can get information about the format of an ontology from its manager
226  OWLOntologyFormat format = manager.getOntologyFormat(pizzaOntology);
227  // We can save the ontology in a different format Lets save the ontology
228  // in owl/xml format
229  OWLXMLOntologyFormat owlxmlFormat = new OWLXMLOntologyFormat();
230  // Some ontology formats support prefix names and prefix IRIs. In our
231  // case we loaded the pizza ontology from an rdf/xml format, which
232  // supports prefixes. When we save the ontology in the new format we
233  // will copy the prefixes over so that we have nicely abbreviated IRIs
234  // in the new ontology document
235  if (format.isPrefixOWLOntologyFormat()) {
236  owlxmlFormat.copyPrefixesFrom(format.asPrefixOWLOntologyFormat());
237  }
238  manager.saveOntology(pizzaOntology, owlxmlFormat, IRI.create(file.toURI()));
239  // We can also dump an ontology to System.out by specifying a different
240  // OWLOntologyOutputTarget Note that we can write an ontology to a
241  // stream in a similar way using the StreamOutputTarget class

243  OWLOntologyDocumentTarget documentTarget = new SystemOutDocumentTarget();
244  // Try another format - The Manchester OWL Syntax
245  ManchesterOWLSyntaxOntologyFormat manSyntaxFormat = new ManchesterOWLSyntaxOntologyFormat();

247  manSyntaxFormat.copyPrefixesFrom(format.asPrefixOWLOntologyFormat());
248  }
249  manager.saveOntology(pizzaOntology, manSyntaxFormat, new StreamDocumentTarget(

258  public void shouldAccessEntities() throws OWLOntologyCreationException {
259  // In order to get access to objects that represent entities we need a
260  // data factory.
261  OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
262  // We can get a reference to a data factory from an OWLOntologyManager.
263  OWLDataFactory factory = manager.getOWLDataFactory();
264  // In OWL, entities are named objects that are used to build class
265  // expressions and axioms. They include classes, properties (object,
266  // data and annotation), named individuals and datatypes. All entities
267  // may be obtained from an OWLDataFactory. Let's create an object to
268  // represent a class. In this case, we'll choose
269  // http://www.semanticweb.org/owlapi/ontologies/ontology#A as the IRI
270  // for our class. There are two ways we can create classes (and other
271  // entities). The first is by specifying the full IRI. First we create
272  // an IRI object:
273  IRI iri = IRI.create("http://www.semanticweb.org/owlapi/ontologies/ontology#A");
274  // Now we create the class
275  OWLClass clsAMethodA = factory.getOWLClass(iri);
276  // The second is to use a prefix manager and specify abbreviated IRIs.
277  // This is useful for creating lots of entities with the same prefix
278  // IRIs. First create our prefix manager and specify that the default
279  // prefix IRI (bound to the empty prefix name) is
280  // http://www.semanticweb.org/owlapi/ontologies/ontology#
281  PrefixManager pm = new DefaultPrefixManager(
282  "http://www.semanticweb.org/owlapi/ontologies/ontology#");
283  // Now we use the prefix manager and just specify an abbreviated IRI
284  @SuppressWarnings("unused")
285  OWLClass clsAMethodB = factory.getOWLClass(":A", pm);
286  // Note that clsAMethodA will be equal to clsAMethodB because they are
287  // both OWLClass objects and have the same IRI. Creating entities in the
288  // above manner does not "add them to an ontology". They are merely
289  // objects that allow us to reference certain objects (classes etc.) for
290  // use in class expressions, and axioms (which can be added to an
291  // ontology). Lets create an ontology, and add a declaration axiom to
292  // the ontology that declares the above class
293  OWLOntology ontology = manager.createOntology(IRI
294  .create("http://www.semanticweb.org/owlapi/ontologies/ontology"));
295  // We can add a declaration axiom to the ontology, that essentially adds
296  // the class to the signature of our ontology.
297  OWLDeclarationAxiom declarationAxiom = factory
298  .getOWLDeclarationAxiom(clsAMethodA);
299  manager.addAxiom(ontology, declarationAxiom);
300  // Note that it isn't necessary to add declarations to an ontology in
301  // order to use an entity. For some ontology formats (e.g. the
302  // Manchester Syntax), declarations will automatically be added in the
303  // saved version of the ontology.
304  }

312  public void shouldBuildDataRanges() throws OWLOntologyCreationException,
313  OWLOntologyStorageException {
314  // OWLDataRange is the superclass of all data ranges in the OWL API.
315  // Data ranges are used as the types of literals, as the ranges for data
316  // properties, as filler for data reatrictions. Get hold of a manager to
317  // work with
318  OWLOntologyManager manager = OWLManager.createOWLOntologyManager();
319  OWLDataFactory factory = manager.getOWLDataFactory();
320  // OWLDatatype represents named datatypes in OWL. These are a bit like
321  // classes whose instances are data values OWLDatatype objects are
322  // obtained from a data factory. The OWL2Datatype enum defines built in
323  // OWL 2 Datatypes Get hold of the integer datatype
324  OWLDatatype integer = factory.getOWLDatatype(OWL2Datatype.XSD_INTEGER.getIRI());
325  // For common data types there are some convenience methods of
326  // OWLDataFactory. For example
327  OWLDatatype integerDatatype = factory.getIntegerOWLDatatype();
328  OWLDatatype floatDatatype = factory.getFloatOWLDatatype();
329  OWLDatatype doubleDatatype = factory.getDoubleOWLDatatype();
330  OWLDatatype booleanDatatype = factory.getBooleanOWLDatatype();
331  // The top datatype (analgous to owl:Thing) is rdfs:Literal, which can
332  // be obtained from the data factory
333  OWLDatatype rdfsLiteral = factory.getTopDatatype();
334  // Custom data ranges can be built up from these basic datatypes. For
335  // example, it is possible to restrict a datatype using facets from XML
336  // Schema Datatypes. For example, lets create a data range that
337  // describes integers that are greater or equal to 18 To do this, we
338  // restrict the xsd:integer datatype using the xsd:minInclusive facet
339  // with a value of 18. Get hold of a literal that is an integer value 18
340  OWLLiteral eighteen = factory.getOWLLiteral(18);
341  // Now create the restriction. The OWLFacet enum provides an enumeration
342  // of the various facets that can be used
343  OWLDatatypeRestriction integerGE18 = factory.getOWLDatatypeRestriction(integer,
344  OWLFacet.MIN_INCLUSIVE, eighteen);
345  // We could use this datatype in restriction, as the range of data
346  // properties etc. For example, if we want to restrict the range of the
347  // :hasAge data property to 18 or more we specify its range as this data
348  // range
349  PrefixManager pm = new DefaultPrefixManager(
350  "http://www.semanticweb.org/ontologies/dataranges#");
351  OWLDataProperty hasAge = factory.getOWLDataProperty(":hasAge", pm);
352  OWLDataPropertyRangeAxiom rangeAxiom = factory.getOWLDataPropertyRangeAxiom(
353  hasAge, integerGE18);
354  OWLOntology ontology = manager.createOntology(IRI
355  .create("http://www.semanticweb.org/ontologies/dataranges"));
356  // Add the range axiom to our ontology
357  manager.addAxiom(ontology, rangeAxiom);
358  // For creating datatype restrictions on integers or doubles there are
359  // some convenience methods on OWLDataFactory For example: Create a data
360  // range of integers greater or equal to 60
361  OWLDatatypeRestriction integerGE60 = factory
362  .getOWLDatatypeMinInclusiveRestriction(60);
363  // Create a data range of integers less than 16
364  OWLDatatypeRestriction integerLT16 = factory
365  .getOWLDatatypeMaxExclusiveRestriction(18);
366  // In OWL 2 it is possible to represent the intersection, union and
367  // complement of data types For example, we could create a union of data
368  // ranges of the data range integer less than 16 or integer greater or
369  // equal to 60
370  OWLDataUnionOf concessionaryAge = factory.getOWLDataUnionOf(integerLT16,
371  integerGE60);
372  // We can also coin names for custom data ranges. To do this we use an
373  // OWLDatatypeDefintionAxiom Get hold of a named datarange (datatype)
374  // that will be used to assign a name to our above datatype
375  OWLDatatype concessionaryAgeDatatype = factory.getOWLDatatype(
376  ":ConcessionaryAge", pm);
377  // Now create a datatype definition axiom
378  OWLDatatypeDefinitionAxiom datatypeDef = factory.getOWLDatatypeDefinitionAxiom(
379  concessionaryAgeDatatype, concessionaryAge);
380  // Add the definition to our ontology
381  manager.addAxiom(ontology, datatypeDef);
382  // Dump our ontology
383  manager.saveOntology(ontology, new StreamDocumentTarget(
384  new ByteArrayOutputStream()));
385  }

394  public void shouldUseDataranges() throws OWLOntologyCreationException {
395  OWLOntologyManager man = OWLManager.createOWLOntologyManager();
396  String base = "http://org.semanticweb.datarangeexample";
397  OWLOntology ont = man.createOntology(IRI.create(base));
398  // We want to add an axiom to our ontology that states that adults have
399  // an age greater than 18. To do this, we will create a restriction
400  // along a hasAge property, with a filler that corresponds to the set of
401  // integers greater than 18. First get a reference to our hasAge
402  // property
403  OWLDataFactory factory = man.getOWLDataFactory();
404  OWLDataProperty hasAge = factory.getOWLDataProperty(IRI.create(base + "hasAge"));
405  // For completeness, we will make hasAge functional by adding an axiom
406  // to state this
407  OWLFunctionalDataPropertyAxiom funcAx = factory
408  .getOWLFunctionalDataPropertyAxiom(hasAge);
409  man.applyChange(new AddAxiom(ont, funcAx));
410  // Now create the data range which correponds to int greater than 18. To
411  // do this, we get hold of the int datatype and then restrict it with a
412  // minInclusive facet restriction.
413  OWLDatatype intDatatype = factory.getIntegerOWLDatatype();
414  // Create the value "18", which is an int.
415  OWLLiteral eighteenConstant = factory.getOWLLiteral(18);
416  // Now create our custom datarange, which is int greater than or equal
417  // to 18. To do this, we need the minInclusive facet
418  OWLFacet facet = OWLFacet.MIN_INCLUSIVE;
419  // Create the restricted data range by applying the facet restriction
420  // with a value of 18 to int
421  OWLDataRange intGreaterThan18 = factory.getOWLDatatypeRestriction(intDatatype,
422  facet, eighteenConstant);
423  // Now we can use this in our datatype restriction on hasAge
424  OWLClassExpression thingsWithAgeGreaterOrEqualTo18 = factory
425  .getOWLDataSomeValuesFrom(hasAge, intGreaterThan18);
426  // Now we want to say all adults have an age that is greater or equal to
427  // 18 - i.e. Adult is a subclass of hasAge some int[>= 18] Obtain a
428  // reference to the Adult class
429  OWLClass adult = factory.getOWLClass(IRI.create(base + "#Adult"));
430  // Now make adult a subclass of the things that have an age greater to
431  // or equal to 18
432  OWLSubClassOfAxiom ax = factory.getOWLSubClassOfAxiom(adult,
433  thingsWithAgeGreaterOrEqualTo18);
434  // Add our axiom to the ontology
435  man.applyChange(new AddAxiom(ont, ax));
436  }
