80  public OWLObjectRenderer(OWLOntology ontology, Writer writer) {
81  this.ontology = ontology;
82  this.writer = writer;
83  writeEnitiesAsURIs = true;
84  prefixManager = new DefaultPrefixManager();
85  OWLOntologyFormat ontologyFormat = ontology.getOWLOntologyManager()
86  .getOntologyFormat(ontology);
87  if (ontologyFormat instanceof PrefixOWLOntologyFormat) {
88  PrefixOWLOntologyFormat prefixFormat = (PrefixOWLOntologyFormat) ontologyFormat;
89  for (String prefixName : prefixFormat.getPrefixNames()) {
90  String prefix = prefixFormat.getPrefix(prefixName);
91  prefixManager.setPrefix(prefixName, prefix);
92  }
93  }
94  if (!ontology.isAnonymous()) {
95  String defPrefix = ontology.getOntologyID().getOntologyIRI() + "#";
96  prefixManager.setDefaultPrefix(defPrefix);
97  }
98  focusedObject = ontology.getOWLOntologyManager().getOWLDataFactory()
99  .getOWLThing();
100  }

163  @Override
164  public void visit(OWLOntology ontology1) {
165 
166  writePrefixes();
167  write("\n\n");
168  write(ONTOLOGY);
169  write("(");
170  if (!ontology1.isAnonymous()) {
171  writeFullIRI(ontology1.getOntologyID().getOntologyIRI());
172  if (ontology1.getOntologyID().getVersionIRI() != null) {
173  write("\n");
174  writeFullIRI(ontology1.getOntologyID().getVersionIRI());
175  }
176  write("\n");
177  }
178  for (OWLImportsDeclaration decl : ontology1.getImportsDeclarations()) {
179  write(IMPORT);
180  write("(");
181  writeFullIRI(decl.getIRI());
182  write(")\n");
183  }
184  for (OWLAnnotation ontologyAnnotation : ontology1.getAnnotations()) {
185  ontologyAnnotation.accept(this);
186  write("\n");
187  }
188  write("\n");
189  Set<OWLAxiom> writtenAxioms = new HashSet<OWLAxiom>();
190  List<OWLEntity> signature = new ArrayList<OWLEntity>(ontology1.getSignature());
191  Collections.sort(signature);
192  for (OWLEntity ent : signature) {
193  writeDeclarations(ent, writtenAxioms);
194  }
195  for (OWLEntity ent : signature) {
196  writeAxioms(ent, writtenAxioms);
197  }
198  List<OWLAxiom> remainingAxioms = new ArrayList<OWLAxiom>(ontology1.getAxioms());
199  remainingAxioms.removeAll(writtenAxioms);
200  for (OWLAxiom ax : remainingAxioms) {
201  ax.accept(this);
202  write("\n");
203  }
204  write(")");
205  // write("\n// ");
206  // write(VersionInfo.getVersionInfo().getGeneratedByMessage());
207  }

219  private Set<OWLAxiom>
220  writeAxioms(OWLEntity entity, Set<OWLAxiom> alreadyWrittenAxioms) {
221  Set<OWLAxiom> writtenAxioms = new HashSet<OWLAxiom>();
222  setFocusedObject(entity);
223  writtenAxioms.addAll(writeDeclarations(entity, alreadyWrittenAxioms));
224  writtenAxioms.addAll(writeAnnotations(entity));
225  List<OWLAxiom> axs = new ArrayList<OWLAxiom>();
226  axs.addAll(entity.accept(new OWLEntityVisitorEx<Set<? extends OWLAxiom>>() {
227  @Override
228  public Set<? extends OWLAxiom> visit(OWLClass cls) {
229  return ontology.getAxioms(cls);
230  }

232  @Override
233  public Set<? extends OWLAxiom> visit(OWLObjectProperty property) {
234  return ontology.getAxioms(property);
235  }
236 
237  @Override
238  public Set<? extends OWLAxiom> visit(OWLDataProperty property) {
239  return ontology.getAxioms(property);
240  }
241 
242  @Override
243  public Set<? extends OWLAxiom> visit(OWLNamedIndividual individual) {
244  return ontology.getAxioms(individual);
245  }
246 
247  @Override
248  public Set<? extends OWLAxiom> visit(OWLDatatype datatype) {
249  return ontology.getAxioms(datatype);
250  }
251 
252  @Override
253  public Set<? extends OWLAxiom> visit(OWLAnnotationProperty property) {
254  return ontology.getAxioms(property);
255  }
256  }));
257  Collections.sort(axs);
258  for (OWLAxiom ax : axs) {
259  if (alreadyWrittenAxioms.contains(ax)) {
260  continue;
261  }
262  if (ax.getAxiomType().equals(AxiomType.DIFFERENT_INDIVIDUALS)) {
263  continue;
264  }
265  if (ax.getAxiomType().equals(AxiomType.DISJOINT_CLASSES)
266  && ((OWLDisjointClassesAxiom) ax).getClassExpressions().size() > 2) {
267  continue;
268  }
269  ax.accept(this);
270  writtenAxioms.add(ax);
271  write("\n");
272  }
273  alreadyWrittenAxioms.addAll(writtenAxioms);
274  return writtenAxioms;
275  }

327  private void write(Collection<? extends OWLObject> objects) {
328  if (objects.size() > 2) {
329  for (Iterator<? extends OWLObject> it = objects.iterator(); it.hasNext();) {
330  it.next().accept(this);
331  if (it.hasNext()) {
332  write(" ");
333  }
334  }
335  } else if (objects.size() == 2) {
336  Iterator<? extends OWLObject> it = objects.iterator();
337  OWLObject objA = it.next();
338  OWLObject objB = it.next();
339  OWLObject lhs, rhs;
340  if (objA.equals(focusedObject)) {
341  lhs = objA;
342  rhs = objB;
343  } else {
344  lhs = objB;
345  rhs = objA;
346  }
347  lhs.accept(this);
348  writeSpace();
349  rhs.accept(this);
350  } else if (objects.size() == 1) {
351  objects.iterator().next().accept(this);
352  }
353  }
