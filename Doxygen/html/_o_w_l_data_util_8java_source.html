123  public static <N extends Number> OWLDatatype getDatatype(OWLDataFactory dataFactory, N n) {
124  XSDVocabulary v = numberTypeMap.get(n.getClass());
125  if (v == null) {
126  throw new OWLRuntimeException("Don't know how to translate " + n.getClass());
127  }
128  return dataFactory.getOWLDatatype(v.getIRI());
129  }
