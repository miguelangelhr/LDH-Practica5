101  public OWLXMLWriter(Writer writer, OWLOntology ontology) {
102  XMLWriterNamespaceManager nsm = new XMLWriterNamespaceManager(Namespaces.OWL.toString());
103  nsm.setPrefix("xsd", Namespaces.XSD.toString());
104  nsm.setPrefix("rdf", Namespaces.RDF.toString());
105  nsm.setPrefix("rdfs", Namespaces.RDFS.toString());
106  nsm.setPrefix("xml", Namespaces.XML.toString());
107  String base = Namespaces.OWL.toString();
108  if (ontology != null && !ontology.isAnonymous()) {
109  base = ontology.getOntologyID().getOntologyIRI().toString();
110  }
111  this.writer = XMLWriterFactory.getInstance().createXMLWriter(writer, nsm, base);
112  }

127  public void writePrefix(String prefixName, String iri) throws IOException {
128  writer.writeStartElement(OWLXMLVocabulary.PREFIX.getIRI().toString());
129  if (prefixName.endsWith(":")) {
130  String attName = prefixName.substring(0, prefixName.length() - 1);
131  writer.writeAttribute(OWLXMLVocabulary.NAME_ATTRIBUTE.getIRI().toString(), attName);
132  }
133  else {
134  writer.writeAttribute(OWLXMLVocabulary.NAME_ATTRIBUTE.getIRI().toString(), prefixName);
135  }
136  writer.writeAttribute(OWLXMLVocabulary.IRI_ATTRIBUTE.getIRI().toString(), iri);
137  writer.writeEndElement();
138  iriPrefixMap.put(iri, prefixName);
139  }

147  public String getIRIString(IRI iri) {
148  String prefixName = iriPrefixMap.get(iri.getNamespace());
149  if (prefixName == null) {
150  return iri.toString();
151  }
152  if(iri.getFragment()==null){
153  return prefixName;
154  }
155  return prefixName + iri.getFragment();
156  }

228  public void writeIRIAttribute(IRI iri) {
229  try {
230  String attName = OWLXMLVocabulary.IRI_ATTRIBUTE.getIRI().toString();
231  String value = iri.toString();
232  if (value.startsWith(writer.getXMLBase())) {
233  writer.writeAttribute(attName, value.substring(writer.getXMLBase().length(), value.length()));
234  }
235  else {
236  String val = getIRIString(iri);
237  if (!val.equals(iri.toString())) {
238  writer.writeAttribute(OWLXMLVocabulary.ABBREVIATED_IRI_ATTRIBUTE.getIRI().toString(), val);
239  }
240  else {
241  writer.writeAttribute(attName, val);
242  }
243  }
244  }
245  catch (IOException e) {
246  throw new OWLRuntimeException(e);
247  }
248  }

257  public void writeIRIElement(IRI iri) {
258  try {
259  String iriString = iri.toString();
260  if (iriString.startsWith(writer.getXMLBase())) {
261  writeStartElement(OWLXMLVocabulary.IRI_ELEMENT);
262  writeTextContent(iriString.substring(writer.getXMLBase().length(), iriString.length()));
263  writeEndElement();
264  }
265  else {
266  String val = getIRIString(iri);
267  if (!val.equals(iriString)) {
268  writeStartElement(OWLXMLVocabulary.ABBREVIATED_IRI_ELEMENT);
269  writer.writeTextContent(val);
270  writeEndElement();
271  }
272  else {
273  writeStartElement(OWLXMLVocabulary.IRI_ELEMENT);
274  writer.writeTextContent(val);
275  writeEndElement();
276  }
277  }
278  }
279  catch (IOException e) {
280  throw new OWLRuntimeException(e);
281  }
