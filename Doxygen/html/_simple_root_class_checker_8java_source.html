93  // Rules for determining if a class is a direct subclass of Thing
94 
95  // 1) It isn't referenced by ANY subclass axiom or equivalent class axioms
96  // 2) It is reference only by subclass axioms, but doesn't appear on the LHS of these axioms
97  // 3) It is on the LHS of a subclass axiom where the RHS is Thing
98  // 4) It is referenced only by equivalent class axioms, where all other operands in these axioms are named
99  // 5) It is not referenced by subclass axioms and is not referenced by any equivalent class axiom where there is
100  // at least one operand in the equivalent class axiom which is an intersection containing a named operand i.e.
101  // EquivalentClasses(A (B and hasP some C)) would not be a subclass of Thing
102 
103  private final RootClassCheckerHelper checker = new RootClassCheckerHelper();
104 
105  protected final NamedSuperChecker superChecker = new NamedSuperChecker();
106 
107 
108 
109  @Override
110  public boolean isRootClass(OWLClass cls) {
111 
112  for (OWLOntology ont : ontologies) {
113  for (OWLAxiom ax : ont.getReferencingAxioms(cls)) {
114  checker.setOWLClass(cls);
115  ax.accept(checker);
116  if (!checker.isRoot()) {
117  return false;
118  }
119  }
120  }
121  return true;
122  }

156  private class RootClassCheckerHelper extends OWLAxiomVisitorAdapter {
157 
158  private boolean isRoot;
159 
160  private OWLClass cls;
161 
162  public RootClassCheckerHelper() {}
163  public void setOWLClass(OWLClass cls) {
164  // Start off with the assumption that the class is
165  // a root class. This means if the class isn't referenced
166  // by any equivalent class axioms or subclass axioms then
167  // we correctly identify it as a root
168  isRoot = true;
169  this.cls = cls;
170  }
171 
172 
173  public boolean isRoot() {
174  return isRoot;
175  }
176 
177 
178  @Override
179  public void visit(OWLSubClassOfAxiom axiom) {
180  if (axiom.getSubClass().equals(cls)) {
181  superChecker.reset();
182  axiom.getSuperClass().accept(superChecker);
183  isRoot = !superChecker.namedSuper;
184  }
185  }
186 
187 
188  @Override
189  public void visit(OWLEquivalentClassesAxiom axiom) {
190  Set<OWLClassExpression> descs = axiom.getClassExpressions();
191  if (!descs.contains(cls)) {
192  return;
193  }
194  for (OWLClassExpression desc : descs) {
195  if (desc.equals(cls)) {
196  continue;
197  }
198  superChecker.reset();
199  desc.accept(superChecker);
200  if (superChecker.namedSuper) {
201  isRoot = false;
202  return;
203  }
204  }
205  isRoot = !superChecker.namedSuper;
206  }
207  }
