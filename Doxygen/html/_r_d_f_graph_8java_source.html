137  public boolean isAnonymousNodeSharedSubject(RDFResourceNode node) {
138  if(!node.isAnonymous()) {
139  return false;
140  }
141  int count = 0;
142  for(RDFTriple triple : triples) {
143  if(!triple.getObject().isLiteral()) {
144  RDFResourceNode object = (RDFResourceNode) triple.getObject();
145  if(object.equals(node)) {
146  count++;
147  if(count > 1) {
148  return true;
149  }
150  }
151  }
152  }
153  return false;
154  }

167  private void rebuildAnonRoots() {
168  rootAnonymousNodes = new HashSet<RDFResourceNode>();
169  for (RDFTriple triple : triples) {
170  rootAnonymousNodes.add(triple.getSubject());
171  }
172  for (RDFTriple triple : triples) {
173  if (!triple.getObject().isLiteral()) {
174  rootAnonymousNodes.remove(triple.getObject());
175  }
176  if (!triple.getSubject().isAnonymous()) {
177  rootAnonymousNodes.remove(triple.getSubject());
178  }
179  }
180  }
