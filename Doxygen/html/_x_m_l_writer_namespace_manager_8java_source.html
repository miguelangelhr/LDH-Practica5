106  public String getQName(String name) {
107  if (name.startsWith(defaultNamespace)) {
108  return name.substring(defaultNamespace.length(), name.length());
109  }
110  for (String ns : namespacePrefixMap.keySet()) {
111  if (name.startsWith(ns)) {
112  String localName = name.substring(ns.length(), name.length());
113  return namespacePrefixMap.get(ns) + ":" + localName;
114  }
115  }
116  return name;
117  }

120  public void createPrefixForNamespace(String namespace) {
121  if (namespace.equals(defaultNamespace)) {
122  return;
123  }
124  if (wellknownNamespaces.containsKey(namespace)) {
125  setPrefix(wellknownNamespaces.get(namespace), namespace);
126  }
127  if (!namespacePrefixMap.containsKey(namespace)) {
128  int counter = 1;
129  while (prefixNamespaceMap.get("p" + counter) != null) {
130  counter++;
131  }
132  setPrefix("p" + counter, namespace);
133  }
134  }
