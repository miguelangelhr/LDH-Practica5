150  private String getDefaultOntologyTagValue(IRI baseIRI) {
151  URI baseURI = baseIRI.toURI();
152  String baseURIPath = baseURI.getPath();
153  if(baseURIPath == null) {
154  return baseIRI.toString();
155  }
156  int lastSlashIndex = baseURIPath.lastIndexOf("/");
157  if(lastSlashIndex == -1 || lastSlashIndex + 1 == baseURIPath.length()) {
158  return baseURIPath;
159  }
160  return baseURIPath.substring(lastSlashIndex + 1);
161  }

402  @Override
403  public void handleTagValue(String tag, String value, String qualifierBlock, String comment) {
404  try {
405  TagValueHandler handler = handlerMap.get(tag);
406  if (handler != null) {
407  handler.handle(currentId, value, qualifierBlock, comment);
408  }
409  else if (inHeader) {
410  if (tag.equals(IMPORT_TAG_NAME)) {
411  IRI uri = IRI.create(value.trim());
412  OWLImportsDeclaration decl = owlOntologyManager.getOWLDataFactory().getOWLImportsDeclaration(uri);
413  owlOntologyManager.makeLoadImportRequest(decl, configuration);
414  owlOntologyManager.applyChange(new AddImport(ontology, decl));
415  }
416  else {
417  // Ontology annotations
418  OWLLiteral con = getDataFactory().getOWLLiteral(unescapeTagValue(value));
419  OWLAnnotationProperty property = getDataFactory().getOWLAnnotationProperty(getIRIFromTagName(tag));
420  OWLAnnotation anno = getDataFactory().getOWLAnnotation(property, con);
421  owlOntologyManager.applyChange(new AddOntologyAnnotation(ontology, anno));
422  }
423  }
424  else if (currentId != null) {
425  // Add as annotation
426  if (configuration.isLoadAnnotationAxioms()) {
427  IRI subject = getIRI(currentId);
428  OWLLiteral con = getDataFactory().getOWLLiteral(unescapeTagValue(value));
429  IRI annotationPropertyIRI = getIRIFromTagName(tag);
430  OWLAnnotationProperty property = getDataFactory().getOWLAnnotationProperty(annotationPropertyIRI);
431  OWLAnnotation anno = getDataFactory().getOWLAnnotation(property, con);
432  OWLAnnotationAssertionAxiom ax = getDataFactory().getOWLAnnotationAssertionAxiom(subject, anno);
433  owlOntologyManager.addAxiom(ontology, ax);
434  OWLDeclarationAxiom annotationPropertyDeclaration = getDataFactory().getOWLDeclarationAxiom(property);
435  owlOntologyManager.addAxiom(ontology, annotationPropertyDeclaration);
436  }
437  }
438 
439  }
440  catch (UnloadableImportException e) {
441  logger.severe(e.getMessage());
442  }
443  }

445  public String unescapeTagValue(String value) {
446  String unquoted;
447  if (value.startsWith("\"") && value.endsWith("\"")) {
448  unquoted = value.substring(1, value.length() - 1);
449  }
450  else {
451  unquoted = value;
452  }
453  StringBuilder sb = new StringBuilder();
454  for (int i = 0; i < unquoted.length(); i++) {
455  char ch = unquoted.charAt(i);
456  if (ch != '\\') {
457  sb.append(ch);
458  }
459  }
460  return sb.toString();
461  }

539  private IRI getIRI(String s) {
540  String trimmed = s.trim();
541  IRI iri = uriCache.get(trimmed);
542  if (iri != null) {
543  return iri;
544  }
545  OWLOntologyID ontologyID = getOntology().getOntologyID();
546  OBOIdType type = OBOIdType.getIdType(trimmed);
547  if (type == null) {
548  throw new OWLRuntimeException("Not a valid OBO ID: " + s);
549  }
550  IRI freshIRI = type.getIRIFromOBOId(ontologyID, idSpaceManager, trimmed);
551  uriCache.put(trimmed, freshIRI);
552  return freshIRI;
553  }

562  public OWLAnnotation parseXRef(String xref) {
563  Matcher matcher = XREF_PATTERN.matcher(xref);
564  if (matcher.matches()) {
565  OWLDataFactory df = getDataFactory();
566  String xrefQuotedString = matcher.group(XREF_QUOTED_STRING_GROUP);
567  // the quoted string is a description of the xref. I can't find anywhere to put this.
568  // Just add as a comment for now
569  Set<OWLAnnotation> xrefDescriptions;
570  if (xrefQuotedString != null) {
571  xrefDescriptions = new HashSet<OWLAnnotation>();
572  xrefDescriptions.add(df.getOWLAnnotation(df.getRDFSComment(), df.getOWLLiteral(xrefQuotedString)));
573  }
574  else {
575  xrefDescriptions = Collections.emptySet();
576  }
577  String xrefId = matcher.group(XREF_ID_GROUP).trim();
578  OBOIdType idType = OBOIdType.getIdType(xrefId);
579  OWLAnnotationValue annotationValue;
580  if (idType != null) {
581  annotationValue = getIRIFromOBOId(xrefId);
582  }
583  else {
584  annotationValue = getDataFactory().getOWLLiteral(xrefId);
585  }
586  OWLAnnotationProperty xrefProperty = df.getOWLAnnotationProperty(OBOVocabulary.XREF.getIRI());
587  return df.getOWLAnnotation(xrefProperty, annotationValue, xrefDescriptions);
588  }
589  else {
590  OWLDataFactory df = getDataFactory();
591  OWLAnnotationProperty xrefProperty = df.getOWLAnnotationProperty(OBOVocabulary.XREF.getIRI());
592  return df.getOWLAnnotation(xrefProperty, df.getOWLLiteral(xref));
593  }
