133  void handleChanges(List<? extends OWLOntologyChange> changes) {
134  if (ontologies == null) {
135  return;
136  }
137  Set<OWLEntity> processed = new HashSet<OWLEntity>();
138  for (OWLOntologyChange chg : changes) {
139  if (ontologies.contains(chg.getOntology())) {
140  if (chg.isAddAxiom()) {
141  AddAxiom addAx = (AddAxiom) chg;
142  for (OWLEntity ent : addAx.getSignature()) {
143  if (!processed.contains(ent)) {
144  processed.add(ent);
145  add(ent);
146  }
147  }
148  } else if (chg.isRemoveAxiom()) {
149  RemoveAxiom remAx = (RemoveAxiom) chg;
150  for (OWLEntity ent : remAx.getSignature()) {
151  if (!processed.contains(ent)) {
152  processed.add(ent);
153  boolean stillRef = false;
154  for (OWLOntology ont : ontologies) {
155  if (ont.containsEntityInSignature(ent)) {
156  stillRef = true;
157  break;
158  }
159  }
160  if (!stillRef) {
161  remove(ent);
162  }
163  }
164  }
165  }
166  }
167  }
168  }
