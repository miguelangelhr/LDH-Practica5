180  public String getPrefix(String namespace) {
181  String prefix = namespace2PrefixMap.get(namespace);
182  if (prefix != null) {
183  return prefix;
184  }
185  // We need to generate a candidate prefix
186  prefix = generatePrefix(namespace);
187  if (prefix == null) {
188  // For some reason, we couldn't generate a decent prefix
189  // Compute an auto generated prefix
190  do {
191  prefix = "p" + candidateIndex++;
192  } while (namespace2PrefixMap.containsValue(prefix));
193  }
194  namespace2PrefixMap.put(namespace, prefix);
195  return prefix;
196  }

213  public String generatePrefix(String namespace) {
214 
215  String prefix = standardNamespacePrefixMappings.get(namespace);
216  if (prefix != null) {
217  namespace2PrefixMap.put(namespace, prefix);
218  return prefix;
219  }
220 
221  int startIndex = -1;
222  for (int i = namespace.length() - 1; i > -1; i--) {
223  char curChar = namespace.charAt(i);
224  boolean isStartChar = XMLUtils.isNCNameStartChar(curChar);
225  if (isStartChar || startIndex == -1) {
226  if (isStartChar) {
227  startIndex = i;
228  }
229  }
230  else if (!XMLUtils.isNCNameChar(curChar)) {
231  break;
232  }
233  }
234  if (startIndex == -1) {
235  return null;
236  }
237  int endIndex = startIndex + 1;
238  for (int i = startIndex; endIndex < namespace.length() && i < namespace.length(); i++) {
239  char curChar = namespace.charAt(endIndex);
240  // We include any NCNameChar except a full stop (.) so
241  // that if the URI looks like a file with an extension the
242  // extension is removed.
243  if (XMLUtils.isNCNameChar(curChar) && curChar != '.') {
244  endIndex = i + 1;
245  }
246  else {
247  break;
248  }
249  }
250  String computedPrefix = namespace.substring(startIndex, endIndex);
251  String candidatePrefix = computedPrefix;
252  int index = 2;
253  while (namespace2PrefixMap.containsValue(candidatePrefix) ||
254  standardNamespacePrefixMappings.containsValue(candidatePrefix)) {
255  candidatePrefix = computedPrefix + index;
256  index++;
257  }
258  return candidatePrefix;
259  }
