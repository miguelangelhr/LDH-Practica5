486  public static class ThreadSafeConditionalCopySet<T> implements Set<T> {
487  private volatile boolean copyDone = false;
488  private Collection<T> delegate;
489  private final ReadWriteLock lock = new ReentrantReadWriteLock();
490  private final Lock readLock = lock.readLock();
491  private final Lock writeLock = lock.writeLock();
492  private final static int maxContains = 10;
493  private final AtomicInteger containsCounter = new AtomicInteger(0);
494 
497  public ThreadSafeConditionalCopySet(Collection<T> source) {
498  this.delegate = new ArrayList<T>(source);
499  }

503  try {
504  readLock.lock();

506  return false;
507  }
508  if (this == obj) {
509  return true;

575  @Override
576  public boolean contains(Object arg0) {
577  /*
578  * note: the check on the number of contains and on the copyDone
579  * flag is intentionally not made inside a lock; these operations
580  * will not stop other threads from accessing the collection, and
581  * the lack of synchronization means that a few contains/containsAll
582  * calls might not be recorded; the result is that the copy
583  * optimization would be delayed. This is not an issue, since
584  * maxContains is only a rough estimate.
585  */
586  if (!copyDone) {
587  if (containsCounter.incrementAndGet() >= maxContains && !copyDone) {
588  try {
589  writeLock.lock();
590  // many calls to contains, inefficient if the delegate
591  // is not a set
592  // copyDone is doublechecked, but here it's protected by
593  // the write
594  // lock as in all other instances in which its value is
595  // changed
596  if (!copyDone && !(delegate instanceof Set)) {
597  copyDone = true;
598  delegate = new SyncSet<T>(delegate);
599  }
600  // skip the second portion of the method: no need to
601  // reacquire
602  // the lock, it's already a write lock
603  return delegate.contains(arg0);
604  } finally {
605  writeLock.unlock();
606  }
607  }
608  }
609  try {
610  readLock.lock();
611  return delegate.contains(arg0);
612  } finally {
613  readLock.unlock();
614  }
615  }
