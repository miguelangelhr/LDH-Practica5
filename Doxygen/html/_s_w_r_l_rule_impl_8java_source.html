119  @Override
120  public boolean containsAnonymousClassExpressions() {
121  if (containsAnonymousClassExpressions == null) {
122  for (SWRLAtom atom : head) {
123  if (atom instanceof SWRLClassAtom) {
124  if (((SWRLClassAtom) atom).getPredicate().isAnonymous()) {
125  containsAnonymousClassExpressions = Boolean.TRUE;
126  break;
127  }
128  }
129  }
130  if (containsAnonymousClassExpressions == null) {
131  for (SWRLAtom atom : body) {
132  if (atom instanceof SWRLClassAtom) {
133  if (((SWRLClassAtom) atom).getPredicate().isAnonymous()) {
134  containsAnonymousClassExpressions = Boolean.TRUE;
135  break;
136  }
137  }
138  }
139  }
140  if (containsAnonymousClassExpressions == null) {
141  containsAnonymousClassExpressions = Boolean.FALSE;
142  }
143  }
144  return containsAnonymousClassExpressions.booleanValue();
145  }
